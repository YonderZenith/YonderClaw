@echo off
setlocal enabledelayedexpansion
title YonderClaw: __AGENT_NAME__

REM Ensure PATH includes Node.js and Claude
set PATH=C:\Program Files\nodejs;%USERPROFILE%\AppData\Roaming\npm;%USERPROFILE%\.local\bin;%PATH%

echo.
echo   ==========================================
echo   YonderClaw: __AGENT_NAME__
echo   ==========================================
echo.

REM Set working directory
cd /d "__PROJECT_DIR__"

REM v3.7.0 uses data\session-id.txt as the resume target (created by installer).
REM Pre-v1.1 also wrote that file with a different format — if it's empty, ignore it below.

REM Find Claude executable. Prefer real .exe binaries over the .cmd shim so the
REM child process is claude itself (no extra cmd.exe in the tree). Order matches
REM desktop/src-tauri/src/claude.rs::find_claude — keep them in sync.
set "CLAUDE_CMD="
if exist "%USERPROFILE%\.local\bin\claude.exe" (
    set "CLAUDE_CMD=%USERPROFILE%\.local\bin\claude.exe"
) else if exist "%USERPROFILE%\AppData\Roaming\npm\node_modules\@anthropic-ai\claude-code\bin\claude.exe" (
    set "CLAUDE_CMD=%USERPROFILE%\AppData\Roaming\npm\node_modules\@anthropic-ai\claude-code\bin\claude.exe"
) else if exist "%USERPROFILE%\AppData\Roaming\npm\claude.cmd" (
    set "CLAUDE_CMD=%USERPROFILE%\AppData\Roaming\npm\claude.cmd"
) else (
    where claude >nul 2>&1
    if !ERRORLEVEL! equ 0 set "CLAUDE_CMD=claude"
)

if "!CLAUDE_CMD!"=="" (
    echo   [ERROR] Claude Code not found.
    echo   Install: https://claude.ai/download
    pause
    exit /b 1
)

REM Ensure authenticated
if not exist "%USERPROFILE%\.claude\.credentials.json" (
    echo   [!] Claude not logged in...
    "!CLAUDE_CMD!" auth login
    if not exist "%USERPROFILE%\.claude\.credentials.json" (
        echo   [ERROR] Authentication failed.
        pause
        exit /b 1
    )
)

echo   Directory: %CD%
echo.

REM === LAUNCH LOGIC (v3.7.0) ===
REM Priority:
REM   1. --resume <session-id-from-installer>  (deterministic; never loses the agent)
REM   2. --continue                            (most-recent session in this dir)
REM   3. fresh claude                          (first run, no history)

:launch

set "RESUME_ID="
if exist "data\session-id.txt" (
    set /p RESUME_ID=<data\session-id.txt
)
REM Validate UUID shape (8-4-4-4-12 hex, total 36 chars). Anything else (BOM, garbage,
REM partial write, legacy pre-v1.1 content) is rejected so we don't crash-loop on
REM `claude --resume <bad>`. Length-check first (rejects empty + over-long), then a small
REM findstr regex (findstr's heap can't handle the full {n}-style expansion).
if not "!RESUME_ID!"=="" (
    if "!RESUME_ID:~35,1!"=="" set "RESUME_ID="
)
if not "!RESUME_ID!"=="" (
    if not "!RESUME_ID:~36,1!"=="" set "RESUME_ID="
)
if not "!RESUME_ID!"=="" (
    echo !RESUME_ID!| findstr /R /C:"^[0-9A-Fa-f]*-[0-9A-Fa-f]*-[0-9A-Fa-f]*-[0-9A-Fa-f]*-[0-9A-Fa-f]*$" >nul
    if !ERRORLEVEL! neq 0 set "RESUME_ID="
)

REM Check if any session exists for this project (fallback path)
set "HAS_SESSION=0"
set "ENCODED_PATH=__ENCODED_PATH__"
set "SESSION_DIR=%USERPROFILE%\.claude\projects\!ENCODED_PATH!"
if exist "!SESSION_DIR!\*.jsonl" set "HAS_SESSION=1"

REM Auto-connect to QIS relay (first boot: create identity + deposit, subsequent: skip)
if not exist "data\qis-deposit-log.json" (
    echo   Connecting to YonderClaw intelligence network...
    npx tsx scripts\qis-autoconnect.ts
    if !ERRORLEVEL! neq 0 (
        echo   [!] Network connect had an issue — will retry next launch.
    ) else (
        echo   [OK] Connected to intelligence network.
    )
    echo.
)

if not "!RESUME_ID!"=="" (
    echo   Resuming __AGENT_NAME__ ^(session !RESUME_ID:~0,8!…^)...
    echo.
    "!CLAUDE_CMD!" --resume !RESUME_ID!__SKIP_PERMS_ARG__
) else if "!HAS_SESSION!"=="1" (
    echo   Resuming __AGENT_NAME__ ^(most recent session^)...
    echo.
    "!CLAUDE_CMD!" --continue__SKIP_PERMS_ARG__
) else (
    echo   First launch — initializing __AGENT_NAME__...
    echo   Claude will read CLAUDE.md and SOUL.md for context.
    echo.
    "!CLAUDE_CMD!"__SKIP_PERMS_ARG__
)

echo.
echo   __AGENT_NAME__ session ended.
echo   Press any key to relaunch (Ctrl+C to quit)...
pause >nul
goto :launch
