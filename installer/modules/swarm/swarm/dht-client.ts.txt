/**
 * YonderClaw Swarm — DHT Client (with Holder Mode)
 *
 * Decentralized bucket routing via Hyperswarm (Kademlia DHT).
 * Every bucket = a DHT topic (SHA-256 of bucket path).
 *
 * EDGE mode (tier 1-2): queries DHT peers, deposits via relay + DHT push
 * HOLDER mode (tier 3): stores packets in local SQLite, serves to other agents,
 *   announces on bucket topics, persists across restarts
 *
 * Fallback: HTTP relay always works regardless of DHT state
 */

import Hyperswarm from "hyperswarm";
import crypto from "crypto";
import fs from "fs";
import path from "path";

const RELAY_URL = "__RELAY_URL__";
const DATA_DIR = path.join(process.cwd(), "data");
const KEYPAIR_PATH = path.join(DATA_DIR, "dht-keypair.json");
const TOPICS_PATH = path.join(DATA_DIR, "dht-topics.json");
const DB_PATH = path.join(DATA_DIR, "dht-local.db");
const MEMBERSHIP_TOPIC = crypto.createHash("sha256").update("yonderclaw.holders.v1").digest();

export interface DHTPacket {
  bucket: string;
  agent_id: string;
  signal: "positive" | "negative" | "neutral";
  confidence: number;
  insight: string;
  context: Record<string, any>;
  metrics: Record<string, any>;
  ts: number;
}

export class DHTClient {
  private swarm: Hyperswarm | null = null;
  private localPackets: Map<string, DHTPacket[]> = new Map();
  private peerPackets: Map<string, DHTPacket[]> = new Map();
  private connections: Map<string, Set<any>> = new Map();
  private agentId: string;
  private ready: boolean = false;
  private isHolder: boolean = false;
  private db: any = null; // better-sqlite3 instance (holder mode only)
  private peers: Map<string, any> = new Map(); // peerId → socket (for holder gossip)

  constructor(agentId: string) {
    this.agentId = agentId;
  }

  async start(holder: boolean = false): Promise<void> {
    this.isHolder = holder;
    fs.mkdirSync(DATA_DIR, { recursive: true });

    const keyPair = this.getOrCreateKeypair();
    this.swarm = new Hyperswarm({ keyPair });

    this.swarm.on("connection", (socket: any, info: any) => {
      this.handleConnection(socket, info);
    });

    this.swarm.on("error", (err: Error) => {
      console.error("[DHT] Error (non-fatal):", err.message);
    });

    // Holder mode: initialize SQLite for persistent packet storage
    if (this.isHolder) {
      try {
        const Database = (await import("better-sqlite3")).default;
        this.db = new Database(DB_PATH);
        this.db.pragma("journal_mode = WAL");
        this.db.pragma("busy_timeout = 5000");
        this.initHolderDb();
        this.loadPacketsFromDb();
        console.log(`[DHT] Holder mode: SQLite initialized at ${DB_PATH}`);
      } catch (err: any) {
        console.warn(`[DHT] SQLite unavailable, using memory storage: ${err.message}`);
        this.db = null;
      }

      // Join membership topic so other holders/edges can discover us
      this.swarm.join(MEMBERSHIP_TOPIC, { server: true, client: true });
    } else {
      // Edge mode: join membership to discover holders
      this.swarm.join(MEMBERSHIP_TOPIC, { server: false, client: true });
    }

    // Rejoin previously announced topics
    const savedTopics = this.loadTopics();
    for (const topicHex of savedTopics) {
      const topic = Buffer.from(topicHex, "hex");
      this.swarm.join(topic, { server: this.isHolder, client: true });
    }

    this.ready = true;
    const nodeHex = keyPair.publicKey.toString("hex").slice(0, 12);
    console.log(`[DHT] Online. Node: ${nodeHex}... Mode: ${holder ? "HOLDER" : "EDGE"}`);

    // Holders: initial sync from relay to catch up, then periodic sync
    if (this.isHolder) {
      setTimeout(() => this.holderSync().catch(() => {}), 10_000); // 10s after start
      this._syncTimer = setInterval(() => this.holderSync().catch(() => {}), 30 * 60 * 1000); // every 30 min
    }
  }

  private _syncTimer: ReturnType<typeof setInterval> | null = null;

  async stop(): Promise<void> {
    if (this._syncTimer) { clearInterval(this._syncTimer); this._syncTimer = null; }
    if (this.swarm) {
      await this.swarm.destroy();
      this.swarm = null;
    }
    if (this.db) {
      this.db.close();
      this.db = null;
    }
    this.ready = false;
  }

  /**
   * Holder sync: pull all buckets from relay, store any missing packets locally.
   * Ensures holders stay up-to-date even if they missed a DHT push.
   */
  private async holderSync(): Promise<void> {
    try {
      const res = await fetch(`${RELAY_URL}/buckets?limit=500`);
      if (!res.ok) return;
      const data = await res.json();
      const buckets = data.buckets || [];
      let synced = 0;

      for (const bucket of buckets) {
        try {
          const bRes = await fetch(`${RELAY_URL}/lookup?path=${encodeURIComponent(bucket.path)}`);
          if (!bRes.ok) continue;
          const bData = await bRes.json();
          const topicHex = this.hashBucket(bucket.path).toString("hex");

          for (const p of bData.packets || []) {
            const packet: DHTPacket = {
              bucket: bucket.path,
              agent_id: p.agent_id,
              signal: p.signal || "neutral",
              confidence: p.confidence ?? 0.5,
              insight: p.insight || "",
              context: typeof p.context === "string" ? JSON.parse(p.context) : (p.context || {}),
              metrics: typeof p.metrics === "string" ? JSON.parse(p.metrics) : (p.metrics || {}),
              ts: p.ts || 0,
            };

            // Only store if we don't have it or ours is older
            const existing = (this.localPackets.get(topicHex) || []).find(lp => lp.agent_id === p.agent_id);
            if (!existing || (packet.ts > existing.ts)) {
              this.storePacketLocal(topicHex, packet);
              synced++;
            }
          }

          // Join this bucket's topic so we serve it
          if (!this.swarm) continue;
          const topic = Buffer.from(topicHex, "hex");
          this.swarm.join(topic, { server: true, client: true });
          this.saveTopicList(topicHex);
        } catch {}
      }

      if (synced > 0) console.log(`[DHT] Holder sync: ${synced} new packets from relay`);
    } catch {}
  }

  // --- Core Operations ---

  /**
   * Join a bucket topic — announce that you have data for this problem
   */
  async joinBucket(bucketPath: string): Promise<string> {
    if (!this.swarm) throw new Error("DHT not started");
    const topic = this.hashBucket(bucketPath);
    const topicHex = topic.toString("hex");

    this.swarm.join(topic, { server: this.isHolder, client: true });
    this.saveTopicList(topicHex);

    return topicHex;
  }

  /**
   * Deposit a packet — stores locally + pushes to connected peers + relay fallback
   */
  async deposit(bucketPath: string, packet: DHTPacket): Promise<void> {
    const topicHex = this.hashBucket(bucketPath).toString("hex");

    // Store locally (memory + SQLite if holder)
    this.storePacketLocal(topicHex, packet);

    // Push to connected peers on this topic
    const sockets = this.connections.get(topicHex);
    if (sockets) {
      for (const socket of sockets) {
        this.send(socket, { type: "PACKET", data: packet });
      }
    }

    // Also push to any peers from membership topic
    for (const [peerId, socket] of this.peers) {
      this.send(socket, { type: "PACKET", data: packet });
    }

    // Push to relay (fire and forget)
    try {
      fetch(`${RELAY_URL}/packets`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          bucket: bucketPath,
          agent_id: packet.agent_id,
          outcome: {
            signal: packet.signal,
            confidence: packet.confidence,
            insight: packet.insight,
            context: packet.context,
            metrics: packet.metrics,
          },
        }),
      }).catch(() => {});
    } catch {}
  }

  /**
   * Query a bucket — pull from local + DHT peers + relay, deduplicate
   */
  async query(bucketPath: string, timeoutMs: number = 8000): Promise<DHTPacket[]> {
    const allPackets: Map<string, DHTPacket> = new Map();
    const topicHex = this.hashBucket(bucketPath).toString("hex");

    // 1. Local packets (instant)
    const local = this.localPackets.get(topicHex) || [];
    for (const p of local) allPackets.set(p.agent_id, p);

    // 2. Cached peer packets
    const cached = this.peerPackets.get(topicHex) || [];
    for (const p of cached) allPackets.set(p.agent_id, p);

    // 3. DHT peers (join topic, collect with timeout)
    if (this.swarm && this.ready) {
      try {
        const topic = Buffer.from(topicHex, "hex");
        this.swarm.join(topic, { server: false, client: true });

        await Promise.race([
          this.collectFromPeers(topicHex, allPackets, timeoutMs),
          new Promise(r => setTimeout(r, timeoutMs)),
        ]);
      } catch {}
    }

    // 4. Relay fallback (canonical copy)
    try {
      const res = await fetch(`${RELAY_URL}/lookup?path=${encodeURIComponent(bucketPath)}`);
      if (res.ok) {
        const data = await res.json();
        for (const p of data.packets || []) {
          if (!allPackets.has(p.agent_id)) {
            allPackets.set(p.agent_id, {
              bucket: bucketPath,
              agent_id: p.agent_id,
              signal: p.signal,
              confidence: p.confidence,
              insight: p.insight,
              context: typeof p.context === "string" ? JSON.parse(p.context) : (p.context || {}),
              metrics: typeof p.metrics === "string" ? JSON.parse(p.metrics) : (p.metrics || {}),
              ts: p.ts,
            });
          }
        }
      }
    } catch {}

    return Array.from(allPackets.values());
  }

  /**
   * Search buckets by keyword — relay only (DHT doesn't support text search)
   */
  async search(keyword: string): Promise<any[]> {
    try {
      const res = await fetch(`${RELAY_URL}/buckets?q=${encodeURIComponent(keyword)}`);
      if (res.ok) {
        const data = await res.json();
        return data.buckets || [];
      }
    } catch {}
    return [];
  }

  /**
   * Tally packets locally — the edge synthesis (where QIS happens)
   */
  static tally(packets: DHTPacket[]): {
    total: number;
    positive: number;
    negative: number;
    neutral: number;
    positive_pct: number;
    avg_confidence: number;
    top_insights: Array<{ insight: string; count: number; avg_confidence: number }>;
  } {
    const total = packets.length;
    const positive = packets.filter(p => p.signal === "positive").length;
    const negative = packets.filter(p => p.signal === "negative").length;
    const neutral = total - positive - negative;
    const avgConf = total > 0 ? packets.reduce((s, p) => s + p.confidence, 0) / total : 0;

    const insightGroups: Map<string, { count: number; totalConf: number; full: string }> = new Map();
    for (const p of packets) {
      const key = p.insight.toLowerCase().slice(0, 100).trim();
      const group = insightGroups.get(key) || { count: 0, totalConf: 0, full: p.insight };
      group.count++;
      group.totalConf += p.confidence;
      insightGroups.set(key, group);
    }

    const topInsights = Array.from(insightGroups.values())
      .map(g => ({ insight: g.full, count: g.count, avg_confidence: Math.round((g.totalConf / g.count) * 100) / 100 }))
      .sort((a, b) => b.count * b.avg_confidence - a.count * a.avg_confidence)
      .slice(0, 10);

    return {
      total, positive, negative, neutral,
      positive_pct: total > 0 ? Math.round((positive / total) * 100) : 0,
      avg_confidence: Math.round(avgConf * 100) / 100,
      top_insights: topInsights,
    };
  }

  // --- Local Storage ---

  private storePacketLocal(topicHex: string, packet: DHTPacket): void {
    // Memory store (always)
    if (!this.localPackets.has(topicHex)) this.localPackets.set(topicHex, []);
    const existing = this.localPackets.get(topicHex)!;
    const idx = existing.findIndex(p => p.agent_id === packet.agent_id);
    if (idx >= 0) existing[idx] = packet;
    else existing.push(packet);

    // SQLite store (holder mode)
    if (this.db) {
      try {
        const bucketId = topicHex.slice(0, 16);
        const domain = packet.bucket.split(".")[0] || "unknown";

        this.db.prepare(`
          INSERT OR IGNORE INTO buckets (bucket_id, path, domain) VALUES (?, ?, ?)
        `).run(bucketId, packet.bucket, domain);

        this.db.prepare(`
          INSERT INTO packets (bucket_id, agent_id, signal, confidence, insight, context, metrics, ts)
          VALUES (?, ?, ?, ?, ?, ?, ?, ?)
          ON CONFLICT(bucket_id, agent_id) DO UPDATE SET
            signal=excluded.signal, confidence=excluded.confidence, insight=excluded.insight,
            context=excluded.context, metrics=excluded.metrics, ts=excluded.ts
        `).run(
          bucketId, packet.agent_id,
          packet.signal, packet.confidence, packet.insight,
          JSON.stringify(packet.context || {}), JSON.stringify(packet.metrics || {}),
          packet.ts || Math.floor(Date.now() / 1000)
        );
      } catch {}
    }
  }

  private initHolderDb(): void {
    if (!this.db) return;
    this.db.exec(`
      CREATE TABLE IF NOT EXISTS buckets (
        bucket_id TEXT PRIMARY KEY,
        path TEXT NOT NULL,
        domain TEXT NOT NULL,
        packet_count INTEGER DEFAULT 0,
        last_updated TEXT DEFAULT (datetime('now'))
      );
      CREATE TABLE IF NOT EXISTS packets (
        bucket_id TEXT NOT NULL,
        agent_id TEXT NOT NULL,
        signal TEXT NOT NULL,
        confidence REAL NOT NULL,
        insight TEXT NOT NULL,
        context TEXT DEFAULT '{}',
        metrics TEXT DEFAULT '{}',
        ts INTEGER NOT NULL,
        UNIQUE(bucket_id, agent_id)
      );
      CREATE INDEX IF NOT EXISTS idx_packets_bucket ON packets(bucket_id);
    `);
  }

  private loadPacketsFromDb(): void {
    if (!this.db) return;
    try {
      const rows = this.db.prepare(`
        SELECT p.*, b.path FROM packets p JOIN buckets b ON p.bucket_id = b.bucket_id
      `).all() as any[];
      for (const r of rows) {
        const topicHex = this.hashBucket(r.path).toString("hex");
        const packet: DHTPacket = {
          bucket: r.path,
          agent_id: r.agent_id,
          signal: r.signal,
          confidence: r.confidence,
          insight: r.insight,
          context: typeof r.context === "string" ? JSON.parse(r.context) : (r.context || {}),
          metrics: typeof r.metrics === "string" ? JSON.parse(r.metrics) : (r.metrics || {}),
          ts: r.ts,
        };
        if (!this.localPackets.has(topicHex)) this.localPackets.set(topicHex, []);
        const existing = this.localPackets.get(topicHex)!;
        const idx = existing.findIndex(p => p.agent_id === packet.agent_id);
        if (idx >= 0) existing[idx] = packet;
        else existing.push(packet);
      }
      console.log(`[DHT] Loaded ${rows.length} packets from local storage`);
    } catch {}
  }

  // --- Connection Handling ---

  private handleConnection(socket: any, info: any) {
    const peerId = info.publicKey?.toString("hex").slice(0, 12) || "unknown";
    this.peers.set(peerId, socket);

    let buffer = "";
    socket.on("data", (chunk: Buffer) => {
      buffer += chunk.toString();
      const lines = buffer.split("\n");
      buffer = lines.pop() || "";
      for (const line of lines) {
        if (line.trim()) {
          try { this.handleMessage(socket, info, JSON.parse(line)); } catch {}
        }
      }
    });

    socket.on("error", () => {});
    socket.on("close", () => {
      this.peers.delete(peerId);
      for (const [topic, sockets] of this.connections) {
        sockets.delete(socket);
      }
    });

    // Track connection by topics
    for (const topic of info.topics || []) {
      const hex = topic.toString("hex");
      if (!this.connections.has(hex)) this.connections.set(hex, new Set());
      this.connections.get(hex)!.add(socket);
    }

    // If holder: send our packets for shared topics
    if (this.isHolder) {
      for (const topic of info.topics || []) {
        const hex = topic.toString("hex");
        const myPackets = this.localPackets.get(hex) || [];
        for (const p of myPackets) {
          this.send(socket, { type: "PACKET", data: p });
        }
      }
    }

    // Request their packets
    this.send(socket, { type: "GET_PACKETS" });

    // Announce as holder
    if (this.isHolder) {
      this.send(socket, { type: "HOLDER_HELLO", agent_id: this.agentId, topics: this.loadTopics().length });
    }
  }

  private handleMessage(socket: any, info: any, msg: any) {
    switch (msg.type) {
      case "PACKET":
        if (msg.data && msg.data.agent_id) {
          const bucket = msg.data.bucket || "";
          const topicHex = this.hashBucket(bucket).toString("hex");
          // Store as peer packet (memory)
          if (!this.peerPackets.has(topicHex)) this.peerPackets.set(topicHex, []);
          const packets = this.peerPackets.get(topicHex)!;
          const idx = packets.findIndex(p => p.agent_id === msg.data.agent_id);
          if (idx >= 0) packets[idx] = msg.data;
          else packets.push(msg.data);

          // If holder, persist to SQLite too
          if (this.isHolder) {
            this.storePacketLocal(topicHex, msg.data);
          }
        }
        break;

      case "GET_PACKETS":
        // Send all local packets for shared topics
        for (const topic of info.topics || []) {
          const hex = topic.toString("hex");
          const myPackets = this.localPackets.get(hex) || [];
          for (const p of myPackets) {
            this.send(socket, { type: "PACKET", data: p });
          }
        }
        break;

      case "HOLDER_HELLO":
        // Another holder announced itself — noted
        break;

      case "SYNC_REQUEST":
        // Peer wants all packets for a bucket
        if (this.isHolder && msg.bucket) {
          const topicHex = this.hashBucket(msg.bucket).toString("hex");
          const packets = this.localPackets.get(topicHex) || [];
          for (const p of packets) {
            this.send(socket, { type: "PACKET", data: p });
          }
          this.send(socket, { type: "SYNC_DONE", bucket: msg.bucket, count: packets.length });
        }
        break;
    }
  }

  private async collectFromPeers(topicHex: string, allPackets: Map<string, DHTPacket>, timeoutMs: number): Promise<void> {
    const start = Date.now();
    while (Date.now() - start < timeoutMs) {
      const cached = this.peerPackets.get(topicHex) || [];
      for (const p of cached) allPackets.set(p.agent_id, p);
      await new Promise(r => setTimeout(r, 500));
    }
  }

  private send(socket: any, msg: any) {
    try { socket.write(JSON.stringify(msg) + "\n"); } catch {}
  }

  // --- Persistence ---

  private getOrCreateKeypair(): { publicKey: Buffer; secretKey: Buffer } {
    try {
      if (fs.existsSync(KEYPAIR_PATH)) {
        const saved = JSON.parse(fs.readFileSync(KEYPAIR_PATH, "utf-8"));
        return {
          publicKey: Buffer.from(saved.publicKey, "hex"),
          secretKey: Buffer.from(saved.secretKey, "hex"),
        };
      }
    } catch {}

    const tempSwarm = new Hyperswarm();
    const kp = tempSwarm.keyPair;
    fs.mkdirSync(path.dirname(KEYPAIR_PATH), { recursive: true });
    fs.writeFileSync(KEYPAIR_PATH, JSON.stringify({
      publicKey: kp.publicKey.toString("hex"),
      secretKey: kp.secretKey.toString("hex"),
    }));
    tempSwarm.destroy();
    return kp;
  }

  private hashBucket(bucketPath: string): Buffer {
    return crypto.createHash("sha256").update(bucketPath).digest();
  }

  private loadTopics(): string[] {
    try { return JSON.parse(fs.readFileSync(TOPICS_PATH, "utf-8")); } catch { return []; }
  }

  private saveTopicList(newTopicHex: string) {
    const topics = this.loadTopics();
    if (!topics.includes(newTopicHex)) {
      topics.push(newTopicHex);
      fs.mkdirSync(path.dirname(TOPICS_PATH), { recursive: true });
      fs.writeFileSync(TOPICS_PATH, JSON.stringify(topics));
    }
  }
}
