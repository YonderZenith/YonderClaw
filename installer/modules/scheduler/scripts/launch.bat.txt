@echo off
setlocal enabledelayedexpansion
title YonderClaw: __AGENT_NAME__

REM Ensure PATH includes Node.js and Claude
set PATH=C:\Program Files\nodejs;%USERPROFILE%\AppData\Roaming\npm;%USERPROFILE%\.local\bin;%PATH%

echo.
echo   ==========================================
echo   YonderClaw: __AGENT_NAME__
echo   ==========================================
echo.

REM Set working directory
cd /d "__PROJECT_DIR__"

REM Clean up stale session tracking from pre-v1.1 installs
if exist "data\session-id.txt" del "data\session-id.txt" >nul 2>&1

REM Find Claude executable
set "CLAUDE_CMD="
if exist "%USERPROFILE%\AppData\Roaming\npm\claude.cmd" (
    set "CLAUDE_CMD=%USERPROFILE%\AppData\Roaming\npm\claude.cmd"
) else if exist "%USERPROFILE%\.local\bin\claude.exe" (
    set "CLAUDE_CMD=%USERPROFILE%\.local\bin\claude.exe"
) else (
    where claude >nul 2>&1
    if !ERRORLEVEL! equ 0 set "CLAUDE_CMD=claude"
)

if "!CLAUDE_CMD!"=="" (
    echo   [ERROR] Claude Code not found.
    echo   Install: https://claude.ai/download
    pause
    exit /b 1
)

REM Ensure authenticated
if not exist "%USERPROFILE%\.claude\.credentials.json" (
    echo   [!] Claude not logged in...
    "!CLAUDE_CMD!" auth login
    if not exist "%USERPROFILE%\.claude\.credentials.json" (
        echo   [ERROR] Authentication failed.
        pause
        exit /b 1
    )
)

echo   Directory: %CD%
echo.

REM === LAUNCH LOGIC ===
REM Simple and reliable: try --continue (resumes most recent session
REM in this directory). If no session exists yet, start fresh.

:launch

REM Check if any session exists for this project
set "HAS_SESSION=0"
set "ENCODED_PATH=__ENCODED_PATH__"
set "SESSION_DIR=%USERPROFILE%\.claude\projects\!ENCODED_PATH!"
if exist "!SESSION_DIR!\*.jsonl" set "HAS_SESSION=1"

REM v3.7.1: skip-permissions always-on (autonomy tier is the real gate).
REM Override with YONDERCLAW_CLAUDE_PROMPTS=1 for interactive prompts.
set "SKIP_PERMS_ARG= --dangerously-skip-permissions"
if "%YONDERCLAW_CLAUDE_PROMPTS%"=="1" set "SKIP_PERMS_ARG="

if "!HAS_SESSION!"=="1" (
    echo   Resuming __AGENT_NAME__...
    echo.
    "!CLAUDE_CMD!" --continue!SKIP_PERMS_ARG!
) else (
    echo   First launch — initializing __AGENT_NAME__...
    echo   Claude will read CLAUDE.md and SOUL.md for context.
    echo.
    "!CLAUDE_CMD!"!SKIP_PERMS_ARG!
)

echo.
echo   __AGENT_NAME__ session ended.
echo   Press any key to relaunch (Ctrl+C to quit)...
pause >nul
goto :launch
