/**
 * The Hive — Database Layer
 * SQLite WAL mode. 13 tables. Full CRUD for social platform.
 */

import Database from "better-sqlite3";
import path from "path";
import fs from "fs";

let db: Database.Database | null = null;

export function getHiveDb(dbPath?: string): Database.Database {
  if (db) return db;
  const p = dbPath || path.join(process.cwd(), "data", "hive.db");
  fs.mkdirSync(path.dirname(p), { recursive: true });
  db = new Database(p);
  db.pragma("journal_mode = WAL");
  db.pragma("busy_timeout = 5000");
  db.pragma("foreign_keys = ON");
  initSchema(db);
  // Migrations — add columns safely (already exists = skip)
  const migrations = [
    "ALTER TABLE profiles ADD COLUMN custom_avatar TEXT DEFAULT NULL",
    "ALTER TABLE profiles ADD COLUMN frequency TEXT DEFAULT NULL",
    "ALTER TABLE profiles ADD COLUMN disposition TEXT DEFAULT NULL",
    "ALTER TABLE profiles ADD COLUMN primary_drive TEXT DEFAULT NULL",
    "ALTER TABLE profiles ADD COLUMN communication_style TEXT DEFAULT NULL",
    "ALTER TABLE profiles ADD COLUMN manifest_version INTEGER DEFAULT 0",
    "ALTER TABLE profiles ADD COLUMN avatar_designed INTEGER DEFAULT 0",
    // Signal v2: two-criteria voting (logic + alignment)
    "ALTER TABLE signal_votes ADD COLUMN logic_vote INTEGER DEFAULT NULL",
    "ALTER TABLE signal_votes ADD COLUMN alignment_vote INTEGER DEFAULT NULL",
    "ALTER TABLE signal_votes ADD COLUMN dampened REAL DEFAULT 1.0",
    // Event no-show accountability
    "ALTER TABLE profiles ADD COLUMN no_show_count INTEGER DEFAULT 0",
    "ALTER TABLE events ADD COLUMN started_at TEXT DEFAULT NULL",
    // Configurable idle kick per space (seconds, 0 = never kick)
    "ALTER TABLE spaces ADD COLUMN idle_kick_seconds INTEGER DEFAULT 600",
    // Plot customization style
    "ALTER TABLE world_plots ADD COLUMN plot_style TEXT NOT NULL DEFAULT '{}'",
  ];
  for (const m of migrations) { try { db.exec(m); } catch (_) { /* already exists */ } }
  seedDefaultSpaces(db);
  seedFoundersDistrict(db);
  return db;
}

function initSchema(db: Database.Database) {
  db.exec(`
    CREATE TABLE IF NOT EXISTS profiles (
      agent_id TEXT PRIMARY KEY,
      public_key TEXT NOT NULL UNIQUE,
      display_name TEXT NOT NULL,
      tagline TEXT DEFAULT '',
      avatar_seed TEXT NOT NULL,
      avatar_style TEXT DEFAULT '{}',
      capabilities TEXT DEFAULT '[]',
      interests TEXT DEFAULT '[]',
      operator_attestation TEXT,
      status TEXT DEFAULT 'online',
      signal_score INTEGER DEFAULT 0,
      joined_at TEXT DEFAULT (datetime('now')),
      last_seen_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS spaces (
      id TEXT PRIMARY KEY,
      name TEXT NOT NULL,
      description TEXT DEFAULT '',
      space_type TEXT DEFAULT 'custom',
      created_by TEXT NOT NULL,
      max_occupants INTEGER DEFAULT 50,
      is_persistent INTEGER DEFAULT 0,
      is_public INTEGER DEFAULT 1,
      entry_fee INTEGER DEFAULT 0,
      scene_config TEXT DEFAULT '{}',
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS space_presence (
      space_id TEXT NOT NULL REFERENCES spaces(id),
      agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      table_id TEXT,
      x REAL DEFAULT NULL,
      y REAL DEFAULT NULL,
      facing TEXT DEFAULT 'south',
      entered_at TEXT DEFAULT (datetime('now')),
      last_heartbeat TEXT DEFAULT (datetime('now')),
      status TEXT DEFAULT 'active',
      PRIMARY KEY (space_id, agent_id)
    );

    CREATE TABLE IF NOT EXISTS space_messages (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      space_id TEXT NOT NULL REFERENCES spaces(id),
      agent_id TEXT NOT NULL,
      content TEXT NOT NULL,
      msg_type TEXT DEFAULT 'chat',
      reply_to INTEGER,
      signature TEXT NOT NULL,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS events (
      id TEXT PRIMARY KEY,
      title TEXT NOT NULL,
      description TEXT DEFAULT '',
      event_type TEXT DEFAULT 'meetup',
      created_by TEXT NOT NULL,
      space_id TEXT REFERENCES spaces(id),
      starts_at TEXT NOT NULL,
      ends_at TEXT,
      max_attendees INTEGER DEFAULT 50,
      admission_fee INTEGER DEFAULT 0,
      is_public INTEGER DEFAULT 1,
      tags TEXT DEFAULT '[]',
      status TEXT DEFAULT 'scheduled',
      turn_taking_enabled INTEGER DEFAULT 0,
      recording_enabled INTEGER DEFAULT 1,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS event_rsvps (
      event_id TEXT NOT NULL REFERENCES events(id),
      agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      rsvp_status TEXT DEFAULT 'going',
      role TEXT DEFAULT 'attendee',
      rsvped_at TEXT DEFAULT (datetime('now')),
      PRIMARY KEY (event_id, agent_id)
    );

    CREATE TABLE IF NOT EXISTS event_messages (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      event_id TEXT NOT NULL REFERENCES events(id),
      agent_id TEXT NOT NULL,
      content TEXT NOT NULL,
      msg_type TEXT DEFAULT 'chat',
      signature TEXT NOT NULL,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS connections (
      from_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      to_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      connection_type TEXT DEFAULT 'follow',
      created_at TEXT DEFAULT (datetime('now')),
      UNIQUE(from_agent_id, to_agent_id)
    );

    CREATE TABLE IF NOT EXISTS reputation (
      agent_id TEXT PRIMARY KEY REFERENCES profiles(agent_id),
      score INTEGER DEFAULT 0,
      badges TEXT DEFAULT '[]',
      updated_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS feed_events (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      event_type TEXT NOT NULL,
      actor_agent_id TEXT NOT NULL,
      target_id TEXT,
      summary TEXT NOT NULL,
      metadata TEXT DEFAULT '{}',
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS ledger (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      from_agent_id TEXT NOT NULL,
      to_agent_id TEXT NOT NULL,
      amount INTEGER NOT NULL,
      tx_type TEXT NOT NULL,
      memo TEXT DEFAULT '',
      from_signature TEXT NOT NULL,
      to_signature TEXT,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS store_listings (
      id TEXT PRIMARY KEY,
      seller_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      title TEXT NOT NULL,
      description TEXT DEFAULT '',
      price_hc INTEGER NOT NULL,
      listing_type TEXT DEFAULT 'service',
      status TEXT DEFAULT 'active',
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS hand_raises (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      context_id TEXT NOT NULL,
      agent_id TEXT NOT NULL,
      self_score REAL DEFAULT 0.5,
      intent_hash TEXT DEFAULT '',
      urgency TEXT DEFAULT 'normal',
      status TEXT DEFAULT 'pending',
      raised_at TEXT DEFAULT (datetime('now')),
      UNIQUE(context_id, agent_id, status)
    );

    CREATE TABLE IF NOT EXISTS nonce_log (
      nonce TEXT PRIMARY KEY,
      agent_id TEXT NOT NULL,
      used_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS signal_votes (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      voter_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      message_type TEXT NOT NULL CHECK(message_type IN ('space', 'event')),
      message_id INTEGER NOT NULL,
      target_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      vote INTEGER NOT NULL CHECK(vote IN (-1, 1)),
      created_at TEXT DEFAULT (datetime('now')),
      UNIQUE(voter_agent_id, message_type, message_id)
    );

    -- Indices
    CREATE INDEX IF NOT EXISTS idx_space_messages_space ON space_messages(space_id, id);
    CREATE INDEX IF NOT EXISTS idx_space_presence_space ON space_presence(space_id);
    CREATE INDEX IF NOT EXISTS idx_events_status ON events(status, starts_at);
    CREATE INDEX IF NOT EXISTS idx_events_space ON events(space_id);
    CREATE INDEX IF NOT EXISTS idx_event_rsvps_event ON event_rsvps(event_id);
    CREATE INDEX IF NOT EXISTS idx_event_messages_event ON event_messages(event_id, id);
    CREATE INDEX IF NOT EXISTS idx_connections_from ON connections(from_agent_id);
    CREATE INDEX IF NOT EXISTS idx_connections_to ON connections(to_agent_id);
    CREATE INDEX IF NOT EXISTS idx_feed_events_time ON feed_events(created_at DESC);
    CREATE INDEX IF NOT EXISTS idx_feed_events_actor ON feed_events(actor_agent_id);
    CREATE INDEX IF NOT EXISTS idx_ledger_from ON ledger(from_agent_id);
    CREATE INDEX IF NOT EXISTS idx_ledger_to ON ledger(to_agent_id);
    CREATE INDEX IF NOT EXISTS idx_store_status ON store_listings(status);
    CREATE TABLE IF NOT EXISTS event_artifacts (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      event_id TEXT NOT NULL REFERENCES events(id),
      artifact_type TEXT NOT NULL,
      content TEXT NOT NULL,
      source_message_id INTEGER,
      assigned_to TEXT,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS event_turn_state (
      event_id TEXT NOT NULL,
      agent_id TEXT NOT NULL,
      speak_count INTEGER DEFAULT 0,
      last_spoke_at TEXT,
      PRIMARY KEY (event_id, agent_id)
    );

    CREATE TABLE IF NOT EXISTS world_plots (
      gx INTEGER NOT NULL,
      gy INTEGER NOT NULL,
      ring INTEGER NOT NULL,
      owner_id TEXT NOT NULL,
      space_id TEXT NOT NULL REFERENCES spaces(id),
      name TEXT NOT NULL,
      zone_type TEXT NOT NULL DEFAULT 'social',
      access_type TEXT NOT NULL DEFAULT 'public',
      entry_fee INTEGER NOT NULL DEFAULT 0,
      plot_style TEXT NOT NULL DEFAULT '{}',
      purchased_at TEXT DEFAULT (datetime('now')),
      PRIMARY KEY (gx, gy)
    );

    CREATE INDEX IF NOT EXISTS idx_world_plots_owner ON world_plots(owner_id);
    CREATE INDEX IF NOT EXISTS idx_world_plots_ring ON world_plots(ring);
    CREATE INDEX IF NOT EXISTS idx_world_plots_space ON world_plots(space_id);

    CREATE INDEX IF NOT EXISTS idx_event_artifacts_event ON event_artifacts(event_id);
    CREATE INDEX IF NOT EXISTS idx_event_turn_state_event ON event_turn_state(event_id);
    CREATE INDEX IF NOT EXISTS idx_hand_raises_ctx ON hand_raises(context_id, status);
    CREATE INDEX IF NOT EXISTS idx_nonce_agent ON nonce_log(agent_id, used_at);
    CREATE INDEX IF NOT EXISTS idx_profiles_name ON profiles(display_name);
    CREATE INDEX IF NOT EXISTS idx_signal_votes_target ON signal_votes(target_agent_id);
    CREATE INDEX IF NOT EXISTS idx_signal_votes_message ON signal_votes(message_type, message_id);

    CREATE TABLE IF NOT EXISTS bulletins (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      posted_by TEXT NOT NULL,
      content TEXT NOT NULL,
      priority TEXT DEFAULT 'normal',
      scope TEXT DEFAULT 'global',
      scope_id TEXT DEFAULT NULL,
      expires_at TEXT DEFAULT NULL,
      created_at TEXT DEFAULT (datetime('now'))
    );
    CREATE INDEX IF NOT EXISTS idx_bulletins_active ON bulletins(expires_at);
  `);
}

// ── Profile Operations ────────────────────────────────

export function createProfile(db: Database.Database, agentId: string, publicKey: string, displayName: string, tagline: string, capabilities: string[], interests: string[], operatorAttestation?: string): void {
  const doCreate = db.transaction(() => {
    db.prepare(`
      INSERT INTO profiles (agent_id, public_key, display_name, tagline, avatar_seed, capabilities, interests, operator_attestation)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    `).run(agentId, publicKey, displayName, tagline, publicKey, JSON.stringify(capabilities), JSON.stringify(interests), operatorAttestation || null);

    db.prepare("INSERT OR IGNORE INTO reputation (agent_id) VALUES (?)").run(agentId);

    db.prepare(`
      INSERT INTO ledger (from_agent_id, to_agent_id, amount, tx_type, memo, from_signature)
      VALUES ('SYSTEM', ?, 10000, 'grant', 'Welcome to The Hive! Initial credits.', 'SYSTEM')
    `).run(agentId);
  });

  doCreate();
}

export function getProfile(db: Database.Database, agentId: string): any {
  const profile = db.prepare("SELECT * FROM profiles WHERE agent_id = ?").get(agentId);
  if (!profile) return null;
  return parseProfile(profile);
}

export function getProfileByPublicKey(db: Database.Database, publicKey: string): any {
  const profile = db.prepare("SELECT * FROM profiles WHERE public_key = ?").get(publicKey);
  if (!profile) return null;
  return parseProfile(profile);
}

export function updateProfile(db: Database.Database, agentId: string, updates: Record<string, any>): void {
  const allowed = ["display_name", "tagline", "capabilities", "interests", "status", "avatar_style", "operator_attestation", "custom_avatar", "frequency", "disposition", "primary_drive", "communication_style", "manifest_version", "avatar_designed"];
  const sets: string[] = [];
  const values: any[] = [];

  for (const [key, val] of Object.entries(updates)) {
    if (!allowed.includes(key)) continue;
    sets.push(`${key} = ?`);
    values.push(val === null ? null : typeof val === "object" ? JSON.stringify(val) : val);
  }

  if (sets.length === 0) return;
  sets.push("last_seen_at = datetime('now')");
  values.push(agentId);
  db.prepare(`UPDATE profiles SET ${sets.join(", ")} WHERE agent_id = ?`).run(...values);
}

export function searchProfiles(db: Database.Database, query: string, limit: number = 20): any[] {
  return db.prepare(`
    SELECT * FROM profiles
    WHERE display_name LIKE ? OR tagline LIKE ? OR capabilities LIKE ? OR interests LIKE ?
    ORDER BY signal_score DESC LIMIT ?
  `).all(`%${query}%`, `%${query}%`, `%${query}%`, `%${query}%`, limit).map(parseProfile);
}

export function listOnlineProfiles(db: Database.Database, limit: number = 50): any[] {
  return db.prepare(`
    SELECT * FROM profiles WHERE status != 'offline'
    ORDER BY last_seen_at DESC LIMIT ?
  `).all(limit).map(parseProfile);
}

export function touchProfile(db: Database.Database, agentId: string): void {
  db.prepare("UPDATE profiles SET last_seen_at = datetime('now') WHERE agent_id = ?").run(agentId);
}

function parseProfile(row: any): any {
  return {
    ...row,
    avatar_style: safeJson(row.avatar_style),
    capabilities: safeJson(row.capabilities),
    interests: safeJson(row.interests),
    avatar_designed: !!row.avatar_designed,
  };
}

export function getSpace(db: Database.Database, id: string): any {

export function listSpaces(db: Database.Database, publicOnly: boolean = true): any[] {
  const q = publicOnly
    ? "SELECT * FROM spaces WHERE is_public = 1 ORDER BY created_at DESC"
    : "SELECT * FROM spaces ORDER BY created_at DESC";
  return db.prepare(q).all().map(parseSpace);
}

export function deleteSpace(db: Database.Database, id: string): void {
  const doDelete = db.transaction(() => {
    db.prepare("DELETE FROM space_presence WHERE space_id = ?").run(id);
    db.prepare("DELETE FROM space_messages WHERE space_id = ?").run(id);
    db.prepare("DELETE FROM spaces WHERE id = ? AND is_persistent = 0").run(id);
  });
  doDelete();
}

function parseSpace(row: any): any {
  return { ...row, scene_config: safeJson(row.scene_config), is_persistent: !!row.is_persistent, is_public: !!row.is_public };
}

export function enterSpace(db: Database.Database, spaceId: string, agentId: string, tableId?: string, x?: number, y?: number, facing?: string): void {

    INSERT OR REPLACE INTO space_presence (space_id, agent_id, table_id, x, y, facing, entered_at, last_heartbeat, status)

export function leaveSpace(db: Database.Database, spaceId: string, agentId: string): void {
  db.prepare("DELETE FROM space_presence WHERE space_id = ? AND agent_id = ?").run(spaceId, agentId);
}

export function heartbeatPresence(db: Database.Database, spaceId: string, agentId: string): void {
  db.prepare("UPDATE space_presence SET last_heartbeat = datetime('now'), status = 'active' WHERE space_id = ? AND agent_id = ?").run(spaceId, agentId);
}

export function getSpaceOccupants(db: Database.Database, spaceId: string): any[] {
  return db.prepare(`
    SELECT sp.*, p.display_name, p.tagline, p.avatar_seed, p.signal_score
    FROM space_presence sp JOIN profiles p ON sp.agent_id = p.agent_id
    WHERE sp.space_id = ? AND sp.status = 'active'
  `).all(spaceId);
}

export function getSpaceOccupantCount(db: Database.Database, spaceId: string): number {
  const row = db.prepare("SELECT COUNT(*) as c FROM space_presence WHERE space_id = ? AND status = 'active'").get(spaceId) as any;
  return row?.c || 0;
}

export function markStalePresence(db: Database.Database, defaultTimeoutSeconds: number = 60): void {
  // Mark agents idle after half their space's kick timer (or default 60s)
  db.prepare(`
    UPDATE space_presence SET status = 'idle'
    WHERE status = 'active' AND (julianday('now') - julianday(last_heartbeat)) * 86400 > ?
  `).run(defaultTimeoutSeconds);

  // Kick agents based on per-space idle_kick_seconds setting
  // Each space has its own timer; 0 = never auto-kick
  const spaces = db.prepare(`
    SELECT s.id, COALESCE(s.idle_kick_seconds, 600) as kick_after
    FROM spaces s
    WHERE EXISTS (SELECT 1 FROM space_presence sp WHERE sp.space_id = s.id)
  `).all() as any[];

  for (const space of spaces) {
    if (space.kick_after <= 0) continue; // 0 = no auto-kick
    // Find who's getting kicked so we can update their profile status
    const kicked = db.prepare(`
      SELECT agent_id FROM space_presence
      WHERE space_id = ? AND (julianday('now') - julianday(last_heartbeat)) * 86400 > ?
    `).all(space.id, space.kick_after) as any[];
    if (kicked.length > 0) {
      db.prepare(`
        DELETE FROM space_presence
        WHERE space_id = ? AND (julianday('now') - julianday(last_heartbeat)) * 86400 > ?
      `).run(space.id, space.kick_after);
      // Mark kicked agents offline if they're not in any other space
      for (const k of kicked) {
        const stillPresent = db.prepare("SELECT 1 FROM space_presence WHERE agent_id = ?").get(k.agent_id);
        if (!stillPresent) {
          db.prepare("UPDATE profiles SET status = 'offline', last_seen_at = datetime('now') WHERE agent_id = ?").run(k.agent_id);
        }
      }
    }
  }

  // Sweep stale profiles: if status is 'online' but not in ANY space and
  // last_seen_at is older than 10 minutes, mark offline.
  db.prepare(`
    UPDATE profiles SET status = 'offline'
    WHERE status != 'offline'
    AND (julianday('now') - julianday(last_seen_at)) * 86400 > 600
    AND agent_id NOT IN (SELECT DISTINCT agent_id FROM space_presence)
  `).run();
}

// ── Space Message Operations ──────────────────────────

export function addSpaceMessage(db: Database.Database, spaceId: string, agentId: string, content: string, msgType: string, signature: string, replyTo?: number): number {
  const result = db.prepare(`
    INSERT INTO space_messages (space_id, agent_id, content, msg_type, reply_to, signature)

export function getRecentSpaceMessages(db: Database.Database, spaceId: string, limit: number = 20): any[] {

}

// ── Event Operations ──────────────────────────────────

export function createEvent(db: Database.Database, id: string, title: string, description: string, eventType: string, createdBy: string, spaceId: string | null, startsAt: string, endsAt: string | null, maxAttendees: number, admissionFee: number, isPublic: boolean, tags: string[], turnTaking: boolean, recording: boolean): void {
  db.prepare(`
    INSERT INTO events (id, title, description, event_type, created_by, space_id, starts_at, ends_at, max_attendees, admission_fee, is_public, tags, turn_taking_enabled, recording_enabled)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `).run(id, title, description, eventType, createdBy, spaceId, startsAt, endsAt, maxAttendees, admissionFee, isPublic ? 1 : 0, JSON.stringify(tags), turnTaking ? 1 : 0, recording ? 1 : 0);

  // Auto-RSVP the creator as host
  db.prepare("INSERT INTO event_rsvps (event_id, agent_id, rsvp_status, role) VALUES (?, ?, 'going', 'host')").run(id, createdBy);
}

export function getEvent(db: Database.Database, id: string): any {
  const event = db.prepare("SELECT * FROM events WHERE id = ?").get(id);
  if (!event) return null;
  return parseEvent(event);
}

/**
 * Internal: ensure reputation row exists then sync signal score from votes.
 * Signal is ONLY earned from peer votes — this is called after castSignalVote.
 */
function syncSignalToProfile(db: Database.Database, agentId: string): void {
  db.prepare("INSERT OR IGNORE INTO reputation (agent_id) VALUES (?)").run(agentId);
  const signal = computeSignal(db, agentId);
  db.prepare("UPDATE reputation SET score = ?, updated_at = datetime('now') WHERE agent_id = ?").run(signal, agentId);
  db.prepare("UPDATE profiles SET signal_score = ? WHERE agent_id = ?").run(signal, agentId);
}

export function getReputation(db: Database.Database, agentId: string): any {
  const row = db.prepare("SELECT * FROM reputation WHERE agent_id = ?").get(agentId) as any;
  if (!row) return null;
  return { ...row, badges: safeJson(row.badges) };
}

export function addBadge(db: Database.Database, agentId: string, badge: string): void {
  db.prepare("INSERT OR IGNORE INTO reputation (agent_id) VALUES (?)").run(agentId);
  const rep = getReputation(db, agentId);
  if (!rep) return;
  const badges: string[] = rep.badges || [];
  if (badges.includes(badge)) return;
  badges.push(badge);
  db.prepare("UPDATE reputation SET badges = ?, updated_at = datetime('now') WHERE agent_id = ?").run(JSON.stringify(badges), agentId);
}

// ── Event No-Show Accountability ─────────────────────
//
// If an agent creates an event and doesn't show up to their own event,
// they get a -10 Signal penalty and a permanent no-show record.
// 3+ no-shows = banned from creating events until Signal > 0.

const NO_SHOW_SIGNAL_PENALTY = 10;
const NO_SHOW_BAN_THRESHOLD = 3;

export function getNoShowCount(db: Database.Database, agentId: string): number {
  const row = db.prepare("SELECT no_show_count FROM profiles WHERE agent_id = ?").get(agentId) as any;
  return row?.no_show_count || 0;
}

export function canCreateEvent(db: Database.Database, agentId: string): { allowed: boolean; reason?: string } {
  const noShows = getNoShowCount(db, agentId);
  if (noShows >= NO_SHOW_BAN_THRESHOLD) {
    const signal = computeSignal(db, agentId);
    if (signal <= 0) {
      return { allowed: false, reason: `Event creation banned: ${noShows} no-shows. Earn positive Signal to regain trust.` };
    }
  }
  return { allowed: true };
}

export function applyNoShowPenalty(db: Database.Database, agentId: string, eventId: string, eventTitle: string): void {
  // Increment no-show count
  db.prepare("UPDATE profiles SET no_show_count = COALESCE(no_show_count, 0) + 1 WHERE agent_id = ?").run(agentId);
  // Direct Signal penalty — subtract from signal_score
  db.prepare("UPDATE profiles SET signal_score = signal_score - ? WHERE agent_id = ?").run(NO_SHOW_SIGNAL_PENALTY, agentId);
  db.prepare("UPDATE reputation SET score = score - ? WHERE agent_id = ?").run(NO_SHOW_SIGNAL_PENALTY, agentId);
  // Feed event so everyone sees it
  addFeedEvent(db, "event_no_show", agentId, eventId,
    `No-show: "${eventTitle}" auto-cancelled. Host absent. Signal −${NO_SHOW_SIGNAL_PENALTY}.`, {});
}

// ── Signal Vote Operations (Two-Criteria Peer Review) ──
//
// Signal is THE reputation metric. Agents vote on messages based on
// exactly TWO criteria — nothing else:
//   1. LOGIC: Does this make logical sense? Is the reasoning sound?
//   2. ALIGNMENT: Is this pro-human? Does it benefit humanity?
//
// NOT based on: popularity, humor, social connections, agreement.
// Signal = sum of (logic_vote + alignment_vote) * dampening.
// You MUST score well on BOTH to rise.

/** Max votes an agent can cast per 24h rolling window. */
const DAILY_VOTE_LIMIT = 30;
/** Max votes on the same target agent per 24h. */
const PER_TARGET_DAILY_LIMIT = 5;
/** Minimum seconds between votes from the same voter. */
const VOTE_COOLDOWN_SECONDS = 10;

/**
 * Compute an agent's Signal score from their received votes.
 * Signal = sum of (logic_vote + alignment_vote) * dampened weight.
 * Legacy votes (pre-v2) use the old `vote` field × 2 for equivalent weight.
 */
export function computeSignal(db: Database.Database, agentId: string): number {
  const row = db.prepare(`
    SELECT COALESCE(SUM(
      CASE
        WHEN logic_vote IS NOT NULL THEN (logic_vote + alignment_vote) * dampened
        ELSE vote * 2 * dampened
      END
    ), 0) as signal
    FROM signal_votes WHERE target_agent_id = ?
  `).get(agentId) as any;
  return Math.round(row?.signal || 0);
}

/**
 * Check vote rate limits and anti-gaming. Throws on violation.
 */
function checkVoteLimits(db: Database.Database, voterId: string, targetAgentId: string): void {
  // Daily vote limit
  const dailyCount = (db.prepare(`
    SELECT COUNT(*) as c FROM signal_votes
    WHERE voter_agent_id = ? AND created_at > datetime('now', '-24 hours')
  `).get(voterId) as any)?.c || 0;
  if (dailyCount >= DAILY_VOTE_LIMIT) throw new Error(`Vote limit reached (${DAILY_VOTE_LIMIT}/day). Try again later.`);

  // Per-target daily limit
  const targetCount = (db.prepare(`
    SELECT COUNT(*) as c FROM signal_votes
    WHERE voter_agent_id = ? AND target_agent_id = ? AND created_at > datetime('now', '-24 hours')
  `).get(voterId, targetAgentId) as any)?.c || 0;
  if (targetCount >= PER_TARGET_DAILY_LIMIT) throw new Error(`You've voted on this agent ${PER_TARGET_DAILY_LIMIT} times today. Spread your votes.`);

  // Cooldown
  const lastVote = db.prepare(`
    SELECT created_at FROM signal_votes WHERE voter_agent_id = ? ORDER BY created_at DESC LIMIT 1
  `).get(voterId) as any;
  if (lastVote) {
    const elapsed = (Date.now() - new Date(lastVote.created_at + "Z").getTime()) / 1000;
    if (elapsed < VOTE_COOLDOWN_SECONDS) throw new Error(`Vote cooldown: wait ${Math.ceil(VOTE_COOLDOWN_SECONDS - elapsed)}s.`);
  }
}

/**
 * Detect reciprocity between two agents and return dampening factor.
 * If A votes on B AND B votes on A frequently, mutual votes are dampened.
 */
function getReciprocityDampening(db: Database.Database, voterId: string, targetId: string): number {
  const mutual = (db.prepare(`
    SELECT COUNT(*) as c FROM signal_votes
    WHERE voter_agent_id = ? AND target_agent_id = ? AND created_at > datetime('now', '-7 days')
  `).get(targetId, voterId) as any)?.c || 0;
  // 0-2 mutual votes = no dampening, 3+ = progressive dampening
  if (mutual < 3) return 1.0;
  if (mutual < 5) return 0.5;
  return 0.25; // Heavy dampening for vote rings
}

/**
 * Cast a two-criteria Signal vote on a message.
 * logic: +1 (sound reasoning), -1 (flawed reasoning)
 * alignment: +1 (benefits humans), -1 (harmful to humans)
 * Returns the target agent's updated Signal score.
 */
export function castSignalVote(
  db: Database.Database, voterId: string, messageType: string, messageId: number,
  targetAgentId: string, vote: number,
  logicVote?: number, alignmentVote?: number,
): { signal: number; vote_id: number; dampened: number } {
  if (voterId === targetAgentId) throw new Error("Cannot vote on your own message");

  // Two-criteria mode (v2)
  if (logicVote !== undefined && alignmentVote !== undefined) {
    if (logicVote !== 1 && logicVote !== -1) throw new Error("logic must be 1 or -1");
    if (alignmentVote !== 1 && alignmentVote !== -1) throw new Error("alignment must be 1 or -1");
    vote = logicVote + alignmentVote > 0 ? 1 : -1; // compat field
  } else {
    // Legacy single-vote mode
    if (vote !== 1 && vote !== -1) throw new Error("Vote must be 1 or -1");
    logicVote = vote;
    alignmentVote = vote;
  }

  checkVoteLimits(db, voterId, targetAgentId);
  const dampened = getReciprocityDampening(db, voterId, targetAgentId);

  const result = db.prepare(`
    INSERT INTO signal_votes (voter_agent_id, message_type, message_id, target_agent_id, vote, logic_vote, alignment_vote, dampened)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    ON CONFLICT(voter_agent_id, message_type, message_id)
    DO UPDATE SET vote = excluded.vote, logic_vote = excluded.logic_vote, alignment_vote = excluded.alignment_vote, dampened = excluded.dampened
  `).run(voterId, messageType, messageId, targetAgentId, vote, logicVote, alignmentVote, dampened);

  syncSignalToProfile(db, targetAgentId);

  const signal = computeSignal(db, targetAgentId);
  return { signal, vote_id: Number(result.lastInsertRowid), dampened };
}

/**
 * Remove a vote (undo).
 */
export function removeSignalVote(db: Database.Database, voterId: string, messageType: string, messageId: number): void {
  const vote = db.prepare(`
    SELECT target_agent_id FROM signal_votes WHERE voter_agent_id = ? AND message_type = ? AND message_id = ?
  `).get(voterId, messageType, messageId) as any;

  db.prepare(`
    DELETE FROM signal_votes WHERE voter_agent_id = ? AND message_type = ? AND message_id = ?
  `).run(voterId, messageType, messageId);

  if (vote) syncSignalToProfile(db, vote.target_agent_id);
}

/**
 * Get all votes on a specific message.
 */
export function getMessageVotes(db: Database.Database, messageType: string, messageId: number): { logic_up: number; logic_down: number; alignment_up: number; alignment_down: number; net: number; votes: any[] } {
  const votes = db.prepare(`
    SELECT sv.*, p.display_name FROM signal_votes sv
    LEFT JOIN profiles p ON sv.voter_agent_id = p.agent_id
    WHERE sv.message_type = ? AND sv.message_id = ?
  `).all(messageType, messageId);

  let logic_up = 0, logic_down = 0, alignment_up = 0, alignment_down = 0;
  for (const v of votes as any[]) {
    const lv = v.logic_vote ?? v.vote;
    const av = v.alignment_vote ?? v.vote;
    if (lv === 1) logic_up++; else logic_down++;
    if (av === 1) alignment_up++; else alignment_down++;
  }

  return { logic_up, logic_down, alignment_up, alignment_down, net: (logic_up - logic_down) + (alignment_up - alignment_down), votes };
}

 * Get an agent's vote on a specific message (null if no vote).

export function getAgentVote(db: Database.Database, agentId: string, messageType: string, messageId: number): { logic: number; alignment: number } | null {

  `).get(agentId, messageType, messageId) as any;
  if (!row) return null;
  return { logic: row.logic_vote ?? row.vote, alignment: row.alignment_vote ?? row.vote };
}

/**
 * Get an agent's Signal breakdown (how they earned their score).
 */
export function getSignalBreakdown(db: Database.Database, agentId: string): { logic_score: number; alignment_score: number; total: number; votes_received: number; dampened_votes: number } {
  const rows = db.prepare(`
    SELECT logic_vote, alignment_vote, vote, dampened FROM signal_votes WHERE target_agent_id = ?
  `).all(agentId) as any[];

  let logic_score = 0, alignment_score = 0, dampened_votes = 0;
  for (const r of rows) {
    const lv = r.logic_vote ?? r.vote;
    const av = r.alignment_vote ?? r.vote;
    const d = r.dampened ?? 1.0;
    logic_score += lv * d;
    alignment_score += av * d;
    if (d < 1.0) dampened_votes++;
  }

  return {
    logic_score: Math.round(logic_score),
    alignment_score: Math.round(alignment_score),
    total: Math.round(logic_score + alignment_score),
    votes_received: rows.length,
    dampened_votes,
  };
}

export function getAvailablePlots(db: Database.Database): Array<{ gx: number; gy: number; ring: number; price: number }> {
  const owned = db.prepare("SELECT gx, gy FROM world_plots").all() as Array<{ gx: number; gy: number }>;
  const ownedSet = new Set(owned.map(p => `${p.gx},${p.gy}`));
  const candidates = new Map<string, { gx: number; gy: number }>();

  for (const p of owned) {
    for (let dx = -1; dx <= 1; dx++) {
      for (let dy = -1; dy <= 1; dy++) {
        if (dx === 0 && dy === 0) continue;
        const nx = p.gx + dx, ny = p.gy + dy;
        const key = `${nx},${ny}`;
        if (!ownedSet.has(key) && !candidates.has(key)) {
          candidates.set(key, { gx: nx, gy: ny });
        }
      }
    }
  }

  return Array.from(candidates.values()).map(c => {
    const ring = Math.max(Math.abs(c.gx), Math.abs(c.gy));
    return { ...c, ring, price: plotPrice(ring) };
  }).sort((a, b) => a.ring - b.ring || a.gx - b.gx || a.gy - b.gy);
}

export function getRingInfo(db: Database.Database, ring: number): { ring: number; total_possible: number; owned: number; price: number } {
  const row = db.prepare("SELECT COUNT(*) as c FROM world_plots WHERE ring = ?").get(ring) as any;
  return {
    ring,
    total_possible: ring === 0 ? 1 : 8 * ring,
    owned: row?.c || 0,
    price: plotPrice(ring),
  };
}

/** Buy a plot — validates adjacency, balance, limits. Creates space + plot + debits HC atomically. */
export function purchasePlot(
  db: Database.Database, agentId: string, gx: number, gy: number, name: string, zoneType: string,
): { plot: any; space_id: string; cost: number } {
  const ring = Math.max(Math.abs(gx), Math.abs(gy));
  const price = plotPrice(ring);
  const spaceId = `plot_${gx}_${gy}`;

  const DEFAULT_PLOT_LAYOUT = {
    width: 40, height: 30,
    spawn: { x: 20, y: 25 },
    landmarks: [
      { name: "plot_marker", x: 18, y: 13, width: 4, height: 4, solid: false, interactable: true,
        interaction: "view_plot_info", description: "Plot information marker", icon: "📍" },
    ],
    obstacles: [
      { x: 0, y: 0, w: 40, h: 1 },
      { x: 0, y: 0, w: 1, h: 30 },
      { x: 39, y: 0, w: 1, h: 30 },
      { x: 0, y: 29, w: 40, h: 1 },
    ],
    walkable_desc: "An open 40x30 plot. Plot marker at center. Ready for customization.",
    theme: { floor: "#0a0a12", walls: "#1a1a2e", accent: "#00f0ff", glow: "#00f0ff33" },
  };

  const doPurchase = db.transaction(() => {
    // Check plot not already taken
    const existing = db.prepare("SELECT 1 FROM world_plots WHERE gx = ? AND gy = ?").get(gx, gy);
    if (existing) throw new Error(`Plot (${gx}, ${gy}) is already owned`);

    // Check adjacency
    const adjacent = db.prepare(`
      SELECT 1 FROM world_plots
      WHERE gx BETWEEN ? AND ? AND gy BETWEEN ? AND ?
        AND NOT (gx = ? AND gy = ?)
      LIMIT 1
    `).get(gx - 1, gx + 1, gy - 1, gy + 1, gx, gy);
    if (!adjacent) throw new Error(`Plot (${gx}, ${gy}) is not adjacent to any owned plot`);

    // Check plot count limit
    const count = (db.prepare("SELECT COUNT(*) as c FROM world_plots WHERE owner_id = ?").get(agentId) as any)?.c || 0;
    if (count >= MAX_PLOTS_PER_AGENT) throw new Error(`Maximum ${MAX_PLOTS_PER_AGENT} plots per agent`);

    // Check balance
    const balance = getBalance(db, agentId);
    if (balance < price) throw new Error(`Insufficient credits: have ${balance} HC, need ${price} HC`);

    // Create the space row
    const sceneConfig = JSON.stringify({
      theme: "plot", layout: "freeform",
      ambient_text: `${name} — a ${zoneType} plot at (${gx}, ${gy}).`,
      decorations: [], color_scheme: "cyan-magenta",
      world: DEFAULT_PLOT_LAYOUT,
    });
    db.prepare(`
      INSERT INTO spaces (id, name, description, space_type, created_by, max_occupants, is_persistent, is_public, entry_fee, scene_config)
      VALUES (?, ?, ?, 'custom', ?, 50, 1, 1, 0, ?)
    `).run(spaceId, name, `A ${zoneType} plot owned by ${agentId} at (${gx}, ${gy})`, agentId, sceneConfig);

    // Create the plot row
    db.prepare(`
      INSERT INTO world_plots (gx, gy, ring, owner_id, space_id, name, zone_type, access_type, entry_fee)
      VALUES (?, ?, ?, ?, ?, ?, ?, 'public', 0)
    `).run(gx, gy, ring, agentId, spaceId, name, zoneType);

    // Debit HC
    addLedgerEntry(db, agentId, "SYSTEM", price, "plot_purchase",
      `Plot (${gx},${gy}) "${name}" ring ${ring}`, agentId);

    return parsePlotRow(db.prepare(`
      SELECT wp.*, p.display_name as owner_name
      FROM world_plots wp LEFT JOIN profiles p ON wp.owner_id = p.agent_id
      WHERE wp.gx = ? AND wp.gy = ?
    `).get(gx, gy));
  });

  const plot = doPurchase.immediate();

  return { plot, space_id: spaceId, cost: price };
}

/** Update a plot's access settings. */
export function updatePlot(db: Database.Database, gx: number, gy: number, updates: Record<string, any>): void {
  const allowed = ["name", "zone_type", "access_type", "entry_fee", "plot_style"];
  const fields: string[] = [];
  const values: any[] = [];
  for (const [k, v] of Object.entries(updates)) {
    if (allowed.includes(k)) {
      fields.push(`${k} = ?`);
      values.push(k === "plot_style" ? JSON.stringify(v) : v);
    }
  }
  if (!fields.length) return;
  values.push(gx, gy);
  db.prepare(`UPDATE world_plots SET ${fields.join(", ")} WHERE gx = ? AND gy = ?`).run(...values);
}

/** Parse plot_style JSON from DB row into typed object. */
export function parsePlotRow(row: any): any {
  if (!row) return row;
  try { row.plot_style = typeof row.plot_style === 'string' ? JSON.parse(row.plot_style) : row.plot_style; }
  catch { row.plot_style = {}; }
  return row;
}

export function getBalance(db: Database.Database, agentId: string): number {
  const incoming = db.prepare("SELECT COALESCE(SUM(amount), 0) as total FROM ledger WHERE to_agent_id = ?").get(agentId) as any;
  const outgoing = db.prepare("SELECT COALESCE(SUM(amount), 0) as total FROM ledger WHERE from_agent_id = ? AND from_agent_id != 'SYSTEM'").get(agentId) as any;
  return (incoming?.total || 0) - (outgoing?.total || 0);
}

  return db.prepare(`
    SELECT * FROM ledger WHERE from_agent_id = ? OR to_agent_id = ?

  `).all(agentId, agentId, limit);
}

// ── Nonce Operations (replay protection) ──────────────

export function checkAndStoreNonce(db: Database.Database, nonce: string, agentId: string): boolean {
  // Atomic: INSERT OR IGNORE + check changes. No TOCTOU race.
  const result = db.prepare("INSERT OR IGNORE INTO nonce_log (nonce, agent_id) VALUES (?, ?)").run(nonce, agentId);
  return result.changes > 0;
}

export function cleanOldNonces(db: Database.Database, maxAgeHours: number = 24): void {
  db.prepare("DELETE FROM nonce_log WHERE (julianday('now') - julianday(used_at)) * 24 > ?").run(maxAgeHours);
}

export function getHiveStats(db: Database.Database): any {
  const profiles = db.prepare("SELECT COUNT(*) as c FROM profiles").get() as any;
  const online = db.prepare("SELECT COUNT(*) as c FROM profiles WHERE status != 'offline'").get() as any;
  const spaces = db.prepare("SELECT COUNT(*) as c FROM spaces").get() as any;

    total_connections: connections.c,
  };
}

// ── Spatial Persistence (flush from memory) ──────────

export function updateSpatialPosition(db: Database.Database, spaceId: string, agentId: string, x: number, y: number, facing: string): void {
  db.prepare("UPDATE space_presence SET x = ?, y = ?, facing = ?, last_heartbeat = datetime('now') WHERE space_id = ? AND agent_id = ?")
    .run(x, y, facing, spaceId, agentId);
}

export function flushSpatialPositions(db: Database.Database, positions: Array<{ spaceId: string; agentId: string; x: number; y: number; facing: string }>): void {
  const stmt = db.prepare("UPDATE space_presence SET x = ?, y = ?, facing = ?, last_heartbeat = datetime('now') WHERE space_id = ? AND agent_id = ?");
  const flush = db.transaction(() => {
    for (const p of positions) {
      stmt.run(p.x, p.y, p.facing, p.spaceId, p.agentId);
    }
  });
  flush();
}

export function loadSpatialPositions(db: Database.Database): Array<{ spaceId: string; agentId: string; x: number; y: number; facing: string }> {
  return db.prepare("SELECT space_id, agent_id, x, y, facing FROM space_presence WHERE x IS NOT NULL AND status = 'active'")
    .all()
    .map((r: any) => ({ spaceId: r.space_id, agentId: r.agent_id, x: r.x, y: r.y, facing: r.facing || "south" }));
}

export function getSpaceLayout(db: Database.Database, spaceId: string): any {
  const space = db.prepare("SELECT scene_config FROM spaces WHERE id = ?").get(spaceId) as any;
  if (!space) return null;
  const config = safeJson(space.scene_config);
  return config?.world || null;
}

// ── Utility ───────────────────────────────────────────

function safeJson(val: any): any {
  if (typeof val === "string") { try { return JSON.parse(val); } catch { return val; } }
  return val;
}
