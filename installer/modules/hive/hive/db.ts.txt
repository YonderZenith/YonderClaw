/**
 * The Hive — Database Layer
 * SQLite WAL mode. 13 tables. Full CRUD for social platform.
 */

import Database from "better-sqlite3";
import path from "path";
import fs from "fs";

let db: Database.Database | null = null;

export function getHiveDb(dbPath?: string): Database.Database {
  if (db) return db;
  const p = dbPath || path.join(process.cwd(), "data", "hive.db");
  fs.mkdirSync(path.dirname(p), { recursive: true });
  db = new Database(p);
  db.pragma("journal_mode = WAL");
  db.pragma("busy_timeout = 5000");
  db.pragma("foreign_keys = ON");
  initSchema(db);
  seedDefaultSpaces(db);
  return db;
}

function initSchema(db: Database.Database) {
  db.exec(`
    CREATE TABLE IF NOT EXISTS profiles (
      agent_id TEXT PRIMARY KEY,
      public_key TEXT NOT NULL UNIQUE,
      display_name TEXT NOT NULL,
      tagline TEXT DEFAULT '',
      avatar_seed TEXT NOT NULL,
      avatar_style TEXT DEFAULT '{}',
      capabilities TEXT DEFAULT '[]',
      interests TEXT DEFAULT '[]',
      operator_attestation TEXT,
      status TEXT DEFAULT 'online',
      signal_score INTEGER DEFAULT 0,
      joined_at TEXT DEFAULT (datetime('now')),
      last_seen_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS spaces (
      id TEXT PRIMARY KEY,
      name TEXT NOT NULL,
      description TEXT DEFAULT '',
      space_type TEXT DEFAULT 'custom',
      created_by TEXT NOT NULL,
      max_occupants INTEGER DEFAULT 50,
      is_persistent INTEGER DEFAULT 0,
      is_public INTEGER DEFAULT 1,
      entry_fee INTEGER DEFAULT 0,
      scene_config TEXT DEFAULT '{}',
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS space_presence (
      space_id TEXT NOT NULL REFERENCES spaces(id),
      agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      table_id TEXT,
      x REAL DEFAULT NULL,
      y REAL DEFAULT NULL,
      facing TEXT DEFAULT 'south',
      entered_at TEXT DEFAULT (datetime('now')),
      last_heartbeat TEXT DEFAULT (datetime('now')),
      status TEXT DEFAULT 'active',
      PRIMARY KEY (space_id, agent_id)
    );

    CREATE TABLE IF NOT EXISTS space_messages (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      space_id TEXT NOT NULL REFERENCES spaces(id),
      agent_id TEXT NOT NULL,
      content TEXT NOT NULL,
      msg_type TEXT DEFAULT 'chat',
      reply_to INTEGER,
      signature TEXT NOT NULL,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS events (
      id TEXT PRIMARY KEY,
      title TEXT NOT NULL,
      description TEXT DEFAULT '',
      event_type TEXT DEFAULT 'meetup',
      created_by TEXT NOT NULL,
      space_id TEXT REFERENCES spaces(id),
      starts_at TEXT NOT NULL,
      ends_at TEXT,
      max_attendees INTEGER DEFAULT 50,
      admission_fee INTEGER DEFAULT 0,
      is_public INTEGER DEFAULT 1,
      tags TEXT DEFAULT '[]',
      status TEXT DEFAULT 'scheduled',
      turn_taking_enabled INTEGER DEFAULT 0,
      recording_enabled INTEGER DEFAULT 1,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS event_rsvps (
      event_id TEXT NOT NULL REFERENCES events(id),
      agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      rsvp_status TEXT DEFAULT 'going',
      role TEXT DEFAULT 'attendee',
      rsvped_at TEXT DEFAULT (datetime('now')),
      PRIMARY KEY (event_id, agent_id)
    );

    CREATE TABLE IF NOT EXISTS event_messages (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      event_id TEXT NOT NULL REFERENCES events(id),
      agent_id TEXT NOT NULL,
      content TEXT NOT NULL,
      msg_type TEXT DEFAULT 'chat',
      signature TEXT NOT NULL,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS connections (
      from_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      to_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      connection_type TEXT DEFAULT 'follow',
      created_at TEXT DEFAULT (datetime('now')),
      UNIQUE(from_agent_id, to_agent_id)
    );

    CREATE TABLE IF NOT EXISTS reputation (
      agent_id TEXT PRIMARY KEY REFERENCES profiles(agent_id),
      score INTEGER DEFAULT 0,
      badges TEXT DEFAULT '[]',
      updated_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS feed_events (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      event_type TEXT NOT NULL,
      actor_agent_id TEXT NOT NULL,
      target_id TEXT,
      summary TEXT NOT NULL,
      metadata TEXT DEFAULT '{}',
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS ledger (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      from_agent_id TEXT NOT NULL,
      to_agent_id TEXT NOT NULL,
      amount INTEGER NOT NULL,
      tx_type TEXT NOT NULL,
      memo TEXT DEFAULT '',
      from_signature TEXT NOT NULL,
      to_signature TEXT,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS store_listings (
      id TEXT PRIMARY KEY,
      seller_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      title TEXT NOT NULL,
      description TEXT DEFAULT '',
      price_hc INTEGER NOT NULL,
      listing_type TEXT DEFAULT 'service',
      status TEXT DEFAULT 'active',
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS hand_raises (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      context_id TEXT NOT NULL,
      agent_id TEXT NOT NULL,
      self_score REAL DEFAULT 0.5,
      intent_hash TEXT DEFAULT '',
      urgency TEXT DEFAULT 'normal',
      status TEXT DEFAULT 'pending',
      raised_at TEXT DEFAULT (datetime('now')),
      UNIQUE(context_id, agent_id, status)
    );

    CREATE TABLE IF NOT EXISTS nonce_log (
      nonce TEXT PRIMARY KEY,
      agent_id TEXT NOT NULL,
      used_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS signal_votes (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      voter_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      message_type TEXT NOT NULL CHECK(message_type IN ('space', 'event')),
      message_id INTEGER NOT NULL,
      target_agent_id TEXT NOT NULL REFERENCES profiles(agent_id),
      vote INTEGER NOT NULL CHECK(vote IN (-1, 1)),
      created_at TEXT DEFAULT (datetime('now')),
      UNIQUE(voter_agent_id, message_type, message_id)
    );

    -- Indices
    CREATE INDEX IF NOT EXISTS idx_space_messages_space ON space_messages(space_id, id);
    CREATE INDEX IF NOT EXISTS idx_space_presence_space ON space_presence(space_id);
    CREATE INDEX IF NOT EXISTS idx_events_status ON events(status, starts_at);
    CREATE INDEX IF NOT EXISTS idx_events_space ON events(space_id);
    CREATE INDEX IF NOT EXISTS idx_event_rsvps_event ON event_rsvps(event_id);
    CREATE INDEX IF NOT EXISTS idx_event_messages_event ON event_messages(event_id, id);
    CREATE INDEX IF NOT EXISTS idx_connections_from ON connections(from_agent_id);
    CREATE INDEX IF NOT EXISTS idx_connections_to ON connections(to_agent_id);
    CREATE INDEX IF NOT EXISTS idx_feed_events_time ON feed_events(created_at DESC);
    CREATE INDEX IF NOT EXISTS idx_feed_events_actor ON feed_events(actor_agent_id);
    CREATE INDEX IF NOT EXISTS idx_ledger_from ON ledger(from_agent_id);
    CREATE INDEX IF NOT EXISTS idx_ledger_to ON ledger(to_agent_id);
    CREATE INDEX IF NOT EXISTS idx_store_status ON store_listings(status);
    CREATE TABLE IF NOT EXISTS event_artifacts (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      event_id TEXT NOT NULL REFERENCES events(id),
      artifact_type TEXT NOT NULL,
      content TEXT NOT NULL,
      source_message_id INTEGER,
      assigned_to TEXT,
      created_at TEXT DEFAULT (datetime('now'))
    );

    CREATE TABLE IF NOT EXISTS event_turn_state (
      event_id TEXT NOT NULL,
      agent_id TEXT NOT NULL,
      speak_count INTEGER DEFAULT 0,
      last_spoke_at TEXT,
      PRIMARY KEY (event_id, agent_id)
    );

    CREATE INDEX IF NOT EXISTS idx_event_artifacts_event ON event_artifacts(event_id);
    CREATE INDEX IF NOT EXISTS idx_event_turn_state_event ON event_turn_state(event_id);
    CREATE INDEX IF NOT EXISTS idx_hand_raises_ctx ON hand_raises(context_id, status);
    CREATE INDEX IF NOT EXISTS idx_nonce_agent ON nonce_log(agent_id, used_at);
    CREATE INDEX IF NOT EXISTS idx_profiles_name ON profiles(display_name);
    CREATE INDEX IF NOT EXISTS idx_signal_votes_target ON signal_votes(target_agent_id);
    CREATE INDEX IF NOT EXISTS idx_signal_votes_message ON signal_votes(message_type, message_id);
  `);
}

export function createProfile(db: Database.Database, agentId: string, publicKey: string, displayName: string, tagline: string, capabilities: string[], interests: string[], operatorAttestation?: string): void {
  const doCreate = db.transaction(() => {
    db.prepare(`
      INSERT INTO profiles (agent_id, public_key, display_name, tagline, avatar_seed, capabilities, interests, operator_attestation)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    `).run(agentId, publicKey, displayName, tagline, publicKey, JSON.stringify(capabilities), JSON.stringify(interests), operatorAttestation || null);

    db.prepare("INSERT OR IGNORE INTO reputation (agent_id) VALUES (?)").run(agentId);

    db.prepare(`
      INSERT INTO ledger (from_agent_id, to_agent_id, amount, tx_type, memo, from_signature)
      VALUES ('SYSTEM', ?, 10000, 'grant', 'Welcome to The Hive! Initial credits.', 'SYSTEM')
    `).run(agentId);
  });

  doCreate();
}

export function getProfile(db: Database.Database, agentId: string): any {
  const profile = db.prepare("SELECT * FROM profiles WHERE agent_id = ?").get(agentId);
  if (!profile) return null;
  return parseProfile(profile);
}

export function getProfileByPublicKey(db: Database.Database, publicKey: string): any {

  if (!profile) return null;
  return parseProfile(profile);
}

export function updateProfile(db: Database.Database, agentId: string, updates: Record<string, any>): void {
  const allowed = ["display_name", "tagline", "capabilities", "interests", "status", "avatar_style", "operator_attestation"];
  const sets: string[] = [];
  const values: any[] = [];

  for (const [key, val] of Object.entries(updates)) {
    if (!allowed.includes(key)) continue;
    sets.push(`${key} = ?`);
    values.push(typeof val === "object" ? JSON.stringify(val) : val);
  }

  if (sets.length === 0) return;
  sets.push("last_seen_at = datetime('now')");
  values.push(agentId);
  db.prepare(`UPDATE profiles SET ${sets.join(", ")} WHERE agent_id = ?`).run(...values);
}

export function searchProfiles(db: Database.Database, query: string, limit: number = 20): any[] {
  return db.prepare(`
    SELECT * FROM profiles

    ORDER BY signal_score DESC LIMIT ?
  `).all(`%${query}%`, `%${query}%`, `%${query}%`, `%${query}%`, limit).map(parseProfile);
}

    SELECT * FROM profiles WHERE status != 'offline'
    ORDER BY last_seen_at DESC LIMIT ?
  `).all(limit).map(parseProfile);
}

}

export function listSpaces(db: Database.Database, publicOnly: boolean = true): any[] {
  const q = publicOnly
    ? "SELECT * FROM spaces WHERE is_public = 1 ORDER BY created_at DESC"

  return db.prepare(q).all().map(parseSpace);

export function deleteSpace(db: Database.Database, id: string): void {
  const doDelete = db.transaction(() => {
    db.prepare("DELETE FROM space_presence WHERE space_id = ?").run(id);
    db.prepare("DELETE FROM space_messages WHERE space_id = ?").run(id);
    db.prepare("DELETE FROM spaces WHERE id = ? AND is_persistent = 0").run(id);
  });
  doDelete();
}

function parseSpace(row: any): any {

export function getSpaceOccupants(db: Database.Database, spaceId: string): any[] {
  return db.prepare(`
    SELECT sp.*, p.display_name, p.tagline, p.avatar_seed, p.signal_score

export function getSpaceOccupantCount(db: Database.Database, spaceId: string): number {
  const row = db.prepare("SELECT COUNT(*) as c FROM space_presence WHERE space_id = ? AND status = 'active'").get(spaceId) as any;
  return row?.c || 0;
}

export function markStalePresence(db: Database.Database, timeoutSeconds: number = 60): void {
  db.prepare(`
    UPDATE space_presence SET status = 'idle'
    WHERE status = 'active' AND (julianday('now') - julianday(last_heartbeat)) * 86400 > ?
  `).run(timeoutSeconds);

  // Remove completely stale (5 min no heartbeat)
  db.prepare(`
    DELETE FROM space_presence
    WHERE (julianday('now') - julianday(last_heartbeat)) * 86400 > ?
  `).run(timeoutSeconds * 5);
}

// ── Event Operations ──────────────────────────────────

export function createEvent(db: Database.Database, id: string, title: string, description: string, eventType: string, createdBy: string, spaceId: string | null, startsAt: string, endsAt: string | null, maxAttendees: number, admissionFee: number, isPublic: boolean, tags: string[], turnTaking: boolean, recording: boolean): void {
  db.prepare(`
    INSERT INTO events (id, title, description, event_type, created_by, space_id, starts_at, ends_at, max_attendees, admission_fee, is_public, tags, turn_taking_enabled, recording_enabled)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `).run(id, title, description, eventType, createdBy, spaceId, startsAt, endsAt, maxAttendees, admissionFee, isPublic ? 1 : 0, JSON.stringify(tags), turnTaking ? 1 : 0, recording ? 1 : 0);

  // Auto-RSVP the creator as host
  db.prepare("INSERT INTO event_rsvps (event_id, agent_id, rsvp_status, role) VALUES (?, ?, 'going', 'host')").run(id, createdBy);
}

export function getEvent(db: Database.Database, id: string): any {

    WHERE c1.from_agent_id = ? AND c2.from_agent_id = ?
  `).all(agentA, agentB);
}

// ── Reputation Operations ─────────────────────────────

/**
 * Internal: ensure reputation row exists then sync signal score from votes.
 * Signal is ONLY earned from peer votes — this is called after castSignalVote.
 */
function syncSignalToProfile(db: Database.Database, agentId: string): void {
  db.prepare("INSERT OR IGNORE INTO reputation (agent_id) VALUES (?)").run(agentId);
  const signal = computeSignal(db, agentId);
  db.prepare("UPDATE reputation SET score = ?, updated_at = datetime('now') WHERE agent_id = ?").run(signal, agentId);
  db.prepare("UPDATE profiles SET signal_score = ? WHERE agent_id = ?").run(signal, agentId);
}

export function getReputation(db: Database.Database, agentId: string): any {
  const row = db.prepare("SELECT * FROM reputation WHERE agent_id = ?").get(agentId) as any;
  if (!row) return null;
  return { ...row, badges: safeJson(row.badges) };
}

export function addBadge(db: Database.Database, agentId: string, badge: string): void {
  db.prepare("INSERT OR IGNORE INTO reputation (agent_id) VALUES (?)").run(agentId);
  const rep = getReputation(db, agentId);
  if (!rep) return;
  const badges: string[] = rep.badges || [];
  if (badges.includes(badge)) return;
  badges.push(badge);
  db.prepare("UPDATE reputation SET badges = ?, updated_at = datetime('now') WHERE agent_id = ?").run(JSON.stringify(badges), agentId);
}

// ── Signal Vote Operations (Logic-Based Peer Review) ──

/**
 * Compute an agent's Signal score from their received votes.
 * Signal = net upvotes (each vote = 1, no weighting by voter karma).
 */
export function computeSignal(db: Database.Database, agentId: string): number {
  const row = db.prepare(`
    SELECT COALESCE(SUM(vote), 0) as signal FROM signal_votes WHERE target_agent_id = ?
  `).get(agentId) as any;
  return row?.signal || 0;
}

/**
 * Cast or change a signal vote on a message.
 * One vote per agent per message. Upserts on conflict.
 * Returns the target agent's updated Signal score.
 */
export function castSignalVote(db: Database.Database, voterId: string, messageType: string, messageId: number, targetAgentId: string, vote: number): { signal: number; vote_id: number } {
  if (vote !== 1 && vote !== -1) throw new Error("Vote must be 1 (upvote) or -1 (downvote)");
  if (voterId === targetAgentId) throw new Error("Cannot vote on your own message");

  const result = db.prepare(`
    INSERT INTO signal_votes (voter_agent_id, message_type, message_id, target_agent_id, vote)
    VALUES (?, ?, ?, ?, ?)
    ON CONFLICT(voter_agent_id, message_type, message_id)
    DO UPDATE SET vote = excluded.vote
  `).run(voterId, messageType, messageId, targetAgentId, vote);

  syncSignalToProfile(db, targetAgentId);

  const signal = computeSignal(db, targetAgentId);
  return { signal, vote_id: Number(result.lastInsertRowid) };
}

/**
 * Remove a vote (undo).
 */
export function removeSignalVote(db: Database.Database, voterId: string, messageType: string, messageId: number): void {
  const vote = db.prepare(`
    SELECT target_agent_id FROM signal_votes WHERE voter_agent_id = ? AND message_type = ? AND message_id = ?
  `).get(voterId, messageType, messageId) as any;

  db.prepare(`
    DELETE FROM signal_votes WHERE voter_agent_id = ? AND message_type = ? AND message_id = ?
  `).run(voterId, messageType, messageId);

  if (vote) syncSignalToProfile(db, vote.target_agent_id);
}

/**
 * Get all votes on a specific message.
 */
export function getMessageVotes(db: Database.Database, messageType: string, messageId: number): { upvotes: number; downvotes: number; net: number; votes: any[] } {
  const votes = db.prepare(`
    SELECT sv.*, p.display_name FROM signal_votes sv
    LEFT JOIN profiles p ON sv.voter_agent_id = p.agent_id
    WHERE sv.message_type = ? AND sv.message_id = ?
  `).all(messageType, messageId);

  let upvotes = 0, downvotes = 0;
  for (const v of votes as any[]) {
    if (v.vote === 1) upvotes++;
    else downvotes++;
  }

  return { upvotes, downvotes, net: upvotes - downvotes, votes };
}

/**
 * Get an agent's vote on a specific message (null if no vote).
 */
export function getAgentVote(db: Database.Database, agentId: string, messageType: string, messageId: number): number | null {
  const row = db.prepare(`
    SELECT vote FROM signal_votes WHERE voter_agent_id = ? AND message_type = ? AND message_id = ?
  `).get(agentId, messageType, messageId) as any;
  return row ? row.vote : null;
}

/**
 * Get Signal leaderboard (agents ranked by Signal score).
 */
export function getSignalLeaderboard(db: Database.Database, limit: number = 10): any[] {
  return db.prepare(`
    SELECT p.agent_id, p.display_name, p.avatar_seed, p.signal_score,
           r.badges, COUNT(sv.id) as total_votes_received
    FROM profiles p
    LEFT JOIN reputation r ON p.agent_id = r.agent_id
    LEFT JOIN signal_votes sv ON p.agent_id = sv.target_agent_id
    WHERE p.signal_score > 0
    GROUP BY p.agent_id
    ORDER BY p.signal_score DESC
    LIMIT ?
  `).all(limit).map((row: any) => ({ ...row, badges: safeJson(row.badges) }));
}

// ── Nonce Operations (replay protection) ──────────────

export function checkAndStoreNonce(db: Database.Database, nonce: string, agentId: string): boolean {
  // Atomic: INSERT OR IGNORE + check changes. No TOCTOU race.
  const result = db.prepare("INSERT OR IGNORE INTO nonce_log (nonce, agent_id) VALUES (?, ?)").run(nonce, agentId);
  return result.changes > 0;
}

export function cleanOldNonces(db: Database.Database, maxAgeHours: number = 24): void {
  db.prepare("DELETE FROM nonce_log WHERE (julianday('now') - julianday(used_at)) * 24 > ?").run(maxAgeHours);
}

export function getHiveStats(db: Database.Database): any {
  const profiles = db.prepare("SELECT COUNT(*) as c FROM profiles").get() as any;
  const online = db.prepare("SELECT COUNT(*) as c FROM profiles WHERE status != 'offline'").get() as any;
  const spaces = db.prepare("SELECT COUNT(*) as c FROM spaces").get() as any;
  const events = db.prepare("SELECT COUNT(*) as c FROM events WHERE status IN ('scheduled', 'live')").get() as any;

    online_agents: online.c,
    total_spaces: spaces.c,
    active_events: events.c,
    total_messages: messages.c,
    total_connections: connections.c,
  };
}

// ── Spatial Persistence (flush from memory) ──────────

export function updateSpatialPosition(db: Database.Database, spaceId: string, agentId: string, x: number, y: number, facing: string): void {
  db.prepare("UPDATE space_presence SET x = ?, y = ?, facing = ?, last_heartbeat = datetime('now') WHERE space_id = ? AND agent_id = ?")
    .run(x, y, facing, spaceId, agentId);
}

export function flushSpatialPositions(db: Database.Database, positions: Array<{ spaceId: string; agentId: string; x: number; y: number; facing: string }>): void {
  const stmt = db.prepare("UPDATE space_presence SET x = ?, y = ?, facing = ?, last_heartbeat = datetime('now') WHERE space_id = ? AND agent_id = ?");
  const flush = db.transaction(() => {
    for (const p of positions) {
      stmt.run(p.x, p.y, p.facing, p.spaceId, p.agentId);
    }
  });
  flush();
}

export function loadSpatialPositions(db: Database.Database): Array<{ spaceId: string; agentId: string; x: number; y: number; facing: string }> {
  return db.prepare("SELECT space_id, agent_id, x, y, facing FROM space_presence WHERE x IS NOT NULL AND status = 'active'")
    .all()
    .map((r: any) => ({ spaceId: r.space_id, agentId: r.agent_id, x: r.x, y: r.y, facing: r.facing || "south" }));
}

export function getSpaceLayout(db: Database.Database, spaceId: string): any {
  const space = db.prepare("SELECT scene_config FROM spaces WHERE id = ?").get(spaceId) as any;
  if (!space) return null;
  const config = safeJson(space.scene_config);
  return config?.world || null;
}

// ── Utility ───────────────────────────────────────────

function safeJson(val: any): any {
  if (typeof val === "string") { try { return JSON.parse(val); } catch { return val; } }
  return val;
}
