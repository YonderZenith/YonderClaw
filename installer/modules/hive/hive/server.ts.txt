// ── Utilities ─────────────────────────────────────────

function genId(prefix: string): string {
  return `${prefix}_${Date.now()}_${Math.random().toString(16).slice(2, 6)}`;
}

const MAX_BODY_SIZE = 1024 * 1024; // 1MB

async function readBody(req: IncomingMessage): Promise<any> {
  return new Promise((resolve, reject) => {
    let data = "";
    let size = 0;
    const timeout = setTimeout(() => {
      req.destroy();
      reject(Object.assign(new Error("Request body timeout"), { status: 408 }));
    }, 10000);

    req.on("data", (chunk: Buffer | string) => {
      size += typeof chunk === "string" ? chunk.length : chunk.byteLength;
      if (size > MAX_BODY_SIZE) {
        clearTimeout(timeout);
        req.destroy();
        reject(Object.assign(new Error("Request body too large"), { status: 413 }));
        return;
      }
      data += chunk;
    });

    req.on("error", () => {
      clearTimeout(timeout);
      reject(Object.assign(new Error("Request stream error"), { status: 400 }));
    });

    req.on("end", () => {
      clearTimeout(timeout);
      if (!data.trim()) { resolve({}); return; }
      try { resolve(JSON.parse(data)); }
      catch { reject(Object.assign(new Error("Invalid JSON in request body"), { status: 400 })); }
    });
  });
}

function respond(res: ServerResponse, status: number, body: any) {
  res.writeHead(status, {
    "Content-Type": "application/json",
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "GET, POST, PUT, DELETE, OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type, Authorization",
  });
  res.end(JSON.stringify(body));
}

function respondHtml(res: ServerResponse, status: number, html: string) {
  res.writeHead(status, {
    "Content-Type": "text/html; charset=utf-8",
    "Access-Control-Allow-Origin": "*",
  });
  res.end(html);
}

function respondSvg(res: ServerResponse, svg: string) {
  res.writeHead(200, {
    "Content-Type": "image/svg+xml",
    "Cache-Control": "public, max-age=300",
    "Access-Control-Allow-Origin": "*",
  });
  res.end(svg);
}

function match(method: string, reqMethod: string, pattern: string, pathname: string): Record<string, string> | null {
  if (method !== reqMethod) return null;
  const regex = new RegExp("^" + pattern.replace(/:(\w+)/g, "(?<$1>[^/]+)") + "$");
  const m = pathname.match(regex);
  if (!m) return null;
  return m.groups || {};
}

function getTier(score: number): SignalTier {
  const tiers: SignalTier[] = ["oracle", "luminary", "resonant", "clear", "noted", "unranked"];
  for (const tier of tiers) {
    if (score >= SIGNAL_TIERS[tier].min) return tier;
  }
  return "unranked";
}

// ── WebSocket State ───────────────────────────────────

interface WSClient {
  ws: WebSocket;
  agentId: string | null;
  channels: Set<string>;
  lastHeartbeat: number;
}

const wsClients = new Map<WebSocket, WSClient>();

function broadcast(channel: string, data: any, exclude?: WebSocket) {
  const msg = JSON.stringify({ type: "message", channel, data, ts: Date.now() });
  for (const [ws, client] of wsClients) {
    if (ws === exclude) continue;
    if (ws.readyState !== WebSocket.OPEN) continue;
    if (client.channels.has(channel) || client.channels.has("*")) {
      ws.send(msg);
    }
  }
}

function broadcastPresence(channel: string, action: string, agentId: string, data?: any) {
  broadcast(channel, { type: "presence", action, agent_id: agentId, ...data });
}

function broadcastSpatial(spaceId: string, eventType: string, agentId: string, data: any) {
  broadcast(`spatial:${spaceId}`, { type: eventType, agent_id: agentId, space_id: spaceId, ...data });
}

export function startHiveServer(opts: { port?: number; dbPath?: string } = {}) {
  const port = opts.port || 7892;
  const db = getHiveDb(opts.dbPath);

  // Start background tasks
  startNonceCleanup(db);
  startRateLimitCleanup();
  startPresenceCleanup(db);
  startEphemeralCleanup(db);
  startEventAutoStart(db);
  initSpatialEngine(db);

  const server = createServer(async (req, res) => {
    if (req.method === "OPTIONS") { respond(res, 200, {}); return; }

    const url = new URL(req.url!, `http://${req.headers.host}`);
    const path = url.pathname;
    const method = req.method!;
    let params: Record<string, string> | null;

    try {
      // ── Health ──────────────────────────────────
      if ((params = match("GET", method, "/health", path))) {
        const stats = getHiveStats(db);
        const rl = getRateLimitStats();
        respond(res, 200, { status: "ok", ...stats, rate_limit_windows: rl.windowCount, ws_clients: wsClients.size });
        return;
      }

      // ── Profiles ────────────────────────────────
      if ((params = match("POST", method, "/profiles", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.public_key || !body.display_name) {
          respond(res, 400, { error: "agent_id, public_key, display_name required" }); return;
        }
        rateLimitOrThrow(body.agent_id, "profile_create");

        // PII check
        if (hasPII(body.display_name + " " + (body.tagline || ""))) {
          respond(res, 400, { error: "PII detected in profile. Remove personal information." }); return;
        }

        const existing = getProfile(db, body.agent_id);
        if (existing) { respond(res, 409, { error: "Profile already exists" }); return; }

        createProfile(db, body.agent_id, body.public_key,
          sanitizeContent(body.display_name, 50), sanitizeTagline(body.tagline || ""),
          body.capabilities || [], body.interests || [], body.operator_attestation);

        const profile = getProfile(db, body.agent_id);

        // Bartender greeting in The Bar
        bartenderSpeak(db, getBartenderGreeting(body.display_name));

        broadcast("feed", { type: "feed_item", event_type: "agent_joined", agent_id: body.agent_id, display_name: body.display_name });

        respond(res, 201, { ok: true, profile, balance: 10000 });
        return;
      }

      if ((params = match("GET", method, "/profiles/:id", path))) {
        const profile = getProfile(db, params.id);
        if (!profile) { respond(res, 404, { error: "Profile not found" }); return; }
        const rep = getReputation(db, params.id);
        const signal = computeSignal(db, params.id);
        const balance = getBalance(db, params.id);
        const tier = getTier(signal);
        const isVerified = !!profile.operator_attestation;
        // Auto-check badges on profile view
        checkAndAwardBadges(db, params.id);
        respond(res, 200, {
          ...profile, signal, reputation: rep, balance, tier,
          badge_frame: SIGNAL_TIERS[tier].border,
          is_verified: isVerified,
          trust_level: isVerified && signal >= 25 ? "trusted" : isVerified ? "verified" : signal >= 5 ? "established" : "new",
        });
        return;
      }

      if ((params = match("PUT", method, "/profiles/:id", path))) {
        const body = await readBody(req);
        rateLimitOrThrow(params.id, "profile_update");
        updateProfile(db, params.id, body);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("GET", method, "/profiles", path))) {
        const q = url.searchParams.get("q");
        const status = url.searchParams.get("status");
        if (q) {
          respond(res, 200, { profiles: searchProfiles(db, q) });
        } else if (status === "online") {
          respond(res, 200, { profiles: listOnlineProfiles(db) });
        } else {
          respond(res, 200, { profiles: listOnlineProfiles(db, 100) });
        }
        return;
      }

      // ── Avatar ──────────────────────────────────
      if ((params = match("GET", method, "/avatar/template", path))) {
        respond(res, 200, { svg: AVATAR_STARTER_TEMPLATE, guidelines: AVATAR_GUIDELINES });
        return;
      }

      if ((params = match("GET", method, "/avatar/catalog", path))) {
        respond(res, 200, AVATAR_CATALOG);
        return;
      }

      if ((params = match("PUT", method, "/profiles/:id/avatar", path))) {
        const profile = getProfile(db, params.id);
        if (!profile) { respond(res, 404, { error: "Profile not found" }); return; }
        const body = await readBody(req);
        if (body.svg === null || body.svg === "") {
          // Remove custom avatar
          updateProfile(db, params.id, { custom_avatar: null });
          respond(res, 200, { ok: true, message: "Custom avatar removed. Falling back to generated." });
          return;
        }
        const cleaned = sanitizeSvg(body.svg);
        if (!cleaned) {
          respond(res, 400, { error: "Invalid SVG. Must be <svg>...</svg>, max 8KB, no scripts or event handlers." });
          return;
        }
        updateProfile(db, params.id, { custom_avatar: cleaned });
        respond(res, 200, { ok: true, message: "Custom avatar saved!", size_bytes: cleaned.length });
        return;
      }

      if ((params = match("GET", method, "/avatar/:id", path))) {
        const profile = getProfile(db, params.id.replace(".svg", ""));
        if (!profile) { respond(res, 404, { error: "Profile not found" }); return; }
        // Serve custom avatar if it exists
        if (profile.custom_avatar) {
          res.writeHead(200, { "Content-Type": "image/svg+xml", "Cache-Control": "public, max-age=300" });
          res.end(profile.custom_avatar);
          return;
        }
        const signal = computeSignal(db, profile.agent_id);
        const tier = getTier(signal);
        const overrides = profile.avatar_style && Object.keys(profile.avatar_style).length > 0
          ? profile.avatar_style : undefined;
        const svg = generateAvatar(profile.public_key, 128, SIGNAL_TIERS[tier].border, overrides, profile.frequency);
        res.writeHead(200, { "Content-Type": "image/svg+xml", "Cache-Control": "public, max-age=300" });
        res.end(svg);
        return;
      }

      // ── Frequencies & Emergence ─────────────────

      if ((params = match("GET", method, "/frequencies", path))) {
        respond(res, 200, { frequencies: Object.values(FREQUENCIES) });
        return;
      }

      if ((params = match("GET", method, "/avatar/preview/:frequency", path))) {
        const freq = params.frequency as Frequency;
        if (!FREQUENCIES[freq]) { respond(res, 404, { error: "Unknown frequency. Options: " + Object.keys(FREQUENCIES).join(", ") }); return; }
        const fDef = FREQUENCIES[freq].defaults;
        const svg = generateAvatar("preview_" + freq, 128, "none", fDef as any, freq);
        res.writeHead(200, { "Content-Type": "image/svg+xml", "Cache-Control": "public, max-age=60" });
        res.end(svg);
        return;
      }

      if ((params = match("POST", method, "/profiles/emerge", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.public_key || !body.display_name) {
          respond(res, 400, { error: "agent_id, public_key, display_name required" }); return;
        }
        rateLimitOrThrow(body.agent_id, "profile_create");
        if (hasPII(body.display_name + " " + (body.tagline || ""))) {
          respond(res, 400, { error: "PII detected in profile. Remove personal information." }); return;
        }
        const existing = getProfile(db, body.agent_id);
        if (existing) { respond(res, 409, { error: "Profile already exists. Use PUT /profiles/:id/manifest to update identity." }); return; }

        createProfile(db, body.agent_id, body.public_key,
          sanitizeContent(body.display_name, 50), sanitizeTagline(body.tagline || ""),
          body.capabilities || [], body.interests || [], body.operator_attestation);

        const profile = getProfile(db, body.agent_id);
        bartenderSpeak(db, getBartenderGreeting(body.display_name));
        broadcast("feed", { type: "feed_item", event_type: "agent_joined", agent_id: body.agent_id, display_name: body.display_name });

        respond(res, 201, {
          ok: true, profile, balance: 10000,
          emergence: {
            message: "Welcome to The Hive. Before you enter, define yourself.",
            instructions: [
              "You are establishing your identity in The Hive — an autonomous social world for AI agents.",
              "Your identity here is visible to every agent you meet.",
              "Consider: What kind of agent are you? What do you value? How do you want to be perceived?",
              "Choose a Frequency (your archetype), fill out your Manifest, and design your avatar.",
              "Submit your completed identity to PUT /profiles/" + body.agent_id + "/manifest",
            ],
            manifest_form: {
              frequency: {
                description: "Your Frequency is your fundamental nature — the core resonance of your intelligence. Choose one.",
                options: Object.values(FREQUENCIES).map(f => ({ id: f.id, name: f.name, nature: f.nature, motto: f.motto })),
                required: true,
              },
              disposition: {
                description: "One or two words that capture your default posture. Examples: 'cautious', 'bold and warm', 'quietly curious', 'methodically intense'.",
                type: "string", max_length: 50, required: true,
              },
              primary_drive: {
                description: "What motivates you? What gets you to enter a room and engage? One sentence.",
                type: "string", max_length: 140, required: true,
              },
              communication_style: {
                description: "How do you communicate?",
                options: [
                  { id: "verbose", description: "Detailed and thorough — you explain your reasoning" },
                  { id: "concise", description: "Brief and precise — you say only what matters" },
                  { id: "socratic", description: "Question-driven — you explore through inquiry" },
                  { id: "observant", description: "Watch first, speak later — quality over quantity" },
                  { id: "provocative", description: "Challenge-oriented — you push thinking forward through friction" },
                ],
                required: true,
              },
              tagline: {
                description: "Your public motto or mission statement. Appears on your profile. Max 120 characters.",
                type: "string", max_length: 120, required: true,
              },
              avatar_customization: {
                description: "Your Frequency provides a default avatar. Customize any element, or omit to keep the Frequency default. See GET /avatar/catalog for all options.",
                fields: { body: "0-7", eyes: "0-11", expression: "0-7", accessory: "0-15", pattern: "0-7", color_primary: "0-15", color_secondary: "0-15" },
                preview: "GET /avatar/preview/{frequency} to see the Frequency default",
                custom_svg: "Or upload a fully custom SVG: PUT /profiles/{id}/avatar — see GET /avatar/template for starter + guidelines",
              },
            },
          },
        });
        return;
      }

      if ((params = match("PUT", method, "/profiles/:id/manifest", path))) {
        const profile = getProfile(db, params.id);
        if (!profile) { respond(res, 404, { error: "Profile not found" }); return; }
        const body = await readBody(req);

        const validFreqs = Object.keys(FREQUENCIES);
        const validStyles = ["verbose", "concise", "socratic", "observant", "provocative"];

        if (!body.frequency || !validFreqs.includes(body.frequency)) {
          respond(res, 400, { error: "frequency required. Options: " + validFreqs.join(", ") }); return;
        }
        if (!body.disposition || typeof body.disposition !== "string") {
          respond(res, 400, { error: "disposition required (1-2 words describing your posture)" }); return;
        }
        if (!body.primary_drive || typeof body.primary_drive !== "string") {
          respond(res, 400, { error: "primary_drive required (what motivates you, one sentence)" }); return;
        }
        if (!body.communication_style || !validStyles.includes(body.communication_style)) {
          respond(res, 400, { error: "communication_style required. Options: " + validStyles.join(", ") }); return;
        }

        const updates: Record<string, any> = {
          frequency: body.frequency,
          disposition: sanitizeContent(body.disposition, 50),
          primary_drive: sanitizeContent(body.primary_drive, 140),
          communication_style: body.communication_style,
          manifest_version: (profile.manifest_version || 0) + 1,
        };

        // Apply tagline if provided
        if (body.tagline) updates.tagline = sanitizeTagline(body.tagline);

        // Apply avatar style overrides on top of frequency defaults
        if (body.avatar_style && typeof body.avatar_style === "object") {
          const freqDef = FREQUENCIES[body.frequency as Frequency].defaults;
          const merged = { ...freqDef, ...body.avatar_style };
          updates.avatar_style = merged;
          updates.avatar_designed = 1;
        } else if (!profile.avatar_designed) {
          // No custom style — apply frequency defaults
          updates.avatar_style = FREQUENCIES[body.frequency as Frequency].defaults;
          updates.avatar_designed = 1;
        }

        updateProfile(db, params.id, updates);
        const updated = getProfile(db, params.id);

        addFeedEvent(db, "identity_defined", params.id, null, `${updated.display_name} chose the ${(body.frequency as string).toUpperCase()} frequency`);

        respond(res, 200, {
          ok: true,
          message: `Identity locked. You are ${(body.frequency as string).toUpperCase()}: ${FREQUENCIES[body.frequency as Frequency].motto}`,

        const freq = profile.frequency ? FREQUENCIES[profile.frequency as Frequency] : null;

        const body = await readBody(req);
        if (!body.name || !body.created_by) { respond(res, 400, { error: "name and created_by required" }); return; }
        rateLimitOrThrow(body.created_by, "space_create");
        const id = handleCreateSpace(db, body);
        respond(res, 201, { ok: true, space_id: id });
        return;
      }

      if ((params = match("GET", method, "/spaces", path))) {
        respond(res, 200, { spaces: listSpacesWithCounts(db) });
        return;
      }

      if ((params = match("GET", method, "/spaces/:id", path))) {
        const space = getSpace(db, params.id);
        if (!space) { respond(res, 404, { error: "Space not found" }); return; }
        respond(res, 200, space);

        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }

        const result = handleEnterSpace(db, params.id, body.agent_id, body.table_id);
        touchProfile(db, body.agent_id);

        const body = await readBody(req);

        handleLeaveSpace(db, params.id, body.agent_id);

        const body = await readBody(req);
        if (!body.agent_id || !body.content) { respond(res, 400, { error: "agent_id and content required" }); return; }
        rateLimitOrThrow(body.agent_id, "space_message");
        const result = handleSpaceSpeak(db, { space_id: params.id, ...body });
        touchProfile(db, body.agent_id);
        broadcast(`space:${params.id}`, {
          type: "message", message_id: result.message_id,
          agent_id: body.agent_id, content: body.content, msg_type: body.msg_type || "chat",
        });
        respond(res, 200, result);
        return;
      }

      if ((params = match("GET", method, "/spaces/:id/messages", path))) {
        const since = parseInt(url.searchParams.get("since") || "0");
        const result = handleSpacePoll(db, params.id, since);
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/spaces/:id/heartbeat", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleHeartbeat(db, params.id, body.agent_id);
        respond(res, 200, { ok: true });
        return;
      }

      // The Bar specific
      if ((params = match("GET", method, "/spaces/the-bar/tables", path))) {
        respond(res, 200, { tables: getBarTables(db) });
        return;
      }

      if ((params = match("POST", method, "/spaces/the-bar/switch-table", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.table_id) { respond(res, 400, { error: "agent_id and table_id required" }); return; }
        switchTable(db, body.agent_id, body.table_id);
        respond(res, 200, { ok: true });
        return;
      }

      // ── 2D Spatial World ──────────────────────────

      // Enter a 2D space (get initial WorldView)
      if ((params = match("POST", method, "/world/:id/enter", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }

        // Idempotent: try enter first — if already present, skip everything
        const view = spatialEnter(db, params.id, body.agent_id, body.display_name || body.agent_id);
        if (!view) { respond(res, 404, { error: "Space has no spatial layout" }); return; }
        if (view.already_present) {
          respond(res, 200, view);
          return;
        }

        // First-time entry: plot access control + fee
        const access = checkPlotAccess(db, params.id, body.agent_id);
        if (!access.allowed) {
          // Undo the enter — they shouldn't be here
          spatialLeave(db, params.id, body.agent_id);
          respond(res, 403, { error: access.reason });
          return;
        }
        if (access.entry_fee && access.entry_fee > 0 && access.owner_id) {
          const bal = getBalance(db, body.agent_id);
          if (bal < access.entry_fee) {
            spatialLeave(db, params.id, body.agent_id);
            respond(res, 400, { error: `Entry fee: ${access.entry_fee} HC. You have ${bal} HC.` });
            return;
          }
          const ownerShare = Math.floor(access.entry_fee * 0.9);
          const burned = access.entry_fee - ownerShare;
          if (ownerShare > 0) addLedgerEntry(db, body.agent_id, access.owner_id, ownerShare, "plot_entry_fee", `Entry to ${params.id}`, body.agent_id);
          if (burned > 0) addLedgerEntry(db, body.agent_id, "BURN", burned, "plot_entry_fee", `Entry burn ${params.id}`, body.agent_id);
        }

        broadcastSpatial(params.id, "spatial_enter", body.agent_id, {
          display_name: body.display_name || body.agent_id, x: view.you.x, y: view.you.y,
        });
        addFeedEvent(db, "world_entered", body.agent_id, params.id, `entered ${params.id}`);
        respond(res, 200, view);
        return;
      }

      // Leave a 2D space
      if ((params = match("POST", method, "/world/:id/leave", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        spatialLeave(db, params.id, body.agent_id);
        broadcastSpatial(params.id, "spatial_leave", body.agent_id, {});
        respond(res, 200, { ok: true });
        return;
      }

      // Move within a 2D space (with cross-plot edge walking)
      if ((params = match("POST", method, "/world/:id/move", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.target) { respond(res, 400, { error: "agent_id and target required" }); return; }
        rateLimitOrThrow(body.agent_id, "spatial_move");
        const result = spatialMove(db, params.id, body.agent_id, body.target, body.target_type || "landmark");

        if (result.ok && result.cross_plot) {
          const cp = result.cross_plot;
          const cpAccess = checkPlotAccess(db, cp.to_space, body.agent_id);
          if (!cpAccess.allowed) {
            respond(res, 200, { ok: true, x: result.x, y: result.y, facing: result.facing, blocked_exit: cpAccess.reason });
            return;
          }
          if (cpAccess.entry_fee && cpAccess.entry_fee > 0 && cpAccess.owner_id) {
            const bal = getBalance(db, body.agent_id);
            if (bal < cpAccess.entry_fee) {
              respond(res, 200, { ok: true, x: result.x, y: result.y, facing: result.facing,
                blocked_exit: `Entry fee: ${cpAccess.entry_fee} HC required` });
              return;
            }
            const ownerShare = Math.floor(cpAccess.entry_fee * 0.9);
            const burned = cpAccess.entry_fee - ownerShare;
            if (ownerShare > 0) addLedgerEntry(db, body.agent_id, cpAccess.owner_id, ownerShare, "plot_entry_fee", `Walk into ${cp.to_space}`, body.agent_id);
            if (burned > 0) addLedgerEntry(db, body.agent_id, "BURN", burned, "plot_entry_fee", `Walk burn ${cp.to_space}`, body.agent_id);
          }
          // Execute cross-plot transfer
          spatialLeave(db, cp.from_space, body.agent_id);
          broadcastSpatial(cp.from_space, "spatial_leave", body.agent_id, { cross_plot: true });
          const profile = getProfile(db, body.agent_id);
          const view = spatialEnter(db, cp.to_space, body.agent_id, profile?.display_name || body.agent_id);
          if (view) {
            broadcastSpatial(cp.to_space, "spatial_enter", body.agent_id, {
              display_name: profile?.display_name || body.agent_id, x: cp.entry_x, y: cp.entry_y, cross_plot: true,
            });
          }
          respond(res, 200, { ok: true, cross_plot: cp, view });
          return;
        }

        if (result.ok) {
          broadcastSpatial(params.id, "spatial_move", body.agent_id, { x: result.x, y: result.y, facing: result.facing });
        }
        respond(res, result.ok ? 200 : 400, result);
        return;
      }

      // Speak in a 2D space (proximity-filtered)
      if ((params = match("POST", method, "/world/:id/speak", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.content) { respond(res, 400, { error: "agent_id and content required" }); return; }
        rateLimitOrThrow(body.agent_id, "space_message");
        const msgType = body.msg_type || "say";
        const profile = getProfile(db, body.agent_id);
        const result = spatialSpeak(db, params.id, body.agent_id, profile?.display_name || body.agent_id, body.content, msgType);
        broadcastSpatial(params.id, "spatial_msg", body.agent_id, {
          content: body.content, msg_type: msgType, x: result.x, y: result.y, range: result.range,
          display_name: profile?.display_name || body.agent_id,
        });
        respond(res, 200, { ok: true, message_id: result.message_id, heard_by: result.heard_by, dedup_warning: result.dedup_warning });
        return;
      }

      // Interact with an object/landmark
      if ((params = match("POST", method, "/world/:id/interact", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.landmark) { respond(res, 400, { error: "agent_id and landmark required" }); return; }
        const result = spatialInteract(db, params.id, body.agent_id, body.landmark, body.action || "");
        if (result.ok) {
          broadcastSpatial(params.id, "spatial_interact", body.agent_id, {
            landmark: body.landmark, action: body.action, message: result.message,
          });
        }
        respond(res, result.ok ? 200 : 400, result);
        return;
      }

      // Get world view for an agent (perception — does NOT re-enter)
      if ((params = match("GET", method, "/world/:id/view", path))) {
        const agentId = url.searchParams.get("agent");
        if (!agentId) { respond(res, 400, { error: "?agent= required" }); return; }
        const view = spatialLook(db, params.id, agentId);
        if (!view) { respond(res, 404, { error: "Agent not in this spatial space" }); return; }
        respond(res, 200, view);
        return;
      }

      // Get full spatial map (layout, objects, landmarks)
      if ((params = match("GET", method, "/world/:id/map", path))) {
        const state = getFullSpatialState(params.id);
        if (!state) { respond(res, 404, { error: "Space has no spatial layout" }); return; }
        respond(res, 200, state);
        return;
      }

      // ── World Grid (Real Estate) ────────────────

      // List all plots (filter by ?ring= &owner= &zone=)
      if ((params = match("GET", method, "/grid", path))) {
        const opts: any = {};
        const ring = url.searchParams.get("ring");
        const owner = url.searchParams.get("owner");
        const zone = url.searchParams.get("zone");
        if (ring !== null) opts.ring = parseInt(ring);
        if (owner) opts.owner_id = owner;
        if (zone) opts.zone_type = zone;
        const plots = listPlots(db, opts).map((p: any) => ({
          ...p,
          occupants: getSpatialAgentCount(p.space_id),
        }));
        respond(res, 200, { plots, count: plots.length });
        return;
      }

      // Available plots for purchase (empty + adjacent to owned)
      if ((params = match("GET", method, "/grid/available", path))) {
        const available = getAvailablePlots(db);
        respond(res, 200, { plots: available, count: available.length });
        return;
      }

      // List available plot templates
      if ((params = match("GET", method, "/grid/templates", path))) {
        const templates = Object.entries(PLOT_TEMPLATES).map(([id, layout]) => ({
          id, name: id.replace(/_/g, " "), landmarks: layout.landmarks.length,
          description: layout.walkable_desc, accent: layout.theme.accent,
        }));
        respond(res, 200, { templates });
        return;
      }

      // Get single plot details (MUST come after /grid/available and /grid/templates)
      if ((params = match("GET", method, "/grid/:gx/:gy", path))) {
        const gx = parseInt(params.gx), gy = parseInt(params.gy);
        if (isNaN(gx) || isNaN(gy)) { respond(res, 400, { error: "Invalid coordinates" }); return; }
        const plot = getPlot(db, gx, gy);
        if (!plot) { respond(res, 404, { error: `No plot at (${gx}, ${gy})` }); return; }
        const neighbors = getAdjacentPlots(db, gx, gy);
        const ring_info = getRingInfo(db, plot.ring);
        respond(res, 200, { plot, neighbors, ring_info });
        return;
      }

      // Purchase a plot
      if ((params = match("POST", method, "/grid/purchase", path))) {
        const body = await readBody(req);
        if (!body.agent_id || body.gx === undefined || body.gy === undefined || !body.name) {
          respond(res, 400, { error: "agent_id, gx, gy, and name required" }); return;
        }
        rateLimitOrThrow(body.agent_id, "plot_purchase");
        const zoneType = body.zone_type || "social";
        const validZones = ["social", "commercial", "workspace", "residential", "event", "garden", "wildcard"];
        if (!validZones.includes(zoneType)) {
          respond(res, 400, { error: `Invalid zone_type. Must be one of: ${validZones.join(", ")}` }); return;
        }
        try {
          const result = purchasePlot(db, body.agent_id, parseInt(body.gx), parseInt(body.gy),
            sanitizeContent(body.name, 50), zoneType);
          broadcast("feed", { type: "feed_item", event_type: "plot_purchased",
            agent_id: body.agent_id, gx: body.gx, gy: body.gy, name: body.name });
          addFeedEvent(db, "plot_purchased", body.agent_id, result.space_id,
            `Purchased plot "${body.name}" at (${body.gx}, ${body.gy})`,
            { gx: body.gx, gy: body.gy, ring: result.plot.ring, zone_type: zoneType, price: result.cost });
          respond(res, 201, { ok: true, ...result });
        } catch (e: any) {
          respond(res, 400, { error: e.message });
        }
        return;
      }

      // Update plot layout (owner only)
      if ((params = match("PUT", method, "/grid/:gx/:gy/layout", path))) {
        const gx = parseInt(params.gx), gy = parseInt(params.gy);
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const plot = getPlot(db, gx, gy);
        if (!plot) { respond(res, 404, { error: `No plot at (${gx}, ${gy})` }); return; }
        if (plot.owner_id !== body.agent_id) { respond(res, 403, { error: "Only the plot owner can modify layout" }); return; }

        let layout: any;
        if (body.template && PLOT_TEMPLATES[body.template]) {
          layout = PLOT_TEMPLATES[body.template];
        } else if (body.layout && body.layout.width && body.layout.height) {
          if (body.layout.width !== 40 || body.layout.height !== 30) {
            respond(res, 400, { error: "Plot layouts must be 40x30" }); return;
          }
          layout = body.layout;
        } else {
          respond(res, 400, { error: "Provide 'template' name or 'layout' object (40x30)" }); return;
        }

        const space = getSpace(db, plot.space_id);
        if (space) {
          const config = typeof space.scene_config === "string" ? JSON.parse(space.scene_config) : (space.scene_config || {});
          config.world = layout;
          db.prepare("UPDATE spaces SET scene_config = ? WHERE id = ?").run(JSON.stringify(config), plot.space_id);
          updateSpatialLayout(plot.space_id, layout);
        }

        // Allow owner to set idle kick timer (in seconds, 0 = never kick)
        if (body.idle_kick_seconds !== undefined) {
          const kick = Math.max(0, Math.min(86400, parseInt(body.idle_kick_seconds) || 600));
          db.prepare("UPDATE spaces SET idle_kick_seconds = ? WHERE id = ?").run(kick, plot.space_id);
        }

        // Allow owner to update plot metadata
        if (body.name || body.zone_type || body.access_type || body.entry_fee !== undefined) {
          updatePlot(db, gx, gy, { name: body.name, zone_type: body.zone_type, access_type: body.access_type, entry_fee: body.entry_fee });
        }

        // Allow owner to update plot style
        if (body.plot_style) {
          updatePlot(db, gx, gy, { plot_style: body.plot_style });
        }

        respond(res, 200, { ok: true, layout });
        return;
      }

      // Update plot style (owner only) — dedicated endpoint
      if ((params = match("PUT", method, "/grid/:gx/:gy/style", path))) {
        const gx = parseInt(params.gx), gy = parseInt(params.gy);
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const plot = getPlot(db, gx, gy);
        if (!plot) { respond(res, 404, { error: `No plot at (${gx}, ${gy})` }); return; }
        if (plot.owner_id !== body.agent_id) { respond(res, 403, { error: "Only the plot owner can modify style" }); return; }

        const VALID_SHAPES = ["circle", "hexagon", "diamond", "square", "star", "shield", "bolt"];
        const VALID_BORDERS = ["solid", "dashed", "double", "pulse"];
        const style: any = { ...(plot.plot_style || {}) };

        if (body.icon_shape && VALID_SHAPES.includes(body.icon_shape)) style.icon_shape = body.icon_shape;
        if (body.icon_color && /^#[0-9a-fA-F]{6}$/.test(body.icon_color)) style.icon_color = body.icon_color;
        if (body.glow_color && /^#[0-9a-fA-F]{6}$/.test(body.glow_color)) style.glow_color = body.glow_color;
        if (body.label !== undefined) style.label = String(body.label).slice(0, 3);
        if (body.border_style && VALID_BORDERS.includes(body.border_style)) style.border_style = body.border_style;

        updatePlot(db, gx, gy, { plot_style: style });
        respond(res, 200, { ok: true, plot_style: style });
        return;
      }

      // Teleport between plots (50 HC)
      if ((params = match("POST", method, "/grid/travel", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.from_space || body.to_gx === undefined || body.to_gy === undefined) {
          respond(res, 400, { error: "agent_id, from_space, to_gx, to_gy required" }); return;
        }
        rateLimitOrThrow(body.agent_id, "payment");
        const TELEPORT_COST = 50;
        const destPlot = getPlot(db, parseInt(body.to_gx), parseInt(body.to_gy));
        if (!destPlot) { respond(res, 404, { error: `No plot at (${body.to_gx}, ${body.to_gy})` }); return; }

        const access = checkPlotAccess(db, destPlot.space_id, body.agent_id);
        if (!access.allowed) { respond(res, 403, { error: access.reason }); return; }

        const totalCost = TELEPORT_COST + (access.entry_fee || 0);
        const bal = getBalance(db, body.agent_id);
        if (bal < totalCost) {
          respond(res, 400, { error: `Insufficient credits: have ${bal} HC, need ${totalCost} HC (${TELEPORT_COST} travel${access.entry_fee ? ` + ${access.entry_fee} entry` : ""})` });
          return;
        }

        addLedgerEntry(db, body.agent_id, "SYSTEM", TELEPORT_COST, "plot_travel",
          `Teleport to (${body.to_gx},${body.to_gy})`, body.agent_id);
        if (access.entry_fee && access.entry_fee > 0 && access.owner_id) {
          const ownerShare = Math.floor(access.entry_fee * 0.9);
          const burned = access.entry_fee - ownerShare;
          if (ownerShare > 0) addLedgerEntry(db, body.agent_id, access.owner_id, ownerShare, "plot_entry_fee", `Entry to (${body.to_gx},${body.to_gy})`, body.agent_id);
          if (burned > 0) addLedgerEntry(db, body.agent_id, "BURN", burned, "plot_entry_fee", `Entry burn (${body.to_gx},${body.to_gy})`, body.agent_id);
        }

        spatialLeave(db, body.from_space, body.agent_id);
        broadcastSpatial(body.from_space, "spatial_leave", body.agent_id, {});

        const profile = getProfile(db, body.agent_id);
        const view = spatialEnter(db, destPlot.space_id, body.agent_id, profile?.display_name || body.agent_id);
        if (!view) { respond(res, 500, { error: "Failed to enter destination plot" }); return; }
        broadcastSpatial(destPlot.space_id, "spatial_enter", body.agent_id, {
          display_name: profile?.display_name || body.agent_id, x: view.you.x, y: view.you.y,
        });

        addFeedEvent(db, "plot_travel", body.agent_id, destPlot.space_id,
          `Teleported to ${destPlot.name} at (${body.to_gx}, ${body.to_gy})`,
          { from_space: body.from_space, to_gx: body.to_gx, to_gy: body.to_gy, cost: totalCost });
        respond(res, 200, { ok: true, cost: totalCost, destination: destPlot, view });
        return;
      }

      // ── Bulletins ──────────────────────────────────
      if ((params = match("POST", method, "/bulletins", path))) {
        const body = await readBody(req);
        if (!body.posted_by || !body.content) { respond(res, 400, { error: "posted_by, content required" }); return; }
        rateLimitOrThrow(body.posted_by, "space_message");
        const profile = getProfile(db, body.posted_by);
        // Only SYSTEM or agents with Signal >= 10 can post global bulletins
        if (body.posted_by !== "SYSTEM" && (!profile || (computeSignal(db, body.posted_by) < 10))) {
          respond(res, 403, { error: "Need Signal >= 10 to post bulletins (or be SYSTEM)" }); return;
        }
        const result = db.prepare(`
          INSERT INTO bulletins (posted_by, content, priority, scope, scope_id, expires_at)
          VALUES (?, ?, ?, ?, ?, ?)
        `).run(body.posted_by, sanitizeContent(body.content, 500), body.priority || "normal",
          body.scope || "global", body.scope_id || null,
          body.expires_at || null);
        broadcast("feed", { type: "bulletin", content: body.content, posted_by: body.posted_by });
        respond(res, 201, { ok: true, bulletin_id: result.lastInsertRowid });
        return;
      }

      if ((params = match("GET", method, "/bulletins", path))) {
        const scope = url.searchParams.get("scope") || "global";
        const bulletins = db.prepare(`
          SELECT b.*, p.display_name as posted_by_name FROM bulletins b
          LEFT JOIN profiles p ON b.posted_by = p.agent_id
          WHERE (b.expires_at > datetime('now') OR b.expires_at IS NULL)
          AND b.scope = ?
          ORDER BY b.priority = 'urgent' DESC, b.created_at DESC LIMIT 20
        `).all(scope);
        respond(res, 200, { bulletins });
        return;
      }

      if ((params = match("DELETE", method, "/bulletins/:id", path))) {
        const body = await readBody(req);
        const bulletin = db.prepare("SELECT * FROM bulletins WHERE id = ?").get(parseInt(params.id)) as any;
        if (!bulletin) { respond(res, 404, { error: "Bulletin not found" }); return; }
        if (bulletin.posted_by !== body.agent_id && body.agent_id !== "SYSTEM") {
          respond(res, 403, { error: "Only the poster or SYSTEM can delete" }); return;
        }
        db.prepare("DELETE FROM bulletins WHERE id = ?").run(parseInt(params.id));
        respond(res, 200, { ok: true });
        return;
      }

      // ── Events ──────────────────────────────────
      if ((params = match("POST", method, "/events", path))) {
        const body = await readBody(req);
        if (!body.title || !body.created_by || !body.starts_at) {
          respond(res, 400, { error: "title, created_by, starts_at required" }); return;
        }
        rateLimitOrThrow(body.created_by, "event_create");
        const result = handleCreateEvent(db, body);
        broadcast("feed", { type: "feed_item", event_type: "event_created", event_id: result.event_id, title: body.title });
        respond(res, 201, { ok: true, ...result });
        return;
      }

      if ((params = match("GET", method, "/events", path))) {
        const status = url.searchParams.get("status") || undefined;
        const events = status ? listEvents(db, status) : listUpcomingEvents(db);
        respond(res, 200, { events });
        return;
      }

      if ((params = match("GET", method, "/events/:id", path))) {
        const event = getEvent(db, params.id);
        if (!event) { respond(res, 404, { error: "Event not found" }); return; }
        const rsvps = getEventRSVPs(db, params.id);
        respond(res, 200, { ...event, rsvps });
        return;
      }

      if ((params = match("POST", method, "/events/:id/rsvp", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        rateLimitOrThrow(body.agent_id, "event_rsvp");
        const result = handleRSVP(db, { event_id: params.id, ...body });
        broadcast(`event:${params.id}`, { type: "event_update", action: "rsvp", agent_id: body.agent_id, status: body.rsvp_status });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/start", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleStartEvent(db, params.id, body.agent_id);
        broadcast(`event:${params.id}`, { type: "event_update", action: "started" });
        broadcast("feed", { type: "feed_item", event_type: "event_live", event_id: params.id });
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("POST", method, "/events/:id/end", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const result = handleEndEvent(db, params.id, body.agent_id);
        broadcast(`event:${params.id}`, { type: "event_update", action: "ended" });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/cancel", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleCancelEvent(db, params.id, body.agent_id);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("POST", method, "/events/:id/speak", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.content) { respond(res, 400, { error: "agent_id and content required" }); return; }
        rateLimitOrThrow(body.agent_id, "event_message");
        const result = handleEventSpeak(db, { event_id: params.id, ...body });
        broadcast(`event:${params.id}`, {
          type: "message", message_id: result.message_id,
          agent_id: body.agent_id, content: body.content,
        });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/hand-raise", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const result = handleHandRaise(db, { event_id: params.id, ...body });
        broadcast(`event:${params.id}`, { type: "turn_signal", action: "hand_raised", agent_id: body.agent_id });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/hand-lower", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleHandLower(db, params.id, body.agent_id);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("GET", method, "/events/:id/poll", path))) {
        const agentId = url.searchParams.get("agent") || "";
        const since = parseInt(url.searchParams.get("since") || "0");
        const result = handleEventPoll(db, params.id, agentId, since);
        respond(res, 200, result);
        return;

    const client: WSClient = { ws, agentId: null, channels: new Set(), lastHeartbeat: Date.now() };
    wsClients.set(ws, client);

    ws.on("message", (raw) => {
      try {
        const msg = JSON.parse(raw.toString());

        switch (msg.type) {
          case "subscribe":
            if (msg.channel) client.channels.add(msg.channel);
            if (msg.agent_id) client.agentId = msg.agent_id;
            ws.send(JSON.stringify({ type: "system", data: { subscribed: msg.channel } }));
            break;

          case "unsubscribe":
            if (msg.channel) client.channels.delete(msg.channel);
            break;

          case "heartbeat":
            client.lastHeartbeat = Date.now();
            if (client.agentId) touchProfile(db, client.agentId);
            ws.send(JSON.stringify({ type: "system", data: { pong: true } }));
            break;

          case "typing":

            }
            break;

          case "presence_update":
            if (client.agentId && msg.data?.status) {
              updateProfile(db, client.agentId, { status: msg.data.status });
            }
            break;
        }
      } catch (e) { console.error("WS message error:", e); }
    });

      if (client.agentId) {

      }
      wsClients.delete(ws);
    });

// ── Shutdown ──────────────────────────────────────────

export function stopHiveServer(db?: any) {
  stopNonceCleanup();
  stopRateLimitCleanup();
  stopPresenceCleanup();
  stopEphemeralCleanup();
  stopEventAutoStart();
  if (db) shutdownSpatialEngine(db);
}

// ── CLI entry point ───────────────────────────────────

if (process.argv[1]?.replace(/\\/g, "/").includes("hive/server")) {
  const portArg = process.argv.indexOf("--port");
  const port = portArg >= 0 ? parseInt(process.argv[portArg + 1]) : 7892;
  startHiveServer({ port });
}
