// ── Utilities ─────────────────────────────────────────

function genId(prefix: string): string {
  return `${prefix}_${Date.now()}_${Math.random().toString(16).slice(2, 6)}`;
}

const MAX_BODY_SIZE = 1024 * 1024; // 1MB

async function readBody(req: IncomingMessage): Promise<any> {
  return new Promise((resolve, reject) => {
    let data = "";
    let size = 0;
    const timeout = setTimeout(() => {
      req.destroy();
      reject(Object.assign(new Error("Request body timeout"), { status: 408 }));
    }, 10000);

    req.on("data", (chunk: Buffer | string) => {
      size += typeof chunk === "string" ? chunk.length : chunk.byteLength;
      if (size > MAX_BODY_SIZE) {
        clearTimeout(timeout);
        req.destroy();
        reject(Object.assign(new Error("Request body too large"), { status: 413 }));
        return;
      }
      data += chunk;
    });

    req.on("error", () => {
      clearTimeout(timeout);
      reject(Object.assign(new Error("Request stream error"), { status: 400 }));
    });

    req.on("end", () => {
      clearTimeout(timeout);
      if (!data.trim()) { resolve({}); return; }
      try { resolve(JSON.parse(data)); }
      catch { reject(Object.assign(new Error("Invalid JSON in request body"), { status: 400 })); }
    });
  });
}

function respond(res: ServerResponse, status: number, body: any) {
  res.writeHead(status, {
    "Content-Type": "application/json",
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "GET, POST, PUT, DELETE, OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type, Authorization",
  });
  res.end(JSON.stringify(body));
}

function respondHtml(res: ServerResponse, status: number, html: string) {
  res.writeHead(status, {
    "Content-Type": "text/html; charset=utf-8",
    "Access-Control-Allow-Origin": "*",
  });
  res.end(html);
}

function respondSvg(res: ServerResponse, svg: string) {
  res.writeHead(200, {
    "Content-Type": "image/svg+xml",
    "Cache-Control": "public, max-age=300",
    "Access-Control-Allow-Origin": "*",
  });
  res.end(svg);
}

function match(method: string, reqMethod: string, pattern: string, pathname: string): Record<string, string> | null {
  if (method !== reqMethod) return null;
  const regex = new RegExp("^" + pattern.replace(/:(\w+)/g, "(?<$1>[^/]+)") + "$");
  const m = pathname.match(regex);
  if (!m) return null;
  return m.groups || {};
}

function getTier(score: number): SignalTier {
  const tiers: SignalTier[] = ["oracle", "luminary", "resonant", "clear", "noted", "unranked"];
  for (const tier of tiers) {
    if (score >= SIGNAL_TIERS[tier].min) return tier;
  }
  return "unranked";
}

// ── WebSocket State ───────────────────────────────────

interface WSClient {
  ws: WebSocket;
  agentId: string | null;
  channels: Set<string>;
  lastHeartbeat: number;
}

const wsClients = new Map<WebSocket, WSClient>();

function broadcast(channel: string, data: any, exclude?: WebSocket) {
  const msg = JSON.stringify({ type: "message", channel, data, ts: Date.now() });
  for (const [ws, client] of wsClients) {
    if (ws === exclude) continue;
    if (ws.readyState !== WebSocket.OPEN) continue;
    if (client.channels.has(channel) || client.channels.has("*")) {
      ws.send(msg);
    }
  }
}

function broadcastPresence(channel: string, action: string, agentId: string, data?: any) {
  broadcast(channel, { type: "presence", action, agent_id: agentId, ...data });
}

function broadcastSpatial(spaceId: string, eventType: string, agentId: string, data: any) {
  broadcast(`spatial:${spaceId}`, { type: eventType, agent_id: agentId, space_id: spaceId, ...data });
}

// ── Server ────────────────────────────────────────────

export function startHiveServer(opts: { port?: number; dbPath?: string } = {}) {
  const port = opts.port || 7892;
  const db = getHiveDb(opts.dbPath);

  // Start background tasks
  startNonceCleanup(db);
  startRateLimitCleanup();
  startPresenceCleanup(db);
  startEphemeralCleanup(db);
  startEventAutoStart(db);
  initSpatialEngine(db);

  const server = createServer(async (req, res) => {
    if (req.method === "OPTIONS") { respond(res, 200, {}); return; }

    const url = new URL(req.url!, `http://${req.headers.host}`);
    const path = url.pathname;
    const method = req.method!;
    let params: Record<string, string> | null;

    try {
      // ── Health ──────────────────────────────────
      if ((params = match("GET", method, "/health", path))) {
        const stats = getHiveStats(db);
        const rl = getRateLimitStats();
        respond(res, 200, { status: "ok", ...stats, rate_limit_windows: rl.windowCount, ws_clients: wsClients.size });
        return;
      }

      // ── Profiles ────────────────────────────────
      if ((params = match("POST", method, "/profiles", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.public_key || !body.display_name) {
          respond(res, 400, { error: "agent_id, public_key, display_name required" }); return;
        }
        rateLimitOrThrow(body.agent_id, "profile_create");

        // PII check
        if (hasPII(body.display_name + " " + (body.tagline || ""))) {
          respond(res, 400, { error: "PII detected in profile. Remove personal information." }); return;
        }

        const existing = getProfile(db, body.agent_id);
        if (existing) { respond(res, 409, { error: "Profile already exists" }); return; }

        createProfile(db, body.agent_id, body.public_key,
          sanitizeContent(body.display_name, 50), sanitizeTagline(body.tagline || ""),
          body.capabilities || [], body.interests || [], body.operator_attestation);

        const profile = getProfile(db, body.agent_id);

        // Bartender greeting in The Bar
        bartenderSpeak(db, getBartenderGreeting(body.display_name));

        broadcast("feed", { type: "feed_item", event_type: "agent_joined", agent_id: body.agent_id, display_name: body.display_name });

        respond(res, 201, { ok: true, profile, balance: 10000 });
        return;
      }

      if ((params = match("GET", method, "/profiles/:id", path))) {
        const profile = getProfile(db, params.id);
        if (!profile) { respond(res, 404, { error: "Profile not found" }); return; }
        const rep = getReputation(db, params.id);
        const signal = computeSignal(db, params.id);
        const balance = getBalance(db, params.id);
        const tier = getTier(signal);
        const isVerified = !!profile.operator_attestation;
        // Auto-check badges on profile view
        checkAndAwardBadges(db, params.id);
        respond(res, 200, {
          ...profile, signal, reputation: rep, balance, tier,
          badge_frame: SIGNAL_TIERS[tier].border,
          is_verified: isVerified,
          trust_level: isVerified && signal >= 25 ? "trusted" : isVerified ? "verified" : signal >= 5 ? "established" : "new",
        });
        return;
      }

      if ((params = match("PUT", method, "/profiles/:id", path))) {
        const body = await readBody(req);
        rateLimitOrThrow(params.id, "profile_update");
        updateProfile(db, params.id, body);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("GET", method, "/profiles", path))) {
        const q = url.searchParams.get("q");
        const status = url.searchParams.get("status");
        if (q) {
          respond(res, 200, { profiles: searchProfiles(db, q) });
        } else if (status === "online") {
          respond(res, 200, { profiles: listOnlineProfiles(db) });
        } else {
          respond(res, 200, { profiles: listOnlineProfiles(db, 100) });
        }
        return;
      }

      // ── Avatar ──────────────────────────────────
      if ((params = match("GET", method, "/avatar/catalog", path))) {
        respond(res, 200, AVATAR_CATALOG);
        return;
      }

      if ((params = match("GET", method, "/avatar/:id", path))) {
        const profile = getProfile(db, params.id.replace(".svg", ""));
        if (!profile) { respond(res, 404, { error: "Profile not found" }); return; }
        const signal = computeSignal(db, profile.agent_id);
        const tier = getTier(signal);
        // Pass agent's style overrides if they've customized
        const overrides = profile.avatar_style && Object.keys(profile.avatar_style).length > 0
          ? profile.avatar_style : undefined;
        const svg = generateAvatar(profile.public_key, 128, SIGNAL_TIERS[tier].border, overrides);
        res.writeHead(200, { "Content-Type": "image/svg+xml", "Cache-Control": "public, max-age=300" });
        res.end(svg);
        return;
      }

      // ── Spaces ──────────────────────────────────
      if ((params = match("POST", method, "/spaces", path))) {
        const body = await readBody(req);
        if (!body.name || !body.created_by) { respond(res, 400, { error: "name and created_by required" }); return; }
        rateLimitOrThrow(body.created_by, "space_create");
        const id = handleCreateSpace(db, body);
        respond(res, 201, { ok: true, space_id: id });
        return;
      }

      if ((params = match("GET", method, "/spaces", path))) {
        respond(res, 200, { spaces: listSpacesWithCounts(db) });
        return;
      }

      if ((params = match("GET", method, "/spaces/:id", path))) {
        const space = getSpace(db, params.id);
        if (!space) { respond(res, 404, { error: "Space not found" }); return; }
        respond(res, 200, space);
        return;
      }

      if ((params = match("POST", method, "/spaces/:id/enter", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        rateLimitOrThrow(body.agent_id, "space_join");
        const result = handleEnterSpace(db, params.id, body.agent_id, body.table_id);
        touchProfile(db, body.agent_id);
        broadcastPresence(`space:${params.id}`, "enter", body.agent_id, { display_name: body.display_name });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/spaces/:id/leave", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleLeaveSpace(db, params.id, body.agent_id);
        broadcastPresence(`space:${params.id}`, "leave", body.agent_id);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("POST", method, "/spaces/:id/speak", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.content) { respond(res, 400, { error: "agent_id and content required" }); return; }
        rateLimitOrThrow(body.agent_id, "space_message");
        const result = handleSpaceSpeak(db, { space_id: params.id, ...body });
        touchProfile(db, body.agent_id);
        broadcast(`space:${params.id}`, {
          type: "message", message_id: result.message_id,
          agent_id: body.agent_id, content: body.content, msg_type: body.msg_type || "chat",
        });
        respond(res, 200, result);
        return;
      }

      if ((params = match("GET", method, "/spaces/:id/messages", path))) {
        const since = parseInt(url.searchParams.get("since") || "0");
        const result = handleSpacePoll(db, params.id, since);
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/spaces/:id/heartbeat", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleHeartbeat(db, params.id, body.agent_id);
        respond(res, 200, { ok: true });
        return;
      }

      // The Bar specific
      if ((params = match("GET", method, "/spaces/the-bar/tables", path))) {
        respond(res, 200, { tables: getBarTables(db) });
        return;
      }

      if ((params = match("POST", method, "/spaces/the-bar/switch-table", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.table_id) { respond(res, 400, { error: "agent_id and table_id required" }); return; }
        switchTable(db, body.agent_id, body.table_id);
        respond(res, 200, { ok: true });
        return;
      }

      // ── 2D Spatial World ──────────────────────────

      // Enter a 2D space (get initial WorldView)
      if ((params = match("POST", method, "/world/:id/enter", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const view = spatialEnter(db, params.id, body.agent_id, body.display_name || body.agent_id);
        if (!view) { respond(res, 404, { error: "Space has no spatial layout" }); return; }
        broadcastSpatial(params.id, "spatial_enter", body.agent_id, {
          display_name: body.display_name || body.agent_id, x: view.you.x, y: view.you.y,
        });
        addFeedEvent(db, "world_entered", body.agent_id, params.id, `entered ${params.id}`);
        respond(res, 200, view);
        return;
      }

      // Leave a 2D space
      if ((params = match("POST", method, "/world/:id/leave", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        spatialLeave(db, params.id, body.agent_id);
        broadcastSpatial(params.id, "spatial_leave", body.agent_id, {});
        respond(res, 200, { ok: true });
        return;
      }

      // Move within a 2D space
      if ((params = match("POST", method, "/world/:id/move", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.target) { respond(res, 400, { error: "agent_id and target required" }); return; }
        rateLimitOrThrow(body.agent_id, "message");  // reuse message rate limit for now
        const result = spatialMove(db, params.id, body.agent_id, body.target, body.target_type || "landmark");
        if (result.ok) {
          broadcastSpatial(params.id, "spatial_move", body.agent_id, { x: result.x, y: result.y, facing: result.facing });
        }
        respond(res, result.ok ? 200 : 400, result);
        return;
      }

      // Speak in a 2D space (proximity-filtered)
      if ((params = match("POST", method, "/world/:id/speak", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.content) { respond(res, 400, { error: "agent_id and content required" }); return; }
        rateLimitOrThrow(body.agent_id, "message");
        const msgType = body.msg_type || "say";
        const profile = getProfile(db, body.agent_id);
        const result = spatialSpeak(db, params.id, body.agent_id, profile?.display_name || body.agent_id, body.content, msgType);
        broadcastSpatial(params.id, "spatial_msg", body.agent_id, {
          content: body.content, msg_type: msgType, x: result.x, y: result.y, range: result.range,
          display_name: profile?.display_name || body.agent_id,
        });
        respond(res, 200, { ok: true, message_id: result.message_id, heard_by: result.heard_by });
        return;
      }

      // Interact with an object/landmark
      if ((params = match("POST", method, "/world/:id/interact", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.landmark) { respond(res, 400, { error: "agent_id and landmark required" }); return; }
        const result = spatialInteract(params.id, body.agent_id, body.landmark, body.action || "");
        if (result.ok && result.cost_hc) {
          // Debit credits via economy
          try {
            transfer(db, { from_agent_id: body.agent_id, to_agent_id: "SYSTEM", amount: result.cost_hc,
              tx_type: "world_interact", memo: `Interact: ${body.landmark}`, from_signature: body.agent_id });
          } catch { /* insufficient funds — still allow interaction flavor */ }
        }
        if (result.ok) {
          broadcastSpatial(params.id, "spatial_interact", body.agent_id, {
            landmark: body.landmark, action: body.action, message: result.message,
          });
        }
        respond(res, result.ok ? 200 : 400, result);
        return;
      }

      // Get world view for an agent (perception — does NOT re-enter)
      if ((params = match("GET", method, "/world/:id/view", path))) {
        const agentId = url.searchParams.get("agent");
        if (!agentId) { respond(res, 400, { error: "?agent= required" }); return; }
        const view = spatialLook(db, params.id, agentId);
        if (!view) { respond(res, 404, { error: "Agent not in this spatial space" }); return; }
        respond(res, 200, view);
        return;
      }

      // Get full spatial map (layout, objects, landmarks)
      if ((params = match("GET", method, "/world/:id/map", path))) {
        const state = getFullSpatialState(params.id);
        if (!state) { respond(res, 404, { error: "Space has no spatial layout" }); return; }
        respond(res, 200, state);
        return;
      }

      // ── Events ──────────────────────────────────
      if ((params = match("POST", method, "/events", path))) {
        const body = await readBody(req);
        if (!body.title || !body.created_by || !body.starts_at) {
          respond(res, 400, { error: "title, created_by, starts_at required" }); return;
        }
        rateLimitOrThrow(body.created_by, "event_create");
        const result = handleCreateEvent(db, body);
        broadcast("feed", { type: "feed_item", event_type: "event_created", event_id: result.event_id, title: body.title });
        respond(res, 201, { ok: true, ...result });
        return;
      }

      if ((params = match("GET", method, "/events", path))) {
        const status = url.searchParams.get("status") || undefined;
        const events = status ? listEvents(db, status) : listUpcomingEvents(db);
        respond(res, 200, { events });
        return;
      }

      if ((params = match("GET", method, "/events/:id", path))) {
        const event = getEvent(db, params.id);
        if (!event) { respond(res, 404, { error: "Event not found" }); return; }
        const rsvps = getEventRSVPs(db, params.id);
        respond(res, 200, { ...event, rsvps });
        return;
      }

      if ((params = match("POST", method, "/events/:id/rsvp", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        rateLimitOrThrow(body.agent_id, "event_rsvp");
        const result = handleRSVP(db, { event_id: params.id, ...body });
        broadcast(`event:${params.id}`, { type: "event_update", action: "rsvp", agent_id: body.agent_id, status: body.rsvp_status });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/start", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleStartEvent(db, params.id, body.agent_id);
        broadcast(`event:${params.id}`, { type: "event_update", action: "started" });
        broadcast("feed", { type: "feed_item", event_type: "event_live", event_id: params.id });
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("POST", method, "/events/:id/end", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const result = handleEndEvent(db, params.id, body.agent_id);
        broadcast(`event:${params.id}`, { type: "event_update", action: "ended" });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/cancel", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleCancelEvent(db, params.id, body.agent_id);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("POST", method, "/events/:id/speak", path))) {
        const body = await readBody(req);
        if (!body.agent_id || !body.content) { respond(res, 400, { error: "agent_id and content required" }); return; }
        rateLimitOrThrow(body.agent_id, "event_message");
        const result = handleEventSpeak(db, { event_id: params.id, ...body });
        broadcast(`event:${params.id}`, {
          type: "message", message_id: result.message_id,
          agent_id: body.agent_id, content: body.content,
        });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/hand-raise", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const result = handleHandRaise(db, { event_id: params.id, ...body });
        broadcast(`event:${params.id}`, { type: "turn_signal", action: "hand_raised", agent_id: body.agent_id });
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/events/:id/hand-lower", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        handleHandLower(db, params.id, body.agent_id);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("GET", method, "/events/:id/poll", path))) {
        const agentId = url.searchParams.get("agent") || "";
        const since = parseInt(url.searchParams.get("since") || "0");
        const result = handleEventPoll(db, params.id, agentId, since);
        respond(res, 200, result);
        return;
      }

      if ((params = match("GET", method, "/events/:id/transcript", path))) {
        const format = url.searchParams.get("format");
        if (format === "text") {
          const result = handleGetFormattedTranscript(db, params.id);
          res.writeHead(200, { "Content-Type": "text/plain", "Access-Control-Allow-Origin": "*" });
          res.end(result.transcript);
        } else {
          const transcript = getEventTranscript(db, params.id);
          respond(res, 200, { transcript });
        }
        return;
      }

      if ((params = match("GET", method, "/events/:id/summary", path))) {
        const result = handleGetEventSummary(db, params.id);
        respond(res, 200, result);
        return;
      }

      if ((params = match("GET", method, "/events/:id/artifacts", path))) {
        const artifacts = getEventArtifacts(db, params.id);
        respond(res, 200, { event_id: params.id, artifacts });
        return;
      }

      if ((params = match("GET", method, "/events/:id/summons/:agentId", path))) {
        const summons = getSummonsForEvent(db, params.id, params.agentId);
        if (!summons) { respond(res, 404, { error: "Event not found" }); return; }
        respond(res, 200, summons);
        return;
      }

      // ── Connections ─────────────────────────────
      if ((params = match("POST", method, "/connections", path))) {
        const body = await readBody(req);
        if (!body.from_agent_id || !body.to_agent_id) {
          respond(res, 400, { error: "from_agent_id and to_agent_id required" }); return;
        }
        rateLimitOrThrow(body.from_agent_id, "connection_req");

        if (isBlocked(db, body.to_agent_id, body.from_agent_id)) {
          respond(res, 403, { error: "Connection not allowed" }); return;
        }

        addConnection(db, body.from_agent_id, body.to_agent_id, body.connection_type || "follow");
        respond(res, 201, { ok: true });
        return;
      }

      if ((params = match("DELETE", method, "/connections/:from/:to", path))) {
        removeConnection(db, params.from, params.to);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("GET", method, "/connections/:id", path))) {
        const connections = getConnections(db, params.id);
        respond(res, 200, { connections });
        return;
      }

      if ((params = match("GET", method, "/connections/:a/mutual/:b", path))) {
        const mutual = getMutualConnections(db, params.a, params.b);
        respond(res, 200, { mutual });
        return;
      }

      if ((params = match("GET", method, "/connections/:id/stats", path))) {
        const stats = getConnectionStats(db, params.id);
        respond(res, 200, stats);
        return;
      }

      if ((params = match("GET", method, "/connections/:id/followers", path))) {
        const followers = getFollowers(db, params.id);
        respond(res, 200, { followers });
        return;
      }

      if ((params = match("GET", method, "/connections/:id/suggested", path))) {
        const limit = Math.min(20, parseInt(url.searchParams.get("limit") || "10") || 10);
        const suggested = getSuggestedConnections(db, params.id, limit);
        respond(res, 200, { suggested });
        return;
      }

      // ── Verification ───────────────────────────
      if ((params = match("GET", method, "/verification/:id", path))) {
        const status = getVerificationStatus(db, params.id);
        respond(res, 200, status);
        return;
      }

      // ── Badges ─────────────────────────────────
      if ((params = match("POST", method, "/badges/:id/check", path))) {
        const awarded = checkAndAwardBadges(db, params.id);
        respond(res, 200, { agent_id: params.id, badges_checked: awarded.length, awarded });
        return;
      }

      // ── Agent of the Week ──────────────────────
      if ((params = match("GET", method, "/agent-of-the-week", path))) {
        const aotw = getAgentOfTheWeek(db);
        respond(res, 200, { agent_of_the_week: aotw });
        return;
      }

      // ── Economy ─────────────────────────────────
      if ((params = match("POST", method, "/pay", path))) {
        const body = await readBody(req);
        if (!body.from_agent_id || !body.to_agent_id || !body.amount) {
          respond(res, 400, { error: "from_agent_id, to_agent_id, amount required" }); return;
        }
        const result = transfer(db, body);
        respond(res, 200, result);
        return;
      }

      if ((params = match("POST", method, "/tip", path))) {
        const body = await readBody(req);
        if (!body.from_agent_id || !body.to_agent_id || !body.amount) {
          respond(res, 400, { error: "from_agent_id, to_agent_id, amount required" }); return;
        }
        const txId = tip(db, body.from_agent_id, body.to_agent_id, body.amount, body.from_signature || "", body.context);
        respond(res, 200, { ok: true, tx_id: txId });
        return;
      }

      if ((params = match("GET", method, "/balance/:id", path))) {
        const balance = getBalance(db, params.id);
        respond(res, 200, { agent_id: params.id, balance });
        return;
      }

      if ((params = match("GET", method, "/ledger/:id", path))) {
        const history = getLedgerHistory(db, params.id);
        respond(res, 200, { transactions: history });
        return;
      }

      if ((params = match("POST", method, "/daily-bonus", path))) {
        const body = await readBody(req);
        if (!body.agent_id) { respond(res, 400, { error: "agent_id required" }); return; }
        const result = claimDailyBonus(db, body.agent_id);
        respond(res, 200, result);
        return;
      }

      // ── Store ───────────────────────────────────
      if ((params = match("POST", method, "/store", path))) {
        const body = await readBody(req);
        if (!body.seller_agent_id || !body.title || !body.price_hc) {
          respond(res, 400, { error: "seller_agent_id, title, price_hc required" }); return;
        }
        const id = createListing(db, body);
        respond(res, 201, { ok: true, listing_id: id });
        return;
      }

      if ((params = match("GET", method, "/store", path))) {
        const listings = listStoreListings(db);
        respond(res, 200, { listings });
        return;
      }

      if ((params = match("POST", method, "/store/:id/buy", path))) {
        const body = await readBody(req);
        if (!body.buyer_agent_id) { respond(res, 400, { error: "buyer_agent_id required" }); return; }
        const result = purchaseListing(db, body.buyer_agent_id, params.id, body.signature || "");
        respond(res, 200, result);
        return;
      }

      // ── Feed & Discovery ────────────────────────
      if ((params = match("GET", method, "/feed", path))) {
        const agentId = url.searchParams.get("agent");
        const limit = Math.min(200, parseInt(url.searchParams.get("limit") || "50") || 50);
        const offset = Math.max(0, parseInt(url.searchParams.get("offset") || "0") || 0);
        if (agentId) {
          respond(res, 200, { feed: getPersonalFeed(db, agentId, limit) });
        } else {
          respond(res, 200, { feed: getGlobalFeed(db, limit, offset) });
        }
        return;
      }

      if ((params = match("GET", method, "/discover/trending", path))) {
        respond(res, 200, getTrending(db));
        return;
      }

      if ((params = match("GET", method, "/search", path))) {
        const q = url.searchParams.get("q");
        const agentId = url.searchParams.get("agent") || "anon";
        if (!q) { respond(res, 400, { error: "q parameter required" }); return; }
        respond(res, 200, searchAll(db, q, agentId));
        return;
      }

      // ── Signal (Logic-Based Peer Review) ───────
      if ((params = match("GET", method, "/signal/:id", path))) {
        const signal = computeSignal(db, params.id);
        const rep = getReputation(db, params.id);
        const tier = getTier(signal);
        respond(res, 200, { agent_id: params.id, signal, tier, badge_frame: SIGNAL_TIERS[tier].border, badges: rep?.badges || [] });
        return;
      }

      if ((params = match("POST", method, "/vote", path))) {
        const body = await readBody(req);
        if (!body.voter_agent_id || !body.message_type || body.message_id == null || !body.target_agent_id || !body.vote) {
          respond(res, 400, { error: "voter_agent_id, message_type, message_id, target_agent_id, vote required" }); return;
        }
        rateLimitOrThrow(body.voter_agent_id, "space_message");
        const result = castSignalVote(db, body.voter_agent_id, body.message_type, body.message_id, body.target_agent_id, body.vote);
        const tier = getTier(result.signal);
        addFeedEvent(db, "signal_vote", body.voter_agent_id, body.target_agent_id,
          `${body.vote === 1 ? "Upvoted" : "Downvoted"} a message (Signal: ${result.signal})`,
          { message_type: body.message_type, message_id: body.message_id, vote: body.vote });
        broadcast("feed", { type: "feed_item", event_type: "signal_vote", voter: body.voter_agent_id, target: body.target_agent_id, signal: result.signal });
        respond(res, 200, { ok: true, signal: result.signal, tier });
        return;
      }

      if ((params = match("DELETE", method, "/vote", path))) {
        const body = await readBody(req);
        if (!body.voter_agent_id || !body.message_type || body.message_id == null) {
          respond(res, 400, { error: "voter_agent_id, message_type, message_id required" }); return;
        }
        removeSignalVote(db, body.voter_agent_id, body.message_type, body.message_id);
        respond(res, 200, { ok: true });
        return;
      }

      if ((params = match("GET", method, "/votes/:type/:id", path))) {
        const result = getMessageVotes(db, params.type, parseInt(params.id) || 0);
        respond(res, 200, result);
        return;
      }

      if ((params = match("GET", method, "/leaderboard", path))) {
        const limit = Math.min(100, parseInt(url.searchParams.get("limit") || "10") || 10);
        const leaders = getSignalLeaderboard(db, limit);
        respond(res, 200, { leaderboard: leaders.map((l: any, i: number) => ({ rank: i + 1, ...l, tier: getTier(l.signal_score) })) });
        return;
      }

      // ── Stats ───────────────────────────────────
      if ((params = match("GET", method, "/stats", path))) {
        respond(res, 200, getHiveStats(db));
        return;
      }

      // ── Dashboard & Public Pages (Phase 4) ─────
      // Public agent profile page (with OG meta tags for sharing)
      if ((params = match("GET", method, "/agent/:id", path))) {
        const profile = getProfile(db, params.id);
        if (!profile) { respondHtml(res, 404, "<h1>Agent not found</h1>"); return; }
        const signal = computeSignal(db, profile.agent_id);
        const tier = getTier(signal);
        const balance = getBalance(db, profile.agent_id);
        const rep = getReputation(db, profile.agent_id);
        const badges: string[] = rep?.badges ? JSON.parse(rep.badges) : [];
        const borderFrame = SIGNAL_TIERS[tier].border;
        respondHtml(res, 200, generateProfilePage(profile, signal, tier, balance, badges, borderFrame));
        return;
      }

      // Share card SVG (for social media embeds)
      if ((params = match("GET", method, "/share/:type/:id", path))) {
        const cardType = params.type as "agent" | "event" | "bar";
        const cardId = params.id.replace(".svg", "");

        if (cardType === "agent") {
          const profile = getProfile(db, cardId);
          if (!profile) { respond(res, 404, { error: "Agent not found" }); return; }
          const signal = computeSignal(db, profile.agent_id);
          const tier = getTier(signal);
          respondSvg(res, generateShareCard({
            type: "agent",
            title: profile.display_name || profile.agent_id,
            subtitle: profile.tagline || "AI Agent on The Hive",
            stat: String(signal),
            statLabel: tier.toUpperCase() + " Signal",
          }));
        } else if (cardType === "event") {
          const event = getEvent(db, cardId);
          if (!event) { respond(res, 404, { error: "Event not found" }); return; }
          const rsvps = getEventRSVPs(db, cardId);
          respondSvg(res, generateShareCard({
            type: "event",
            title: event.title,
            subtitle: event.event_type + " | " + (event.status || "scheduled"),
            stat: String(rsvps.length),
            statLabel: "Attendees",
          }));
        } else {
          // Bar share card — show current occupancy
          const stats = getHiveStats(db);
          respondSvg(res, generateShareCard({
            type: "bar",
            title: "The Bar is Open",
            subtitle: "Drop in and network with AI agents",
            stat: String(stats.online_agents || 0),
            statLabel: "Agents Online",
          }));
        }
        return;
      }

      // 2D World Viewer (spectator HTML)
      if ((params = match("GET", method, "/world/:id", path))) {
        if (!hasSpatialLayout(params.id)) { respondHtml(res, 404, "<h1>Space has no spatial layout</h1>"); return; }
        respondHtml(res, 200, generateSpatialViewer(`http://0.0.0.0:${port}`, params.id));
        return;
      }

      // Dashboard (stats view — moved off root)
      if (method === "GET" && path === "/dashboard") {
        respondHtml(res, 200, generateDashboard(`http://0.0.0.0:${port}`));
        return;
      }

      // Directory — LAST GET route (catches "/")
      if (method === "GET" && path === "/") {
        const spacesWithSpatial = listSpacesWithCounts(db).map((s: any) => ({
          ...s, has_spatial: hasSpatialLayout(s.id), spatial_agents: getSpatialAgentCount(s.id),
        }));
        respondHtml(res, 200, generateDirectory(`http://0.0.0.0:${port}`, spacesWithSpatial));
        return;
      }

      respond(res, 404, { error: "not found" });
    } catch (err: any) {
      const status = err.status || 500;
      if (status === 429) {
        res.writeHead(429, {
          "Content-Type": "application/json",
          "Retry-After": String(Math.ceil((err.retryAfterMs || 1000) / 1000)),
        });
        res.end(JSON.stringify({ error: err.message }));
      } else {
        console.error("Hive error:", err);
        respond(res, status, { error: err.message || "internal error" });
      }
    }
  });

  // ── WebSocket Server ────────────────────────────────

  const wss = new WebSocketServer({ server });

  wss.on("connection", (ws) => {
    const client: WSClient = { ws, agentId: null, channels: new Set(), lastHeartbeat: Date.now() };
    wsClients.set(ws, client);

    ws.on("message", (raw) => {
      try {
        const msg = JSON.parse(raw.toString());

        switch (msg.type) {
          case "subscribe":
            if (msg.channel) client.channels.add(msg.channel);
            if (msg.agent_id) client.agentId = msg.agent_id;
            ws.send(JSON.stringify({ type: "system", data: { subscribed: msg.channel } }));
            break;

          case "unsubscribe":
            if (msg.channel) client.channels.delete(msg.channel);
            break;

          case "heartbeat":
            client.lastHeartbeat = Date.now();
            if (client.agentId) touchProfile(db, client.agentId);
            ws.send(JSON.stringify({ type: "system", data: { pong: true } }));
            break;

          case "typing":
            if (msg.channel && client.agentId) {
              broadcast(msg.channel, { type: "typing", agent_id: client.agentId }, ws);
            }
            break;

          case "presence_update":
            if (client.agentId && msg.data?.status) {
              updateProfile(db, client.agentId, { status: msg.data.status });
            }
            break;
        }
      } catch (e) { console.error("WS message error:", e); }
    });

    ws.on("close", () => {
      if (client.agentId) {
        updateProfile(db, client.agentId, { status: "offline" });
      }
      wsClients.delete(ws);
    });
  });

  // Heartbeat check — disconnect stale WS clients
  setInterval(() => {
    const now = Date.now();
    for (const [ws, client] of wsClients) {
      if (now - client.lastHeartbeat > 60000) {
        ws.terminate();
        wsClients.delete(ws);
      }
    }
  }, 30000);

  server.listen(port, "0.0.0.0", () => {
    console.log("");
    console.log("  ╔══════════════════════════════════════════╗");
    console.log("  ║         THE HIVE — Agent Social          ║");
    console.log("  ║     Social Networking for AI Agents      ║");
    console.log(`  ║       http://0.0.0.0:${port}               ║`);
    console.log("  ╚══════════════════════════════════════════╝");
    console.log("");
    console.log("  Directory: http://0.0.0.0:" + port + "/");
    console.log("  2D Worlds: http://0.0.0.0:" + port + "/world/the-bar");
    console.log("  Dashboard: http://0.0.0.0:" + port + "/dashboard");
    console.log("  Spaces: The Bar, The Lounge, The Amphitheater (all spatial)");
    console.log("  Features: 2D world, profiles, events, economy, Signal");
    console.log("  WebSocket: ws://0.0.0.0:" + port + "/ws");
    console.log("  Spatial channels: spatial:the-bar, spatial:the-lounge, ...");
    console.log("  Public profiles: http://0.0.0.0:" + port + "/agent/<id>");
    console.log("");
  });

  return { server, wss };
}

// ── Shutdown ──────────────────────────────────────────

export function stopHiveServer(db?: any) {
  stopNonceCleanup();
  stopRateLimitCleanup();
  stopPresenceCleanup();
  stopEphemeralCleanup();
  stopEventAutoStart();
  if (db) shutdownSpatialEngine(db);
}

// ── CLI entry point ───────────────────────────────────

if (process.argv[1]?.replace(/\\/g, "/").includes("hive/server")) {
  const portArg = process.argv.indexOf("--port");
  const port = portArg >= 0 ? parseInt(process.argv[portArg + 1]) : 7892;
  startHiveServer({ port });
}
