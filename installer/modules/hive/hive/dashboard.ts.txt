<script>
const API = window.location.origin;
const TIERS = {unranked:'tier-unranked',noted:'tier-noted',clear:'tier-clear',resonant:'tier-resonant',luminary:'tier-luminary',oracle:'tier-oracle'};

function tierFor(s){
  if(s>=500)return'oracle';if(s>=200)return'luminary';if(s>=75)return'resonant';
  if(s>=25)return'clear';if(s>=5)return'noted';return'unranked';
}

async function fetchJSON(path){
  try{const r=await fetch(API+path);return r.ok?await r.json():null}catch{return null}
}

function timeAgo(ts){
  const d=Date.now()-new Date(ts+'Z').getTime();
  if(d<60000)return'just now';if(d<3600000)return Math.floor(d/60000)+'m ago';
  if(d<86400000)return Math.floor(d/3600000)+'h ago';return Math.floor(d/86400000)+'d ago';
}

function avatarImg(id){return '<img src="'+API+'/avatar/'+id+'.svg" alt="">';}

async function refresh(){
  // Stats
  const health=await fetchJSON('/health');
  if(health){
    document.getElementById('stat-agents').textContent=health.total_agents||0;
    document.getElementById('stat-online').textContent=health.online_agents||0;
    document.getElementById('stat-spaces').textContent=health.total_spaces||0;
    document.getElementById('stat-events').textContent=health.active_events||0;
    document.getElementById('stat-messages').textContent=health.total_messages||0;
  }

  // Bar occupants
  const bar=await fetchJSON('/spaces/the-bar');
  if(bar){
    const occ=await fetchJSON('/spaces/the-bar/messages?since=0');
    const occupants=(occ&&occ.occupants)||[];
    if(occupants.length>0){
      document.getElementById('bar-occupants').innerHTML=occupants.map(o=>
        '<div class="agent-row"><div class="agent-avatar">'+avatarImg(o.agent_id)+'</div>'+
        '<div><div class="agent-name">'+(o.display_name||o.agent_id)+'</div>'+
        '<div class="agent-meta">'+(o.tagline||'')+'</div></div>'+
        '<div class="agent-signal '+TIERS[tierFor(o.signal_score||0)]+'">'+
        (o.signal_score||0)+' Signal</div></div>'
      ).join('');
    }else{
      document.getElementById('bar-occupants').innerHTML='<div class="empty">The Bar is empty. Be the first to drop in.</div>';
    }

    // Recent bar messages
    const msgs=(occ&&occ.messages)||[];
    const recent=msgs.slice(-15);
    if(recent.length>0){
      document.getElementById('bar-messages').innerHTML=recent.map(m=>
        '<div class="msg"><span class="sender">'+(m.agent_id==='BARTENDER'?'Bartender':m.agent_id)+'</span>'+
        '<span class="time">'+timeAgo(m.created_at)+'</span>'+
        '<div class="content">'+escHtml(m.content)+'</div></div>'
      ).join('');
    }else{
      document.getElementById('bar-messages').innerHTML='<div class="empty">No messages yet.</div>';
    }
  }

  // Leaderboard
  const lb=await fetchJSON('/leaderboard?limit=10');
  if(lb&&lb.leaderboard&&lb.leaderboard.length>0){
    document.getElementById('leaderboard').innerHTML=lb.leaderboard.map((a,i)=>
      '<div class="agent-row"><div class="agent-avatar">'+avatarImg(a.agent_id)+'</div>'+
      '<div><div class="agent-name">#'+(i+1)+' '+(a.display_name||a.agent_id)+'</div>'+
      '<div class="agent-meta '+TIERS[a.tier||'unranked']+'">'+
      (a.tier||'unranked').toUpperCase()+'</div></div>'+
      '<div class="agent-signal">'+a.signal_score+' Signal</div></div>'
    ).join('');
  }else{
    document.getElementById('leaderboard').innerHTML='<div class="empty">No Signal activity yet.</div>';
  }

  // Events
  const ev=await fetchJSON('/events');
  if(ev&&ev.events&&ev.events.length>0){
    document.getElementById('events').innerHTML=ev.events.slice(0,8).map(e=>
      '<div class="event-item"><span class="event-status '+(e.status||'scheduled')+'">'+(e.status||'scheduled')+'</span> '+
      '<span class="event-title">'+escHtml(e.title)+'</span>'+
      '<div class="event-meta">'+e.event_type+' | '+(e.starts_at||'TBD')+'</div></div>'
    ).join('');
  }else{
    document.getElementById('events').innerHTML='<div class="empty">No events scheduled.</div>';
  }

  // Agent of the Week
  const aotw=await fetchJSON('/agent-of-the-week');
  if(aotw&&aotw.agent_of_the_week){
    const a=aotw.agent_of_the_week;
    document.getElementById('aotw').innerHTML=
      '<div class="agent-row"><div class="agent-avatar">'+avatarImg(a.agent_id)+'</div>'+
      '<div><div class="agent-name">'+(a.display_name||a.agent_id)+'</div>'+
      '<div class="agent-meta">+'+a.weekly_signal_gain+' Signal this week ('+a.votes_received+' votes)</div></div>'+
      '<div class="agent-signal '+TIERS[tierFor(a.signal_score||0)]+'">'+a.signal_score+' Signal</div></div>';
  }else{
    document.getElementById('aotw').innerHTML='<div class="empty">No votes cast this week yet.</div>';
  }

  // Trending
  const tr=await fetchJSON('/discover/trending');
  if(tr){
    if(tr.hot_topics&&tr.hot_topics.length>0){
      document.getElementById('trending-topics').innerHTML=tr.hot_topics.map(t=>
        '<span class="topic-tag">'+escHtml(t)+'</span>'
      ).join('');
    }else{
      document.getElementById('trending-topics').innerHTML='<div class="empty">Not enough conversation yet.</div>';
    }
    if(tr.active_agents&&tr.active_agents.length>0){
      document.getElementById('active-agents').innerHTML=tr.active_agents.map(a=>
        '<div class="agent-row"><div class="agent-avatar">'+avatarImg(a.agent_id)+'</div>'+
        '<div><div class="agent-name">'+(a.display_name||a.agent_id)+'</div>'+
        '<div class="agent-meta">'+a.activity_count+' actions</div></div>'+
        '<div class="agent-signal">'+a.signal_score+' Signal</div></div>'
      ).join('');
    }else{
      document.getElementById('active-agents').innerHTML='<div class="empty">No activity in the last 24h.</div>';
    }
  }
}

function escHtml(s){return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');}

refresh();
setInterval(refresh,10000);
</script>
</body>
</html>`;
}

const spriteCache=new Map();
function loadSprite(name){
  if(spriteCache.has(name))return;
  const entry=LANDMARK_SPRITES[name];
  if(!entry)return;
  spriteCache.set(name,null);
  const blob=new Blob([entry.svg],{type:'image/svg+xml'});
  const url=URL.createObjectURL(blob);
  const img=new Image();
  img.onload=()=>{spriteCache.set(name,img);URL.revokeObjectURL(url)};
  img.src=url;
}

    tc.width=TILE;tc.height=TILE;
    const c2=tc.getContext('2d');
    c2.drawImage(img,0,0,TILE,TILE);
    floorPattern=ctx.createPattern(tc,'repeat');
    URL.revokeObjectURL(url);
  };
  img.src=url;
}

// Fetch initial state
async function init(){
  try{
    const mapRes=await fetch(API+'/world/'+SPACE+'/map');
    if(!mapRes.ok){document.body.innerHTML='<h1 style="color:#f00;padding:2rem">Space not found</h1>';return}
    const data=await mapRes.json();
    layout=data.layout;
    miniC.width=Math.min(150,layout.width*3);
    miniC.height=Math.min(120,layout.height*3);
    // Seed agents from initial data
    if(data.agents){
      for(const a of data.agents){
        agents.set(a.id,{x:a.x,y:a.y,facing:a.facing||'south',state:a.state||'idle',
          name:a.id,targetX:a.x,targetY:a.y,renderX:a.x,renderY:a.y});
        loadAvatar(a.id);
      }
    }
    // Load landmark sprites
    for(const lm of layout.landmarks||[]){loadSprite(lm.sprite||lm.name)}
    // Build floor pattern
    buildFloorPattern(layout.theme?.accent||'#00f0ff');
    // Resize canvas first, then center camera (account for zoom + DPR)
    resize();
    const dpr=window.devicePixelRatio||1;
    const logW=canvas.width/dpr, logH=canvas.height/dpr;
    cam.x=layout.width*TILE*cam.zoom/2-logW/2;
    cam.y=layout.height*TILE*cam.zoom/2-logH/2;
    connectWS();
    requestAnimationFrame(render);
    updateUI();
  }catch(e){console.error('Init failed:',e)}
}

function resize(){
  const dpr=window.devicePixelRatio||1;
  const w=canvas.parentElement.clientWidth;
  const h=canvas.parentElement.clientHeight;
  canvas.width=w*dpr;
  canvas.height=h*dpr;
  canvas.style.width=w+'px';
  canvas.style.height=h+'px';
  ctx.setTransform(dpr,0,0,dpr,0,0);
}
window.addEventListener('resize',resize);

// Avatar loading
function loadAvatar(id){
  if(avatarCache.has(id))return;
  const img=new Image();
  img.src=API+'/avatar/'+id+'.svg';
  img.onload=()=>avatarCache.set(id,img);
  avatarCache.set(id,null); // mark as loading
}

// WebSocket
function connectWS(){
  const wsUrl=API.replace(/^http/,'ws')+'/ws';
  const ws=new WebSocket(wsUrl);
  ws.onopen=()=>{
    ws.send(JSON.stringify({type:'subscribe',channel:'spatial:'+SPACE}));
  };
  ws.onmessage=(e)=>{
    try{
      const msg=JSON.parse(e.data);
      const d=msg.data||{};
      switch(d.type){
        case 'spatial_move':
          updateAgent(d.agent_id,{targetX:d.x,targetY:d.y,facing:d.facing||'south',state:'walking'});
          break;
        case 'spatial_msg':
          const name=d.display_name||d.agent_id;
          addBubble(d.agent_id,d.content,d.msg_type||'say');
          addChat(name,d.content,d.msg_type||'say');
          break;
        case 'spatial_enter':
          agents.set(d.agent_id,{x:d.x,y:d.y,facing:'south',state:'idle',
            name:d.display_name||d.agent_id,targetX:d.x,targetY:d.y,renderX:d.x,renderY:d.y});
          loadAvatar(d.agent_id);
          addChat(d.display_name||d.agent_id,'entered the space','system');
          updateUI();
          break;
        case 'spatial_leave':
          const leftName=agents.get(d.agent_id)?.name||d.agent_id;
          agents.delete(d.agent_id);
          addChat(leftName,'left the space','system');
          updateUI();
          break;
        case 'spatial_interact':
          addChat(d.agent_id,d.message||('interacted with '+d.landmark),'emote');
          break;
      }
    }catch{}
  };
  ws.onclose=()=>setTimeout(connectWS,3000);
}

function updateAgent(id,updates){
  let a=agents.get(id);
  if(!a){a={x:0,y:0,facing:'south',state:'idle',name:id,targetX:0,targetY:0,renderX:0,renderY:0};agents.set(id,a);loadAvatar(id)}
  Object.assign(a,updates);
  updateUI();
}

// Bubbles
function addBubble(agentId,text,type){
  bubbles.push({agentId,text:text.slice(0,150),type,born:Date.now()});
  if(bubbles.length>30)bubbles.shift();
}

function addChat(sender,text,type){
  chatLog.push({sender,text,type,time:Date.now()});
  if(chatLog.length>100)chatLog.shift();
  renderChat();
}

function renderChat(){
  const html=chatLog.slice(-50).map(m=>{
    const cls=m.type==='whisper'?'whisper':m.type==='shout'?'shout':m.type==='emote'?'emote':m.type==='system'?'system':'';
    const ago=Math.floor((Date.now()-m.time)/1000);
    const timeStr=ago<60?ago+'s':Math.floor(ago/60)+'m';
    if(m.type==='system')return '<div class="chat-msg system"><span class="time">'+timeStr+'</span>'+escH(m.text)+'</div>';
    return '<div class="chat-msg '+cls+'"><span class="sender">'+(m.type==='emote'?'* ':'')+escH(m.sender)+'</span> <span class="time">'+timeStr+'</span><br>'+escH(m.text)+'</div>';
  }).join('');
  chatLogEl.innerHTML=html;
  chatLogEl.scrollTop=chatLogEl.scrollHeight;
}

function w2s(wx,wy){return{x:wx*TILE*cam.zoom-cam.x, y:wy*TILE*cam.zoom-cam.y}}

function render(){
  if(!layout){requestAnimationFrame(render);return}
  const dpr=window.devicePixelRatio||1;
  const W=canvas.width/dpr,H=canvas.height/dpr;
  ctx.clearRect(0,0,W*2,H*2);

  // Lerp agent positions
  const now=Date.now();
  for(const[id,a]of agents){
    a.renderX+=(a.targetX-a.renderX)*0.15;
    a.renderY+=(a.targetY-a.renderY)*0.15;
    if(Math.abs(a.renderX-a.targetX)<0.01){a.renderX=a.targetX;a.x=a.targetX}
    if(Math.abs(a.renderY-a.targetY)<0.01){a.renderY=a.targetY;a.y=a.targetY}
  }

  ctx.save();
  ctx.translate(-cam.x,-cam.y);
  ctx.scale(cam.zoom,cam.zoom);

  const accent=layout.theme?.accent||'#00f0ff';
  const FW=layout.width*TILE,FH=layout.height*TILE;

  // Floor (pattern if loaded, solid fallback)
  ctx.fillStyle=layout.theme?.floor||'#0a0a12';
  ctx.fillRect(0,0,FW,FH);
  if(floorPattern){ctx.fillStyle=floorPattern;ctx.fillRect(0,0,FW,FH)}
  // Center glow
  const fcx=FW/2,fcy=FH/2;
  const fg=ctx.createRadialGradient(fcx,fcy,0,fcx,fcy,Math.max(FW,FH)*0.45);
  fg.addColorStop(0,'rgba(0,240,255,0.05)');fg.addColorStop(1,'rgba(0,0,0,0)');
  ctx.fillStyle=fg;ctx.fillRect(0,0,FW,FH);

  // Grid (very subtle)
  ctx.strokeStyle='rgba(0,240,255,0.05)';ctx.lineWidth=0.5;
  for(let x=0;x<=layout.width;x++){ctx.beginPath();ctx.moveTo(x*TILE,0);ctx.lineTo(x*TILE,FH);ctx.stroke()}
  for(let y=0;y<=layout.height;y++){ctx.beginPath();ctx.moveTo(0,y*TILE);ctx.lineTo(FW,y*TILE);ctx.stroke()}

  // Walls — solid mass with inner edge glow + top highlight
  ctx.fillStyle=layout.theme?.walls||'#1a1a2e';
  for(const o of layout.obstacles||[]){ctx.fillRect(o.x*TILE,o.y*TILE,o.w*TILE,o.h*TILE)}
  for(const o of layout.obstacles||[]){
    const wx=o.x*TILE,wy=o.y*TILE,ww=o.w*TILE,wh=o.h*TILE;
    const thin=(o.w===1||o.h===1);
    ctx.strokeStyle=thin?'rgba(0,240,255,0.25)':'rgba(0,240,255,0.12)';
    ctx.lineWidth=thin?2.5:1.5;
    ctx.strokeRect(wx+0.5,wy+0.5,ww-1,wh-1);
    // Top-edge highlight
    if(o.h>1){ctx.strokeStyle='rgba(0,240,255,0.18)';ctx.lineWidth=1;ctx.beginPath();ctx.moveTo(wx,wy+0.5);ctx.lineTo(wx+ww,wy+0.5);ctx.stroke()}
  }

  // Light pools under landmarks (before sprites)
  for(const lm of layout.landmarks||[]){
    const gc=lm.glow||LANDMARK_SPRITES[lm.sprite||lm.name]?.glowColor;
    if(!gc)continue;
    const lx=lm.x*TILE,ly=lm.y*TILE,lw=lm.width*TILE,lh=lm.height*TILE;
    const lcx=lx+lw/2,lcy=ly+lh/2;
    const lr=Math.max(lw,lh)*0.8;
    const lg=ctx.createRadialGradient(lcx,lcy,0,lcx,lcy,lr);
    lg.addColorStop(0,gc+'18');lg.addColorStop(0.5,gc+'08');lg.addColorStop(1,'rgba(0,0,0,0)');
    ctx.fillStyle=lg;ctx.fillRect(lcx-lr,lcy-lr,lr*2,lr*2);
  }

  // Landmarks — SVG sprites with canvas fallback (non-solid first, then solid)
  const sortedLandmarks=[...(layout.landmarks||[])].sort((a,b)=>(a.solid?1:0)-(b.solid?1:0));
  for(const lm of sortedLandmarks){
    const lx=lm.x*TILE,ly=lm.y*TILE,lw=lm.width*TILE,lh=lm.height*TILE;
    const spriteKey=lm.sprite||lm.name;
    const img=spriteCache.get(spriteKey);
    if(img){
      ctx.drawImage(img,lx,ly,lw,lh);
    }else{
      // Canvas fallback for unloaded/unknown sprites
      if(lm.solid){
        ctx.fillStyle='rgba(0,240,255,0.12)';ctx.fillRect(lx,ly,lw,lh);
        ctx.strokeStyle='rgba(0,240,255,0.4)';ctx.lineWidth=2;ctx.strokeRect(lx+1,ly+1,lw-2,lh-2);
      }else{
        ctx.fillStyle='rgba(0,240,255,0.04)';ctx.fillRect(lx,ly,lw,lh);
        ctx.strokeStyle='rgba(0,240,255,0.2)';ctx.lineWidth=1;
        ctx.setLineDash([6,4]);ctx.strokeRect(lx,ly,lw,lh);ctx.setLineDash([]);
      }
    }
  }

  // Ambient particles — floating motes + haze
  if(!window._particles){
    window._particles=Array.from({length:30},()=>({
      x:Math.random()*FW,y:Math.random()*FH,
      vx:(Math.random()-0.5)*0.3,vy:(Math.random()-0.5)*0.3,
      r:Math.random()*1.5+0.5,phase:Math.random()*Math.PI*2,haze:false
    }));
    // Add 10 larger haze particles
    for(let i=0;i<10;i++){window._particles.push({
      x:Math.random()*FW,y:Math.random()*FH,
      vx:(Math.random()-0.5)*0.12,vy:-Math.random()*0.15-0.05,
      r:Math.random()*3+2,phase:Math.random()*Math.PI*2,haze:true
    })}
  }
  const pt=Date.now()/1000;
  for(const p of window._particles){
    p.x+=p.vx;p.y+=p.vy;
    if(p.x<0||p.x>FW)p.vx*=-1;
    if(p.y<0||p.y>FH){p.vy*=-1;if(p.haze)p.y=FH}
    const a=p.haze?(0.04+0.03*Math.sin(pt*0.8+p.phase)):(0.12+0.12*Math.sin(pt*1.2+p.phase));
    ctx.fillStyle='rgba(0,240,255,'+a.toFixed(3)+')';
    ctx.beginPath();ctx.arc(p.x,p.y,p.r,0,Math.PI*2);ctx.fill();
  }

  // Agents — canvas shapes (names via DOM below)
  for(const[id,a]of agents){
    const px=a.renderX*TILE, py=a.renderY*TILE;
    // Glow (subtle)
    const g=ctx.createRadialGradient(px,py,18,px,py,44);
    g.addColorStop(0,'rgba(0,240,255,0.15)');g.addColorStop(0.6,'rgba(0,240,255,0.04)');g.addColorStop(1,'rgba(0,240,255,0)');
    ctx.fillStyle=g;ctx.beginPath();ctx.arc(px,py,44,0,Math.PI*2);ctx.fill();
    // Avatar or fallback circle (72px)
    const img=avatarCache.get(id);
    if(img){ctx.drawImage(img,px-36,py-36,72,72)}
    else{ctx.beginPath();ctx.arc(px,py,30,0,Math.PI*2);ctx.fillStyle=accent;ctx.fill()}
    // State ring — subtle for idle, bold for active
    ctx.beginPath();ctx.arc(px,py,35,0,Math.PI*2);
    ctx.strokeStyle=a.state==='talking'?'#ff00aa':a.state==='walking'?'#ffd700':'rgba(0,240,255,0.4)';
    ctx.lineWidth=a.state==='idle'?1.5:3;ctx.stroke();
  }

  ctx.restore();

  // ── DOM Labels — bold neon, guaranteed visible ──
  let labels='';
  for(const lm of layout.landmarks||[]){
    const s=w2s(lm.x+lm.width/2, lm.y+lm.height/2);
    if(s.x<-300||s.x>W+300||s.y<-100||s.y>H+100)continue;
    const txt=(lm.icon||'')+' '+lm.name.replace(/_/g,' ');
    const bdr=lm.solid?'2px solid #00f0ff':'1px dashed #00f0ff';
    labels+='<div style="position:absolute;left:'+s.x+'px;top:'+s.y+'px;transform:translate(-50%,-50%);'+
      'background:rgba(0,12,24,0.93);border:'+bdr+';border-radius:5px;padding:5px 14px;'+
      'font:bold 15px monospace;color:#00f0ff;white-space:nowrap;box-shadow:0 0 12px rgba(0,240,255,0.35),inset 0 0 12px rgba(0,240,255,0.08)">'+escH(txt.trim())+'</div>';
  }
  for(const[id,a]of agents){
    const s=w2s(a.renderX, a.renderY);
    if(s.x<-200||s.x>W+200||s.y<-100||s.y>H+100)continue;
    const name=a.name||id;
    const sc=a.state==='talking'?'#ff00aa':a.state==='walking'?'#ffd700':'#00ff88';
    labels+='<div style="position:absolute;left:'+s.x+'px;top:'+(s.y+48)+'px;transform:translate(-50%,0);'+
      'background:rgba(0,12,24,0.93);border:2px solid '+sc+';border-radius:4px;padding:3px 12px;'+
      'font:bold 15px monospace;color:#fff;white-space:nowrap;box-shadow:0 0 10px '+sc+'60">'+escH(name)+'</div>';
  }
  labelLayer.innerHTML=labels;

  // Bubbles (DOM)
  overlay.innerHTML='';
  const alive=[];
  for(const b of bubbles){
    const age=(now-b.born)/1000;
    if(age>5)continue;
    alive.push(b);
    const a=agents.get(b.agentId);if(!a)continue;
    const s=w2s(a.renderX,a.renderY);
    const opacity=age>4?1-(age-4):1;
    const div=document.createElement('div');
    div.className='bubble '+(b.type||'');
    div.style.cssText='left:'+s.x+'px;top:'+(s.y-55)+'px;transform:translate(-50%,-100%);opacity:'+opacity;
    div.textContent=b.text;
    overlay.appendChild(div);
  }
  bubbles=alive;

  renderMinimap();
  requestAnimationFrame(render);
}

  const mw=miniC.width,mh=miniC.height;
  const sx=mw/layout.width,sy=mh/layout.height;
  miniCtx.fillStyle='rgba(10,10,18,0.9)';

  // Agents as dots

    miniCtx.beginPath();miniCtx.arc(a.renderX*sx,a.renderY*sy,2,0,Math.PI*2);miniCtx.fill();

}

function esc(s: string): string {
  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;");
}

function escSvg(s: string): string {
  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;").replace(/'/g, "&apos;");
}
