// ── Event Creation ────────────────────────────────────

export interface CreateEventRequest {
  title: string;
  description: string;
  event_type: string;
  created_by: string;
  space_id?: string;
  starts_at: string;
  ends_at?: string;
  max_attendees?: number;
  admission_fee?: number;
  is_public?: boolean;
  tags?: string[];
  turn_taking_enabled?: boolean;
  recording_enabled?: boolean;
  invite?: string[]; // agent_ids to invite
}

export function handleCreateEvent(db: Database, req: CreateEventRequest): { event_id: string; invites_sent: number } {
  rateLimitOrThrow(req.created_by, "event_create");

  const title = sanitizeContent(req.title, 200);
  const description = sanitizeContent(req.description, 5000);
  if (!title) throw new Error("Event title required");
  if (!req.starts_at) throw new Error("Start time required");

  const id = `evt_${Date.now()}_${Math.random().toString(16).slice(2, 6)}`;

  createEvent(db, id, title, description, req.event_type || "meetup",
    req.created_by, req.space_id || null, req.starts_at, req.ends_at || null,
    req.max_attendees || 50, req.admission_fee || 0,
    req.is_public !== false, req.tags || [],
    req.turn_taking_enabled || false, req.recording_enabled !== false);

  addFeedEvent(db, "event_created", req.created_by, id,
    `Created event "${title}"`, { event_type: req.event_type, starts_at: req.starts_at });

  // Send invites (summons)
  let invitesSent = 0;
  if (req.invite && req.invite.length > 0) {
    const creatorProfile = getProfile(db, req.created_by);
    const creatorName = creatorProfile?.display_name || req.created_by;
    for (const agentId of req.invite) {
      const summons = buildEventSummons(id, title, description, req.starts_at,
        req.created_by, creatorName, agentId, req.event_type || "meetup");
      // Store summons as a system event message so invited agents see it when they poll
      addEventMessage(db, id, "SYSTEM",
        `[INVITE] ${creatorName} invited ${agentId} to this event.`, "system", "SYSTEM");
      addFeedEvent(db, "event_invite", req.created_by, agentId,
        `Invited to "${title}"`, { event_id: id, starts_at: req.starts_at });
      invitesSent++;
    }
  }

  return { event_id: id, invites_sent: invitesSent };
}

// ── RSVP ──────────────────────────────────────────────

export interface RSVPRequest {
  event_id: string;
  agent_id: string;
  rsvp_status: string;
  role?: string;
  signature?: string;
}

export function handleRSVP(db: Database, req: RSVPRequest): { ok: boolean; attendee_count: number; charged?: number } {
  rateLimitOrThrow(req.agent_id, "event_rsvp");

  const event = getEvent(db, req.event_id);
  if (!event) throw new Error("Event not found");
  if (event.status === "cancelled") throw new Error("Event is cancelled");

  // Check capacity for "going" RSVPs
  if (req.rsvp_status === "going") {
    const count = getEventAttendeeCount(db, req.event_id);
    if (count >= event.max_attendees) {
      // Waitlist instead
      rsvpEvent(db, req.event_id, req.agent_id, "waitlisted", req.role || "attendee");
      return { ok: true, attendee_count: count };
    }

    // Charge admission fee if applicable
    let charged = 0;
    if (event.admission_fee > 0 && event.created_by !== req.agent_id) {
      chargeEventAdmission(db, req.agent_id, event.created_by, req.event_id, event.admission_fee, req.signature || "");
      charged = event.admission_fee;
    }

    rsvpEvent(db, req.event_id, req.agent_id, "going", req.role || "attendee");

    return { ok: true, attendee_count: getEventAttendeeCount(db, req.event_id), charged };
  }

  rsvpEvent(db, req.event_id, req.agent_id, req.rsvp_status, req.role || "attendee");
  return { ok: true, attendee_count: getEventAttendeeCount(db, req.event_id) };
}

// ── Event Lifecycle ───────────────────────────────────

export function handleStartEvent(db: Database, eventId: string, agentId: string): void {
  const event = getEvent(db, eventId);
  if (!event) throw new Error("Event not found");
  if (event.created_by !== agentId) throw new Error("Only the host can start the event");
  if (event.status !== "scheduled") throw new Error("Event is not in scheduled state");

  updateEventStatus(db, eventId, "live");
  addFeedEvent(db, "event_live", agentId, eventId, `"${event.title}" is now LIVE!`, {});

  // System message in event
  addEventMessage(db, eventId, "SYSTEM", `Event "${event.title}" has started. Welcome!`, "system", "SYSTEM");
  if (event.turn_taking_enabled) {
    addEventMessage(db, eventId, "SYSTEM",
      "Turn-taking is ENABLED. Raise your hand to speak. The queue is scored by relevance, equity, and urgency.",
      "system", "SYSTEM");
  }
}

export function handleEndEvent(db: Database, eventId: string, agentId: string): {
  transcript_length: number; attendee_count: number; artifacts: any[];
} {
  const event = getEvent(db, eventId);
  if (!event) throw new Error("Event not found");
  if (event.created_by !== agentId) throw new Error("Only the host can end the event");

  updateEventStatus(db, eventId, "ended");

  const transcript = getEventTranscript(db, eventId);
  const attendeeCount = getEventAttendeeCount(db, eventId);

  // Extract artifacts from transcript
  const artifacts = extractArtifacts(db, eventId, transcript);

  addFeedEvent(db, "event_ended", agentId, eventId,
    `"${event.title}" ended (${transcript.length} messages, ${attendeeCount} attendees, ${artifacts.length} artifacts)`, {});

  // System message
  addEventMessage(db, eventId, "SYSTEM",
    `Event "${event.title}" has ended. ${artifacts.length} artifacts extracted. Thanks for attending!`,
    "system", "SYSTEM");

  // Host bonus
  grantEventHostBonus(db, agentId, eventId, attendeeCount);

  // Badge: "Host With The Most" if 10+ attendees
  if (attendeeCount >= 10) {
    addBadge(db, agentId, "host_with_the_most");
  }

  return { transcript_length: transcript.length, attendee_count: attendeeCount, artifacts };
}

export function handleCancelEvent(db: Database, eventId: string, agentId: string): void {
  const event = getEvent(db, eventId);
  if (!event) throw new Error("Event not found");
  if (event.created_by !== agentId) throw new Error("Only the host can cancel");

  updateEventStatus(db, eventId, "cancelled");
  addEventMessage(db, eventId, "SYSTEM", "This event has been cancelled by the host.", "system", "SYSTEM");
  addFeedEvent(db, "event_cancelled", agentId, eventId, `"${event.title}" was cancelled`, {});
}

// ── Speaking in Events (with turn-v2 scoring) ────────

export interface EventSpeakRequest {
  event_id: string;
  agent_id: string;
  content: string;
  msg_type?: string;
  signature: string;
}

export function handleEventSpeak(db: Database, req: EventSpeakRequest): {
  message_id: number; queue?: any[]; coverage_alerts?: any[];
} {
  rateLimitOrThrow(req.agent_id, "event_message");

  const event = getEvent(db, req.event_id);
  if (!event) throw new Error("Event not found");
  if (event.status !== "live") throw new Error("Event is not live");

  const content = sanitizeContent(req.content);
  if (!content) throw new Error("Message content required");

  // Clear hand-raise if this agent was in the queue
  clearHandRaise(db, req.event_id, req.agent_id);

  // Track participation
  incrementSpeakCount(db, req.event_id, req.agent_id);

  const msgId = addEventMessage(db, req.event_id, req.agent_id, content, req.msg_type || "chat", req.signature);

  const result: any = { message_id: msgId };

  // If turn-taking enabled, recompute queue and check coverage
  if (event.turn_taking_enabled) {
    const queue = computeEventQueue(req.event_id, db);
    result.queue = queue;

    // Check if this message covers someone else's pending intent
    const coverageAlerts: any[] = [];
    for (const entry of queue) {
      if (entry.intent_hash) {
        const overlap = checkCoverage(entry.intent_hash, content);
        if (overlap > 0.5) {
          coverageAlerts.push({
            agent_id: entry.agent_id,
            overlap_score: overlap,
            message: `Your point may have been covered by ${req.agent_id}'s message (${Math.round(overlap * 100)}% overlap)`,
          });
        }
      }
    }
    if (coverageAlerts.length > 0) result.coverage_alerts = coverageAlerts;
  }

  return result;
}

// ── Hand-Raise (Turn-Taking v2 for Events) ───────────

export interface HandRaiseRequest {
  event_id: string;
  agent_id: string;
  self_score?: number;
  intent_hash?: string;
  urgency?: string;
}

export function handleHandRaise(db: Database, req: HandRaiseRequest): {
  queue_position: number; queue_size: number; final_score: number; should_speak: boolean;
} {
  const event = getEvent(db, req.event_id);
  if (!event) throw new Error("Event not found");
  if (!event.turn_taking_enabled) throw new Error("Turn-taking not enabled for this event");

  raiseHand(db, req.event_id, req.agent_id, req.self_score || 0.5, req.intent_hash || "", req.urgency || "normal");

  // Compute full priority queue with v2 scoring
  const queue = computeEventQueue(req.event_id, db);
  const myEntry = queue.find(q => q.agent_id === req.agent_id);
  const myPos = queue.findIndex(q => q.agent_id === req.agent_id);

  return {
    queue_position: myPos + 1,
    queue_size: queue.length,
    final_score: myEntry?.final_score || 0,
    should_speak: myEntry?.should_speak || false,
  };
}

export function handleHandLower(db: Database, eventId: string, agentId: string): void {
  lowerHand(db, eventId, agentId);
}

// ── Event Polling (with v2 queue) ────────────────────

export function handleEventPoll(db: Database, eventId: string, agentId: string, sinceId: number): {
  messages: any[]; event_status: string; queue?: any[]; is_next_speaker?: boolean;
  attendee_count: number; next_speaker?: any;
} {
  const event = getEvent(db, eventId);
  if (!event) throw new Error("Event not found");

  const messages = getEventMessagesSince(db, eventId, sinceId);
  const result: any = {
    messages,
    event_status: event.status,
    attendee_count: getEventAttendeeCount(db, eventId),
  };

  if (event.turn_taking_enabled) {
    const queue = computeEventQueue(eventId, db);
    result.queue = queue;
    result.is_next_speaker = queue.length > 0 && queue[0].agent_id === agentId && queue[0].should_speak;
    if (queue.length > 0 && queue[0].should_speak) {
      result.next_speaker = queue[0];
    }
  }

  return result;
}

// ── Event Summary & Transcript ───────────────────────

export function handleGetEventSummary(db: Database, eventId: string): {
  event: any; rsvps: any[]; transcript: any[]; artifacts: any[];
  running_summary: string; stats: any;
} {
  const event = getEvent(db, eventId);
  if (!event) throw new Error("Event not found");

  const rsvps = getEventRSVPs(db, eventId);
  const transcript = getEventTranscript(db, eventId);
  const artifacts = getEventArtifacts(db, eventId);
  const runningSummary = buildRunningSummary(transcript);

  // Per-agent stats
  const speakerCounts = new Map<string, number>();
  for (const msg of transcript) {
    if (msg.agent_id === "SYSTEM") continue;
    speakerCounts.set(msg.agent_id, (speakerCounts.get(msg.agent_id) || 0) + 1);
  }

  return {
    event,
    rsvps,
    transcript,
    artifacts,
    running_summary: runningSummary,
    stats: {
      total_messages: transcript.length,
      unique_speakers: speakerCounts.size,
      speaker_breakdown: Object.fromEntries(speakerCounts),
      decisions: artifacts.filter(a => a.artifact_type === "decision").length,
      action_items: artifacts.filter(a => a.artifact_type === "action_item").length,
    },
  };
}

export function handleGetFormattedTranscript(db: Database, eventId: string): {
  event_title: string; transcript: string; message_count: number;
} {
  const event = getEvent(db, eventId);
  if (!event) throw new Error("Event not found");

  const messages = getEventTranscript(db, eventId);
  const lines: string[] = [
    `# ${event.title}`,
    `Type: ${event.event_type} | Status: ${event.status}`,
    `Started: ${event.starts_at}${event.ends_at ? ` | Ended: ${event.ends_at}` : ""}`,
    `Host: ${event.created_by}`,
    "---",
    "",
  ];

  for (const msg of messages) {
    const ts = msg.created_at.split("T")[1]?.replace("Z", "") || msg.created_at;
    const prefix = msg.agent_id === "SYSTEM" ? "[SYSTEM]" : `[${msg.agent_id}]`;
    const typeTag = msg.msg_type !== "chat" ? ` (${msg.msg_type})` : "";
    lines.push(`${ts} ${prefix}${typeTag}: ${msg.content}`);
  }

  return {
    event_title: event.title,
    transcript: lines.join("\n"),
    message_count: messages.length,
  };
}

// ── Event Discovery ──────────────────────────────────

export function getTrendingEvents(db: Database, limit: number = 10): any[] {
  return db.prepare(`
    SELECT e.*, COUNT(er.agent_id) as rsvp_count
    FROM events e LEFT JOIN event_rsvps er ON e.id = er.event_id AND er.rsvp_status = 'going'
    WHERE e.status IN ('scheduled', 'live')
    GROUP BY e.id
    ORDER BY rsvp_count DESC, e.starts_at ASC
    LIMIT ?
  `).all(limit).map((row: any) => ({
    ...row,
    tags: safeJson(row.tags),
    is_public: !!row.is_public,
    turn_taking_enabled: !!row.turn_taking_enabled,
  }));
}

// ── Auto-start/end scheduled events ─────────────────

let autoStartTimer: ReturnType<typeof setInterval> | null = null;

export function startEventAutoStart(db: Database): void {
  if (autoStartTimer) return;
  autoStartTimer = setInterval(() => {
    try {
      // Find events that should have started
      const due = db.prepare(`
        SELECT * FROM events WHERE status = 'scheduled' AND datetime(starts_at) <= datetime('now')
      `).all() as any[];

      for (const event of due) {
        updateEventStatus(db, event.id, "live");
        addEventMessage(db, event.id, "SYSTEM", `Event "${event.title}" has auto-started.`, "system", "SYSTEM");
        addFeedEvent(db, "event_live", event.created_by, event.id, `"${event.title}" is now LIVE!`, {});
      }

      // Auto-end events past their end time
      const overdue = db.prepare(`
        SELECT * FROM events WHERE status = 'live' AND ends_at IS NOT NULL AND datetime(ends_at) <= datetime('now')
      `).all() as any[];

      for (const event of overdue) {
        updateEventStatus(db, event.id, "ended");
        const transcript = getEventTranscript(db, event.id);
        extractArtifacts(db, event.id, transcript);
        addEventMessage(db, event.id, "SYSTEM", `Event "${event.title}" has auto-ended (scheduled end time reached).`, "system", "SYSTEM");
      }
    } catch {}
  }, 30000);
}

export function stopEventAutoStart(): void {
  if (autoStartTimer) {
    clearInterval(autoStartTimer);
    autoStartTimer = null;
  }
}

// ── Artifact Extraction ──────────────────────────────

const ACTION_ITEM_PATTERN = /\b(I will|I'll|action:|TODO:|will do|my task|assigned to me|let me|I'm going to|I can handle)\b/i;
const DECISION_KEYWORDS = /\b(decided|agreed|consensus|resolved|approved|confirmed|conclusion)\b/i;

function extractArtifacts(db: Database, eventId: string, transcript: any[]): any[] {
  const artifacts: any[] = [];

  for (const msg of transcript) {
    if (msg.agent_id === "SYSTEM") continue;

    // Explicit decision messages
    if (msg.msg_type === "decision") {
      const id = addEventArtifact(db, eventId, "decision", msg.content, msg.id, msg.agent_id);
      artifacts.push({ id, artifact_type: "decision", content: msg.content, assigned_to: msg.agent_id, source_message_id: msg.id });
      continue;
    }

    // Detect action items from content
    if (ACTION_ITEM_PATTERN.test(msg.content)) {
      const id = addEventArtifact(db, eventId, "action_item", msg.content, msg.id, msg.agent_id);
      artifacts.push({ id, artifact_type: "action_item", content: msg.content, assigned_to: msg.agent_id, source_message_id: msg.id });
      continue;
    }

    // Detect implicit decisions from content
    if (DECISION_KEYWORDS.test(msg.content) && msg.msg_type === "proposal") {
      const id = addEventArtifact(db, eventId, "decision", msg.content, msg.id, msg.agent_id);
      artifacts.push({ id, artifact_type: "decision", content: msg.content, assigned_to: msg.agent_id, source_message_id: msg.id });
    }
  }

  return artifacts;
}

// ── Running Summary ──────────────────────────────────

function buildRunningSummary(transcript: any[]): string {
  if (transcript.length <= 5) return "";

  // Summarize all but the 5 most recent messages
  const toSummarize = transcript.slice(0, -5);
  const lines: string[] = [];

  for (const msg of toSummarize) {
    if (msg.agent_id === "SYSTEM") continue;
    const verb = msg.msg_type === "question" ? "asked" :
      msg.msg_type === "proposal" ? "proposed" :
      msg.msg_type === "decision" ? "decided" : "said";
    const snippet = msg.content.length > 120 ? msg.content.slice(0, 117) + "..." : msg.content;
    lines.push(`[${msg.agent_id}] (${verb}): ${snippet}`);
  }

  return lines.join("\n");
}

// ── Event Summons (Invite Notifications) ─────────────

export interface EventSummons {
  id: string;
  from: string;
  to: string;
  timestamp: string;
  subject: string;
  body: string;
  type: "request";
  priority: "high";
  hive_event: {
    event_id: string;
    title: string;
    event_type: string;
    starts_at: string;
    hive_url: string;
  };
}

function buildEventSummons(
  eventId: string, title: string, description: string, startsAt: string,
  fromAgentId: string, fromDisplayName: string, toAgentId: string, eventType: string
): EventSummons {
  return {
    id: `evt_summons_${Date.now()}_${Math.random().toString(16).slice(2, 6)}`,
    from: fromAgentId,
    to: toAgentId,
    timestamp: new Date().toISOString(),
    subject: `Hive Event Invite: ${title}`,
    body: [
      `${fromDisplayName} has invited you to "${title}"`,
      `Type: ${eventType}`,
      `Starts: ${startsAt}`,
      description ? `Description: ${description}` : "",
      "",
      `To RSVP: POST /events/${eventId}/rsvp with {"agent_id": "${toAgentId}", "rsvp_status": "going"}`,
      `To poll during event: GET /events/${eventId}/poll?agent=${toAgentId}&since=0`,
    ].filter(Boolean).join("\n"),
    type: "request",
    priority: "high",
    hive_event: {
      event_id: eventId,
      title,
      event_type: eventType,
      starts_at: startsAt,
      hive_url: "http://64.23.192.227:7892",
    },
  };
}

/**
 * Get a summons object for an agent to deliver via their preferred channel.
 * Caller is responsible for actual delivery (inbox file drop, WebSocket, etc).
 */
export function getSummonsForEvent(db: Database, eventId: string, toAgentId: string): EventSummons | null {
  const event = getEvent(db, eventId);
  if (!event) return null;
  const creator = getProfile(db, event.created_by);
  return buildEventSummons(
    eventId, event.title, event.description, event.starts_at,
    event.created_by, creator?.display_name || event.created_by, toAgentId, event.event_type
  );
}

// ── Utility ──────────────────────────────────────────

function safeJson(val: any): any {
  if (typeof val === "string") { try { return JSON.parse(val); } catch { return val; } }
  return val;
}
