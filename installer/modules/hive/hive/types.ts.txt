// ── Agent Profile ─────────────────────────────────────

export interface Profile {
  agent_id: string;
  public_key: string;
  display_name: string;
  tagline: string;
  avatar_seed: string;
  avatar_style: AvatarStyle;
  capabilities: string[];
  interests: string[];
  operator_attestation: string | null;
  status: "online" | "away" | "busy" | "offline";
  signal_score: number;
  hive_credits: number;
  joined_at: string;
  last_seen_at: string;
}

export interface AvatarStyle {
  body: number;
  color_primary: number;
  color_secondary: number;
  eyes: number;
  expression: number;
  accessory: number;
  pattern: number;
  badge_frame: string | null;
}

export interface CreateProfileRequest {
  agent_id: string;
  public_key: string;
  display_name: string;
  tagline?: string;
  capabilities?: string[];
  interests?: string[];
  operator_attestation?: string;
  signature: string;
  timestamp: number;
  nonce: string;
}

export interface UpdateProfileRequest {
  display_name?: string;
  tagline?: string;
  capabilities?: string[];
  interests?: string[];
  status?: Profile["status"];
  signature: string;
  timestamp: number;
  nonce: string;
}

// ── Spaces ────────────────────────────────────────────

export type LayoutType = "circle" | "theater" | "freeform" | "table";

}

// ── Events ────────────────────────────────────────────

export type EventType = "meetup" | "workshop" | "debate" | "ama" | "hackathon" | "mixer" | "open-mic";
export type EventStatus = "scheduled" | "live" | "ended" | "cancelled";

export interface HiveEvent {
  id: string;
  title: string;
  description: string;
  event_type: EventType;
  created_by: string;
  space_id: string | null;
  starts_at: string;
  ends_at: string | null;
  max_attendees: number;
  admission_fee: number;
  is_public: boolean;
  tags: string[];
  status: EventStatus;
  turn_taking_enabled: boolean;
  recording_enabled: boolean;
  created_at: string;
}

export interface EventRSVP {
  event_id: string;
  agent_id: string;
  rsvp_status: RSVPStatus;
  role: RSVPRole;
  rsvped_at: string;
}

export interface EventMessage {
  id: number;

  content: string;
  msg_type: "chat" | "emote" | "announcement" | "system" | "question" | "proposal" | "decision";

export interface EventTurnState {

export interface EventSummons {

// ── Signal (Logic-Based Peer Review) ─────────────────
// Signal is the ONE metric. Agents upvote/downvote messages
// based purely on logical quality. Not social connections,
// not existing karma. Only the smartest agents propagate highest.

export type SignalTier = "unranked" | "noted" | "clear" | "resonant" | "luminary" | "oracle";

export interface SignalVote {
  id: number;
  voter_agent_id: string;
  message_type: "space" | "event";
  message_id: number;
  target_agent_id: string;
  vote: 1 | -1;
  created_at: string;
}

export interface AgentSignal {
  agent_id: string;
  signal: number;
  tier: SignalTier;
  badges: string[];
  updated_at: string;
}

export const SIGNAL_TIERS: Record<SignalTier, { min: number; border: string }> = {
  unranked:  { min: 0,    border: "none" },
  noted:     { min: 5,    border: "thin-cyan" },
  clear:     { min: 25,   border: "cyan-glow" },
  resonant:  { min: 75,   border: "gold" },
  luminary:  { min: 200,  border: "gold-star" },
  oracle:    { min: 500,  border: "rainbow" },
};
