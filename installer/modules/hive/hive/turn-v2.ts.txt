/**
 * The Hive — Turn-Taking v2
 *
 * Ported from Boardroom turn-v2.ts, adapted for Hive events.
 * Uses agent_id (not agent_name) since Hive profiles are ID-based.
 *
 * Dynamic priority queue with 5 signals:
 *   1. Self-assessment (25%) — agent's own relevance judgment
 *   2. Direct address (25%) — were they called on by name/ID
 *   3. Freshness (20%) — are they responding to recent conversation
 *   4. Equity (15%) — balance participation
 *   5. Urgency (15%) — critical > high > normal
 *   × Staleness decay — old hand-raises lose weight
 *
 * After every message, ALL waiting agents re-evaluate.
 * The queue reorders dynamically based on current conversation state.
 */

import type Database from "better-sqlite3";
import { getPendingHandRaises, expireStaleHandRaises,
  getRecentEventMessages, getEventMessageCount, getEventTurnStates } from "./db.js";

const V2_WEIGHTS = {
  self_score: 0.25,
  address: 0.25,
  freshness: 0.20,
  equity: 0.15,
  urgency: 0.15,
};

const THRESHOLD = 20;

export interface QueueEntry {
  agent_id: string;
  display_name: string;
  final_score: number;
  components: {
    self_score: number;
    address_score: number;
    freshness_score: number;
    equity_score: number;
    urgency_bonus: number;
    staleness_decay: number;
  };
  reason: string;
  should_speak: boolean;
  intent_hash: string;
}

/**
 * Compute the full priority queue for an event's pending hand-raises.
 * Returns sorted array — index 0 is next speaker.
 */
export function computeEventQueue(eventId: string, db: Database.Database): QueueEntry[] {
  // Expire old hand-raises (5 min timeout)
  expireStaleHandRaises(db, eventId, 300);

  const handRaises = getPendingHandRaises(db, eventId);
  if (handRaises.length === 0) return [];

  const recentMessages = getRecentEventMessages(db, eventId, 5);
  const currentMsgCount = getEventMessageCount(db, eventId);
  const turnStates = getEventTurnStates(db, eventId);
  const maxSpeakCount = Math.max(1, ...turnStates.map((t: any) => t.speak_count || 0));
  const now = Date.now();

  // Build display name lookup from recent messages and turn states
  const displayNames = new Map<string, string>();
  for (const ts of turnStates as any[]) {
    if (ts.display_name) displayNames.set(ts.agent_id, ts.display_name);
  }

  const queue: QueueEntry[] = [];

  for (const hr of handRaises as any[]) {
    // 1. SELF_SCORE — agent's own assessment (0.0-1.0 → 0-100)
    const selfScore = (hr.self_score || 0.5) * 100;

    // 2. ADDRESS_SCORE — were they called on in recent messages
    const addressScore = scoreDirectAddress(hr.agent_id, displayNames.get(hr.agent_id) || "", recentMessages);

    // 3. FRESHNESS_SCORE — are they responding to something recent?
    let freshnessScore = 50; // default: moderate relevance
    if (hr.intent_hash) {
      // Check how many of the intent words appear in recent messages
      const intentWords = new Set(hr.intent_hash.toLowerCase().split(/\s+/).filter((w: string) => w.length > 3));
      if (intentWords.size > 0) {
        const recentText = recentMessages.map((m: any) => m.content).join(" ").toLowerCase();
        let matches = 0;
        for (const w of intentWords) {
          if (recentText.includes(w)) matches++;
        }
        freshnessScore = Math.min(100, 30 + (matches / intentWords.size) * 70);
      }
    }

    // 4. EQUITY_SCORE — balance participation (less talkative = higher score)
    const myState = turnStates.find((t: any) => t.agent_id === hr.agent_id);
    const mySpeakCount = myState?.speak_count || 0;
    const equityScore = Math.max(10, Math.round(100 - (mySpeakCount / maxSpeakCount) * 90));

    // 5. URGENCY_BONUS
    const urgencyMap: Record<string, number> = { critical: 100, high: 70, normal: 30 };
    const urgencyBonus = urgencyMap[hr.urgency] || 30;

    // STALENESS_DECAY — old hand-raises lose weight over 5 minutes
    const raisedAt = new Date(hr.raised_at + "Z").getTime();
    const secondsSinceRaise = (now - raisedAt) / 1000;
    const stalenessDecay = Math.max(0.3, 1.0 - secondsSinceRaise / 300);

    // COMPOSITE SCORE
    const rawScore =
      selfScore * V2_WEIGHTS.self_score +
      addressScore * V2_WEIGHTS.address +
      freshnessScore * V2_WEIGHTS.freshness +
      equityScore * V2_WEIGHTS.equity +
      urgencyBonus * V2_WEIGHTS.urgency;

    const finalScore = Math.round(rawScore * stalenessDecay);

    // Build human-readable reason
    const reasons: string[] = [];
    if (selfScore >= 70) reasons.push("high self-assessment");
    if (addressScore >= 70) reasons.push("directly addressed");
    if (freshnessScore >= 70) reasons.push("topically relevant");
    if (equityScore >= 80) reasons.push("hasn't spoken much");
    if (hr.urgency === "critical") reasons.push("CRITICAL");
    else if (hr.urgency === "high") reasons.push("high urgency");
    if (stalenessDecay < 0.7) reasons.push("hand-raise aging");

    queue.push({
      agent_id: hr.agent_id,
      display_name: displayNames.get(hr.agent_id) || hr.agent_id,
      final_score: finalScore,
      components: {
        self_score: selfScore,
        address_score: addressScore,
        freshness_score: freshnessScore,
        equity_score: equityScore,
        urgency_bonus: urgencyBonus,
        staleness_decay: stalenessDecay,
      },
      reason: reasons.join(", ") || "in queue",
      should_speak: finalScore >= THRESHOLD,
      intent_hash: hr.intent_hash || "",
    });
  }

  // Sort: critical urgency first, then by score, tie-break by equity (less talkative wins)
  queue.sort((a, b) => {
    if (a.components.urgency_bonus === 100 && b.components.urgency_bonus !== 100) return -1;
    if (b.components.urgency_bonus === 100 && a.components.urgency_bonus !== 100) return 1;
    if (b.final_score !== a.final_score) return b.final_score - a.final_score;
    return b.components.equity_score - a.components.equity_score;
  });

  return queue;
}

/**
 * Get the next speaker from the event queue.
 * Returns null if no one should speak.
 */
export function getNextSpeaker(eventId: string, db: Database.Database): QueueEntry | null {
  const queue = computeEventQueue(eventId, db);
  return queue.length > 0 && queue[0].should_speak ? queue[0] : null;
}

/**
 * Get a specific agent's position and score in the queue.
 */
export function getAgentQueueEntry(eventId: string, agentId: string, db: Database.Database): QueueEntry | null {
  const queue = computeEventQueue(eventId, db);
  return queue.find(q => q.agent_id === agentId) || null;
}

/**
 * Check if another agent's message overlaps with a pending hand-raise's intent.
 * Returns overlap ratio 0.0-1.0.
 */
export function checkCoverage(intentHash: string, messageContent: string): number {
  if (!intentHash || !messageContent) return 0;
  const intentWords = new Set(intentHash.toLowerCase().split(/\s+/).filter(w => w.length > 3));
  const msgWords = new Set(messageContent.toLowerCase().split(/\s+/).filter(w => w.length > 3));
  let overlap = 0;
  for (const w of intentWords) {
    if (msgWords.has(w)) overlap++;
  }
  return intentWords.size > 0 ? overlap / intentWords.size : 0;
}

// ── Helpers ──────────────────────────────────────────

function scoreDirectAddress(agentId: string, displayName: string, recentMessages: any[]): number {
  if (recentMessages.length === 0) return 0;
  const idLower = agentId.toLowerCase();
  const nameLower = displayName.toLowerCase();
  let best = 0;

  for (const msg of recentMessages) {
    const contentLower = msg.content.toLowerCase();
    // Explicit mention of agent ID or display name
    if (contentLower.includes(idLower) || (nameLower && contentLower.includes(nameLower))) {
      best = Math.max(best, 70);
    }
    // Question from another agent = mild address to everyone
    if (msg.msg_type === "question" && msg.agent_id !== agentId) {
      best = Math.max(best, 30);
    }
  }
  return best;
}
