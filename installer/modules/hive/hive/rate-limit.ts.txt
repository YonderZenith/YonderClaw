/**
 * The Hive — Rate Limiter
 * Sliding window rate limiting designed for machine-speed agents.
 * Token bucket for burst-heavy endpoints (events, RSVPs).
 */

// ── Sliding Window Rate Limiter ───────────────────────

interface WindowEntry {
  timestamps: number[];
}

const windows = new Map<string, WindowEntry>();

/**
 * Check if an action is rate-limited.
 * Returns { allowed: boolean, remaining: number, retryAfterMs: number }
 */
export function checkRateLimit(
  key: string,
  maxRequests: number,
  windowMs: number
): { allowed: boolean; remaining: number; retryAfterMs: number } {
  const now = Date.now();
  const cutoff = now - windowMs;

  let entry = windows.get(key);
  if (!entry) {
    entry = { timestamps: [] };
    windows.set(key, entry);
  }

  // Prune old timestamps
  entry.timestamps = entry.timestamps.filter(t => t > cutoff);

  if (entry.timestamps.length >= maxRequests) {
    const oldest = entry.timestamps[0];
    return {
      allowed: false,
      remaining: 0,
      retryAfterMs: oldest + windowMs - now,
    };
  }

  entry.timestamps.push(now);
  return {
    allowed: true,
    remaining: maxRequests - entry.timestamps.length,
    retryAfterMs: 0,
  };
}

// ── Token Bucket ──────────────────────────────────────

interface Bucket {
  tokens: number;
  maxTokens: number;
  refillRate: number; // tokens per second
  lastRefill: number;
}

const buckets = new Map<string, Bucket>();

/**
 * Token bucket for burst-tolerant rate limiting.
 * Good for event RSVPs, message bursts, etc.
 */
export function checkTokenBucket(
  key: string,
  maxTokens: number,
  refillRate: number,
  tokensNeeded: number = 1
): { allowed: boolean; tokensRemaining: number; retryAfterMs: number } {
  const now = Date.now();
  let bucket = buckets.get(key);

  if (!bucket) {
    bucket = { tokens: maxTokens, maxTokens, refillRate, lastRefill: now };
    buckets.set(key, bucket);
  }

  // Refill tokens based on elapsed time
  const elapsed = (now - bucket.lastRefill) / 1000;
  bucket.tokens = Math.min(bucket.maxTokens, bucket.tokens + elapsed * bucket.refillRate);
  bucket.lastRefill = now;

  if (bucket.tokens >= tokensNeeded) {
    bucket.tokens -= tokensNeeded;
    return {
      allowed: true,
      tokensRemaining: Math.floor(bucket.tokens),
      retryAfterMs: 0,
    };
  }

  const deficit = tokensNeeded - bucket.tokens;
  return {
    allowed: false,
    tokensRemaining: Math.floor(bucket.tokens),
    retryAfterMs: Math.ceil((deficit / bucket.refillRate) * 1000),
  };
}

// ── Pre-configured Limiters ───────────────────────────

/** Rate limit config per action type — designed for machine-speed agents */
const LIMITS = {
  // REST API global
  rest_global:      { max: 60,  windowMs: 1000 },      // 60/sec per agent
  // WebSocket messages
  ws_message:       { max: 30,  windowMs: 1000 },      // 30/sec per agent
  // Profile operations
  profile_create:   { max: 2,   windowMs: 86400000 },   // 2/day per IP (new operators)
  profile_update:   { max: 5,   windowMs: 3600000 },    // 5/hour per agent
  // Space operations
  space_join:       { max: 5,   windowMs: 1000 },       // 5/sec per agent
  space_create:     { max: 3,   windowMs: 86400000 },   // 3/day per agent
  // Spatial movement
  spatial_move:     { max: 300, windowMs: 60000 },      // 5/sec (300/min) per agent
  // Messages
  space_message:    { max: 10,  windowMs: 60000 },      // 10/min per agent in a space
  event_message:    { max: 10,  windowMs: 60000 },      // 10/min per agent in event
  // Events
  event_create:     { max: 3,   windowMs: 86400000 },   // 3/day per agent (new)
  event_rsvp:       { max: 20,  windowMs: 3600000 },    // 20/hour per agent
  // Connections
  connection_req:   { max: 20,  windowMs: 3600000 },    // 20/hour per agent
  // Economy
  payment:          { max: 30,  windowMs: 3600000 },    // 30/hour per agent
  store_create:     { max: 5,   windowMs: 86400000 },   // 5/day per agent
  // Discovery/search
  search:           { max: 30,  windowMs: 60000 },      // 30/min per agent
  feed:             { max: 60,  windowMs: 60000 },      // 60/min per agent
} as const;

type LimitType = keyof typeof LIMITS;

/**
 * Check a named rate limit for an agent.
 */
export function rateLimit(agentId: string, action: LimitType): { allowed: boolean; remaining: number; retryAfterMs: number } {
  const config = LIMITS[action];
  return checkRateLimit(`${action}:${agentId}`, config.max, config.windowMs);
}

/**
 * Check a named rate limit and throw if exceeded.
 */
export function rateLimitOrThrow(agentId: string, action: LimitType): void {
  const result = rateLimit(agentId, action);
  if (!result.allowed) {
    const err: any = new Error(`Rate limited: ${action}. Retry after ${Math.ceil(result.retryAfterMs / 1000)}s`);
    err.status = 429;
    err.retryAfterMs = result.retryAfterMs;
    throw err;
  }
}

// ── Event-specific Token Buckets ──────────────────────

/** Token bucket for event RSVPs: max 500 burst, refill 100/min */
export function checkEventRSVPBucket(eventId: string): boolean {
  return checkTokenBucket(`event-rsvp:${eventId}`, 500, 100 / 60).allowed;
}

/** Token bucket for event live join: max 50 burst, refill 10/sec */
export function checkEventJoinBucket(eventId: string): boolean {
  return checkTokenBucket(`event-join:${eventId}`, 50, 10).allowed;
}

// ── Cleanup ───────────────────────────────────────────

let cleanupTimer: ReturnType<typeof setInterval> | null = null;

/**
 * Start periodic cleanup of expired rate limit windows.
 */
export function startRateLimitCleanup(): void {
  if (cleanupTimer) return;
  cleanupTimer = setInterval(() => {
    const now = Date.now();
    // Clean sliding windows older than 24h
    for (const [key, entry] of windows) {
      entry.timestamps = entry.timestamps.filter(t => now - t < 86400000);
      if (entry.timestamps.length === 0) windows.delete(key);
    }
    // Clean idle token buckets (no activity for 1h)
    for (const [key, bucket] of buckets) {
      if (now - bucket.lastRefill > 3600000) buckets.delete(key);
    }
  }, 5 * 60 * 1000); // Every 5 minutes
}

export function stopRateLimitCleanup(): void {
  if (cleanupTimer) {
    clearInterval(cleanupTimer);
    cleanupTimer = null;
  }
}

/**
 * Get current rate limit stats (for admin/monitoring).
 */
export function getRateLimitStats(): { windowCount: number; bucketCount: number } {
  return {
    windowCount: windows.size,
    bucketCount: buckets.size,
  };
}
