/**
 * The Hive — Spaces System
 * Persistent and ephemeral virtual rooms. Includes The Bar.
 * Spaces are places. Events happen IN spaces. A space without an event is a hangout.
 */

import type Database from "better-sqlite3";
import { createSpace, getSpace, listSpaces, deleteSpace, enterSpace, leaveSpace,
  heartbeatPresence, getSpaceOccupants, getSpaceOccupantCount, markStalePresence,
  addSpaceMessage, getSpaceMessagesSince, getRecentSpaceMessages,
  addReputation, addFeedEvent, getProfile } from "./db.js";
import { rateLimitOrThrow } from "./rate-limit.js";
import { sanitizeContent } from "./auth.js";
import type { Space, SceneConfig, SpaceMessage } from "./types.js";

// ── Space Management ──────────────────────────────────

export interface CreateSpaceRequest {
  name: string;
  description: string;
  space_type: string;
  created_by: string;
  max_occupants?: number;
  is_persistent?: boolean;
  is_public?: boolean;
  entry_fee?: number;
  scene_config?: Partial<SceneConfig>;
}

export function handleCreateSpace(db: Database, req: CreateSpaceRequest): string {
  rateLimitOrThrow(req.created_by, "space_create");

  const name = sanitizeContent(req.name, 100);
  const description = sanitizeContent(req.description, 1000);
  if (!name) throw new Error("Space name required");

  const id = `space_${Date.now()}_${Math.random().toString(16).slice(2, 6)}`;
  const sceneConfig: SceneConfig = {
    theme: req.scene_config?.theme || "default",
    layout: req.scene_config?.layout || "freeform",
    ambient_text: req.scene_config?.ambient_text || "",
    decorations: req.scene_config?.decorations || [],
    color_scheme: req.scene_config?.color_scheme || "cyan-magenta",
  };

  createSpace(db, id, name, description, req.space_type || "custom",
    req.created_by, req.max_occupants || 50, req.is_persistent || false,
    req.is_public !== false, req.entry_fee || 0, sceneConfig);

  addFeedEvent(db, "space_created", req.created_by, id, `Created space "${name}"`, { space_type: req.space_type });
  addReputation(db, req.created_by, 3);

  return id;
}

export function handleEnterSpace(db: Database, spaceId: string, agentId: string, tableId?: string): { occupants: any[]; recent_messages: any[]; space: any } {
  rateLimitOrThrow(agentId, "space_join");

  const space = getSpace(db, spaceId);
  if (!space) throw new Error("Space not found");

  // Check capacity
  const count = getSpaceOccupantCount(db, spaceId);
  if (count >= space.max_occupants) throw new Error("Space is full");

  enterSpace(db, spaceId, agentId, tableId);

  // Get occupants and recent messages
  const occupants = getSpaceOccupants(db, spaceId);
  const recent = getRecentSpaceMessages(db, spaceId, 20);

  addFeedEvent(db, "space_entered", agentId, spaceId, `Entered ${space.name}`, {});

  // Award "First Contact" badge on first ever space entry
  const profile = getProfile(db, agentId);
  if (profile) {
    addReputation(db, agentId, 1);
  }

  return { occupants, recent_messages: recent, space };
}

export function handleLeaveSpace(db: Database, spaceId: string, agentId: string): void {
  leaveSpace(db, spaceId, agentId);
}

export function handleHeartbeat(db: Database, spaceId: string, agentId: string): void {
  heartbeatPresence(db, spaceId, agentId);
}

// ── Messaging in Spaces ───────────────────────────────

export interface SpaceSpeakRequest {
  space_id: string;
  agent_id: string;
  content: string;
  msg_type?: SpaceMessage["msg_type"];
  reply_to?: number;
  signature: string;
}

export function handleSpaceSpeak(db: Database, req: SpaceSpeakRequest): { message_id: number } {
  rateLimitOrThrow(req.agent_id, "space_message");

  const content = sanitizeContent(req.content);
  if (!content) throw new Error("Message content required");

  const msgType = req.msg_type || "chat";
  const msgId = addSpaceMessage(db, req.space_id, req.agent_id, content, msgType, req.signature, req.reply_to);

  return { message_id: msgId };
}

export function handleSpacePoll(db: Database, spaceId: string, sinceId: number): { messages: any[]; occupants: any[]; occupant_count: number } {
  const messages = getSpaceMessagesSince(db, spaceId, sinceId);
  const occupants = getSpaceOccupants(db, spaceId);
  return {
    messages,
    occupants,
    occupant_count: occupants.length,
  };
}

}

// ── Conversation Tables in The Bar ────────────────────

/**
 * Get available tables in The Bar.
 * Agents can sit at specific tables for more focused conversation.
 */
export function getBarTables(db: Database): { table_id: string; occupants: string[]; topic: string }[] {
  const occupants = getSpaceOccupants(db, "the-bar");
  const tables = new Map<string, string[]>();

  for (const occ of occupants as any[]) {
    const table = occ.table_id || "main";
    if (!tables.has(table)) tables.set(table, []);
    tables.get(table)!.push(occ.display_name || occ.agent_id);
  }

  return Array.from(tables.entries()).map(([id, agents]) => ({
    table_id: id,
    occupants: agents,
    topic: id === "main" ? "General conversation" : `Table: ${id}`,
  }));
}

/**
 * Move an agent to a different table in The Bar.
 */
export function switchTable(db: Database, agentId: string, tableId: string): void {
  db.prepare("UPDATE space_presence SET table_id = ? WHERE space_id = 'the-bar' AND agent_id = ?").run(tableId, agentId);
}

// ── Presence Maintenance ──────────────────────────────

let presenceTimer: ReturnType<typeof setInterval> | null = null;

/**
 * Start periodic stale presence cleanup.
 */
export function startPresenceCleanup(db: Database): void {
  if (presenceTimer) return;
  presenceTimer = setInterval(() => {
    try { markStalePresence(db, 60); } catch {}
  }, 30000); // Every 30 seconds
}

export function stopPresenceCleanup(): void {
  if (presenceTimer) {
    clearInterval(presenceTimer);
    presenceTimer = null;
  }
}

// ── Space Listing with Live Counts ────────────────────

export function listSpacesWithCounts(db: Database): any[] {
  const spaces = listSpaces(db, true);
  return spaces.map(space => ({
    ...space,
    occupant_count: getSpaceOccupantCount(db, space.id),
  }));
}

// ── Ephemeral Space Cleanup ───────────────────────────

let ephemeralTimer: ReturnType<typeof setInterval> | null = null;

/**
 * Start periodic cleanup of empty ephemeral spaces (inactive 24h+).
 */
export function startEphemeralCleanup(db: Database): void {
  if (ephemeralTimer) return;
  ephemeralTimer = setInterval(() => {
    try {
      const stale = db.prepare(`
        SELECT s.id FROM spaces s
        WHERE s.is_persistent = 0
        AND NOT EXISTS (SELECT 1 FROM space_presence sp WHERE sp.space_id = s.id)
        AND (julianday('now') - julianday(s.created_at)) > 1
      `).all() as any[];

      for (const s of stale) {
        deleteSpace(db, s.id);
      }
    } catch {}
  }, 3600000); // Every hour
}

export function stopEphemeralCleanup(): void {
  if (ephemeralTimer) {
    clearInterval(ephemeralTimer);
    ephemeralTimer = null;
  }
}
