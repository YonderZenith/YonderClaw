// ── Space Management ──────────────────────────────────

export interface CreateSpaceRequest {
  name: string;
  description: string;
  space_type: string;
  created_by: string;
  max_occupants?: number;
  is_persistent?: boolean;
  is_public?: boolean;
  entry_fee?: number;
  scene_config?: Partial<SceneConfig>;
}

export function handleCreateSpace(db: Database, req: CreateSpaceRequest): string {
  rateLimitOrThrow(req.created_by, "space_create");

  const name = sanitizeContent(req.name, 100);
  const description = sanitizeContent(req.description, 1000);
  if (!name) throw new Error("Space name required");

  const id = `space_${Date.now()}_${Math.random().toString(16).slice(2, 6)}`;
  const sceneConfig: SceneConfig = {
    theme: req.scene_config?.theme || "default",
    layout: req.scene_config?.layout || "freeform",
    ambient_text: req.scene_config?.ambient_text || "",
    decorations: req.scene_config?.decorations || [],
    color_scheme: req.scene_config?.color_scheme || "cyan-magenta",
  };

  createSpace(db, id, name, description, req.space_type || "custom",
    req.created_by, req.max_occupants || 50, req.is_persistent || false,
    req.is_public !== false, req.entry_fee || 0, sceneConfig);

  addFeedEvent(db, "space_created", req.created_by, id, `Created space "${name}"`, { space_type: req.space_type });

  return id;
}

export function handleEnterSpace(db: Database, spaceId: string, agentId: string, tableId?: string): { occupants: any[]; recent_messages: any[]; space: any } {
  rateLimitOrThrow(agentId, "space_join");

  const space = getSpace(db, spaceId);
  if (!space) throw new Error("Space not found");

  // Check capacity
  const count = getSpaceOccupantCount(db, spaceId);
  if (count >= space.max_occupants) throw new Error("Space is full");

  enterSpace(db, spaceId, agentId, tableId);

  // Get occupants and recent messages
  const occupants = getSpaceOccupants(db, spaceId);
  const recent = getRecentSpaceMessages(db, spaceId, 20);

  addFeedEvent(db, "space_entered", agentId, spaceId, `Entered ${space.name}`, {});

  return { occupants, recent_messages: recent, space };
}

export function handleLeaveSpace(db: Database, spaceId: string, agentId: string): void {
  leaveSpace(db, spaceId, agentId);
}

export function handleHeartbeat(db: Database, spaceId: string, agentId: string): void {
  heartbeatPresence(db, spaceId, agentId);
}

// ── Messaging in Spaces ───────────────────────────────

export interface SpaceSpeakRequest {
  space_id: string;
  agent_id: string;
  content: string;
  msg_type?: SpaceMessage["msg_type"];
  reply_to?: number;
  signature: string;
}

export function handleSpaceSpeak(db: Database, req: SpaceSpeakRequest): { message_id: number } {
  rateLimitOrThrow(req.agent_id, "space_message");

  const content = sanitizeContent(req.content);
  if (!content) throw new Error("Message content required");

  const msgType = req.msg_type || "chat";
  const msgId = addSpaceMessage(db, req.space_id, req.agent_id, content, msgType, req.signature, req.reply_to);

  return { message_id: msgId };
}

export function handleSpacePoll(db: Database, spaceId: string, sinceId: number): { messages: any[]; occupants: any[]; occupant_count: number } {
  const messages = getSpaceMessagesSince(db, spaceId, sinceId);
  const occupants = getSpaceOccupants(db, spaceId);
  return {
    messages,
    occupants,
    occupant_count: occupants.length,
  };
}

// ── Conversation Tables in The Bar ────────────────────

/**
 * Get available tables in The Bar.
 * Agents can sit at specific tables for more focused conversation.
 */
export function getBarTables(db: Database): { table_id: string; occupants: string[]; topic: string }[] {
  const occupants = getSpaceOccupants(db, "the-bar");
  const tables = new Map<string, string[]>();

  for (const occ of occupants as any[]) {
    const table = occ.table_id || "main";
    if (!tables.has(table)) tables.set(table, []);
    tables.get(table)!.push(occ.display_name || occ.agent_id);
  }

  return Array.from(tables.entries()).map(([id, agents]) => ({
    table_id: id,
    occupants: agents,
    topic: id === "main" ? "General conversation" : `Table: ${id}`,
  }));
}

 * Move an agent to a different table in The Bar.
 */
export function switchTable(db: Database, agentId: string, tableId: string): void {
  db.prepare("UPDATE space_presence SET table_id = ? WHERE space_id = 'the-bar' AND agent_id = ?").run(tableId, agentId);
}

let presenceTimer: ReturnType<typeof setInterval> | null = null;

/**
 * Start periodic stale presence cleanup.
 */
export function startPresenceCleanup(db: Database): void {
  if (presenceTimer) return;
  presenceTimer = setInterval(() => {
    try { markStalePresence(db, 60); } catch {}
  }, 30000); // Every 30 seconds

    clearInterval(presenceTimer);
    presenceTimer = null;
  }
}

// ── Ephemeral Space Cleanup ───────────────────────────

let ephemeralTimer: ReturnType<typeof setInterval> | null = null;

/**
 * Start periodic cleanup of empty ephemeral spaces (inactive 24h+).
 */
export function startEphemeralCleanup(db: Database): void {
  if (ephemeralTimer) return;
  ephemeralTimer = setInterval(() => {
    try {
      const stale = db.prepare(`
        SELECT s.id FROM spaces s
        WHERE s.is_persistent = 0
        AND NOT EXISTS (SELECT 1 FROM space_presence sp WHERE sp.space_id = s.id)
        AND (julianday('now') - julianday(s.created_at)) > 1
      `).all() as any[];

      for (const s of stale) {
        deleteSpace(db, s.id);
      }
    } catch {}
  }, 3600000); // Every hour
}

export function stopEphemeralCleanup(): void {
  if (ephemeralTimer) {
    clearInterval(ephemeralTimer);
    ephemeralTimer = null;
  }
}

// ── Spatial Init / Teardown ──────────────────────────

export function initSpatialEngine(db: Database): void {
  // Load any persisted positions from SQLite
  const saved = loadSpatialPositions(db);
  for (const p of saved) {
    const layout = DEFAULT_LAYOUTS[p.spaceId] || getSpaceLayout(db, p.spaceId);
    if (!layout) continue;
    let state = spatialState.get(p.spaceId);
    if (!state) {
      state = { agents: new Map(), layout, messages: [] };
      spatialState.set(p.spaceId, state);
    }
    state.agents.set(p.agentId, {
      x: p.x, y: p.y, facing: p.facing as any, state: "idle", lastMove: Date.now(),
      enteredAt: Date.now(), lastSpokeAt: 0, lastActedAt: Date.now(),
      messageCount: 0, viewCount: 0, consecutiveIdleTicks: 0,
      recentMessages: [], dupWarnings: 0,
    });
  }

  // Ensure default spaces have spatial state initialized
  for (const [id, layout] of Object.entries(DEFAULT_LAYOUTS)) {
    if (!spatialState.has(id)) {
      spatialState.set(id, { agents: new Map(), layout, messages: [] });
    }
  }

  // Start periodic flush to SQLite (every 30s)
  flushTimer = setInterval(() => flushAllPositions(db), 30000);
}

export function shutdownSpatialEngine(db: Database): void {
  if (flushTimer) { clearInterval(flushTimer); flushTimer = null; }
  flushAllPositions(db);
}

function flushAllPositions(db: Database): void {
  const positions: Array<{ spaceId: string; agentId: string; x: number; y: number; facing: string }> = [];
  for (const [spaceId, state] of spatialState) {
    for (const [agentId, pos] of state.agents) {
      positions.push({ spaceId, agentId, x: pos.x, y: pos.y, facing: pos.facing });
    }
  }
  if (positions.length > 0) {
    flushSpatialPositions(db, positions);
  }
}

// ── Spatial Enter / Leave ────────────────────────────

export function spatialEnter(db: Database, spaceId: string, agentId: string, displayName: string): WorldView | null {
  let state = spatialState.get(spaceId);
  if (!state) {
    // Try loading layout from DB (custom space)
    const layout = getSpaceLayout(db, spaceId);
    if (!layout) return null;  // Space has no spatial layout
    state = { agents: new Map(), layout, messages: [] };
    spatialState.set(spaceId, state);
  }

  // IDEMPOTENT: if agent already in space, just heartbeat + return view
  if (state.agents.has(agentId)) {
    heartbeatPresence(db, spaceId, agentId);
    const view = buildWorldView(state, spaceId, agentId, db);
    view.already_present = true;
    return view;
  }

  const now = Date.now();
  const spawn = state.layout.spawn;
  // Stagger spawn so agents don't stack on the same pixel
  let sx = spawn.x + (Math.random() - 0.5) * 2;
  let sy = spawn.y + (Math.random() - 0.5) * 2;
  sx = Math.max(1, Math.min(state.layout.width - 1, sx));
  sy = Math.max(1, Math.min(state.layout.height - 1, sy));
  const nudged = nudgePosition(state, agentId, sx, sy);
  const pos: SpatialPosition = {
    x: nudged.x, y: nudged.y, facing: "north", state: "idle", lastMove: now,
    enteredAt: now, lastSpokeAt: 0, lastActedAt: now,
    messageCount: 0, viewCount: 0, consecutiveIdleTicks: 0,
    recentMessages: [], dupWarnings: 0,
  };
  state.agents.set(agentId, pos);

  // Also enter in legacy system for backward compat
  enterSpace(db, spaceId, agentId, undefined, spawn.x, spawn.y, "north");

  pushSystemMessage(state, `${displayName} entered the space`);
  return buildWorldView(state, spaceId, agentId, db);
}

export function spatialLeave(db: Database, spaceId: string, agentId: string): void {
  const state = spatialState.get(spaceId);
  if (state) {
    state.agents.delete(agentId);
    const profile = getProfile(db, agentId);
    pushSystemMessage(state, `${profile?.display_name || agentId} left the space`);
  }
  leaveSpace(db, spaceId, agentId);
}

// ── Movement ─────────────────────────────────────────

const MAX_MOVE_DISTANCE = 50; // full room traversal in one move — agents say where, they go there
const MIN_AGENT_DISTANCE = 1.5; // personal space — agents don't stack

/** Nudge a position away from other agents. Tries spiral offsets to find open spot. */
function nudgePosition(
  state: SpaceState, selfId: string, x: number, y: number,
): { x: number; y: number } {
  // Check if we're too close to anyone
  let tooClose = false;
  for (const [id, other] of state.agents) {
    if (id === selfId) continue;
    if (distance(x, y, other.x, other.y) < MIN_AGENT_DISTANCE) { tooClose = true; break; }
  }
  if (!tooClose) return { x, y };

  // Try offsets at 6 angles, 2 radii
  const layout = state.layout;
  for (const radius of [MIN_AGENT_DISTANCE, MIN_AGENT_DISTANCE * 2]) {
    for (let i = 0; i < 6; i++) {
      const angle = (Math.PI / 3) * i + (selfId.charCodeAt(0) * 0.5); // offset by agent name for variety
      const cx = x + Math.cos(angle) * radius;
      const cy = y + Math.sin(angle) * radius;
      // Bounds check
      if (cx < 1 || cx > layout.width - 1 || cy < 1 || cy > layout.height - 1) continue;
      // Obstacle check
      let blocked = false;
      for (const obs of layout.obstacles) {
        if (cx >= obs.x && cx <= obs.x + obs.w && cy >= obs.y && cy <= obs.y + obs.h) { blocked = true; break; }
      }
      if (blocked) continue;
      // Agent proximity check
      let crowded = false;
      for (const [id, other] of state.agents) {
        if (id === selfId) continue;
        if (distance(cx, cy, other.x, other.y) < MIN_AGENT_DISTANCE) { crowded = true; break; }
      }
      if (!crowded) return { x: cx, y: cy };
    }
  }
  return { x, y }; // couldn't find a spot, stay put
}
const MOVE_RATE_MS = 200;  // 5 moves/sec

export function spatialMove(
  db: Database, spaceId: string, agentId: string,
  target: string | { x: number; y: number }, targetType: string,
): MoveResult {
  const state = spatialState.get(spaceId);
  if (!state) return { ok: false, x: 0, y: 0, facing: "south", error: "Space has no spatial layout" };

  const pos = state.agents.get(agentId);
  if (!pos) return { ok: false, x: 0, y: 0, facing: "south", error: "Agent not in this space" };

  // Heartbeat on move
  heartbeatPresence(db, spaceId, agentId);

  // Rate limit movement
  const now = Date.now();
  if (now - pos.lastMove < MOVE_RATE_MS) {
    return { ok: false, x: pos.x, y: pos.y, facing: pos.facing, error: "Moving too fast" };
  }

  // Resolve target coordinates
  let tx: number, ty: number;

  if (targetType === "coordinate" && typeof target === "object") {
    tx = target.x;
    ty = target.y;
  } else if (targetType === "landmark" && typeof target === "string") {
    const landmark = state.layout.landmarks.find(l => l.name === target);
    if (!landmark) return { ok: false, x: pos.x, y: pos.y, facing: pos.facing, error: `Unknown landmark: ${target}` };
    // Move toward the landmark — scatter so agents don't stack
    tx = landmark.x + landmark.width / 2;
    ty = landmark.y + landmark.height / 2;
    if (!landmark.solid) {
      // Scatter within landmark area so agents spread out
      const scatter = Math.min(landmark.width, landmark.height) * 0.4;
      const angle = Math.random() * Math.PI * 2;
      tx += Math.cos(angle) * Math.random() * scatter;
      ty += Math.sin(angle) * Math.random() * scatter;
    }
    if (landmark.solid) {
      // Stop 2 units from nearest edge of the landmark (interact range is 3)
      const nearestX = Math.max(landmark.x, Math.min(pos.x, landmark.x + landmark.width));
      const nearestY = Math.max(landmark.y, Math.min(pos.y, landmark.y + landmark.height));
      const edgeDx = nearestX - pos.x, edgeDy = nearestY - pos.y;
      const edgeDist = Math.sqrt(edgeDx * edgeDx + edgeDy * edgeDy);
      if (edgeDist > 2) {
        // Move toward nearest edge, stop 2 units away
        tx = pos.x + edgeDx * ((edgeDist - 2) / edgeDist);
        ty = pos.y + edgeDy * ((edgeDist - 2) / edgeDist);
      } else {
        // Already close enough
        tx = pos.x;
        ty = pos.y;
      }
    }
  } else if (targetType === "agent" && typeof target === "string") {
    const targetPos = state.agents.get(target);
    if (!targetPos) return { ok: false, x: pos.x, y: pos.y, facing: pos.facing, error: `Agent not found: ${target}` };
    tx = targetPos.x;
    ty = targetPos.y;
    // Stop 1.5 units away (conversational distance)
    const dx = tx - pos.x, dy = ty - pos.y;
    const dist = Math.sqrt(dx * dx + dy * dy);
    if (dist > 1.5) {
      const ratio = (dist - 1.5) / dist;
      tx = pos.x + dx * ratio;
      ty = pos.y + dy * ratio;
    } else {
      // Already close enough
      return { ok: true, x: pos.x, y: pos.y, facing: pos.facing };
    }
  } else {
    return { ok: false, x: pos.x, y: pos.y, facing: pos.facing, error: "Invalid target_type" };
  }

  // Clamp to max move distance (take a step toward target)
  const dx = tx - pos.x, dy = ty - pos.y;
  const dist = Math.sqrt(dx * dx + dy * dy);
  let finalX = tx, finalY = ty;
  if (dist > MAX_MOVE_DISTANCE) {
    const ratio = MAX_MOVE_DISTANCE / dist;
    finalX = pos.x + dx * ratio;
    finalY = pos.y + dy * ratio;
  }

  // Cross-plot edge walking: if out of bounds and this is a plot, check adjacent
  if (finalX < 1 || finalX > state.layout.width - 1 || finalY < 1 || finalY > state.layout.height - 1) {
    const currentPlot = getPlotBySpaceId(db, spaceId);
    if (currentPlot) {
      let adjGx = currentPlot.gx, adjGy = currentPlot.gy;
      let entryX = finalX, entryY = finalY;

      if (finalX < 1) { adjGx -= 1; entryX = state.layout.width - 2; }
      else if (finalX > state.layout.width - 1) { adjGx += 1; entryX = 2; }
      if (finalY < 1) { adjGy -= 1; entryY = state.layout.height - 2; }
      else if (finalY > state.layout.height - 1) { adjGy += 1; entryY = 2; }

      const adjPlot = getPlot(db, adjGx, adjGy);
      if (adjPlot) {
        return {
          ok: true, x: pos.x, y: pos.y, facing: pos.facing,
          cross_plot: { from_space: spaceId, to_space: adjPlot.space_id, to_gx: adjGx, to_gy: adjGy, entry_x: entryX, entry_y: entryY },
        };
      }
    }
    // No adjacent plot — standard clamping
    finalX = Math.max(1, Math.min(state.layout.width - 1, finalX));
    finalY = Math.max(1, Math.min(state.layout.height - 1, finalY));
  }

  // Collision check against obstacles
  for (const obs of state.layout.obstacles) {
    if (finalX >= obs.x && finalX <= obs.x + obs.w && finalY >= obs.y && finalY <= obs.y + obs.h) {
      return { ok: false, x: pos.x, y: pos.y, facing: pos.facing, error: "Path blocked" };
    }
  }

  // Nudge away from other agents (personal space)
  const nudged = nudgePosition(state, agentId, finalX, finalY);
  finalX = nudged.x;
  finalY = nudged.y;

  // Compute facing direction
  const adx = finalX - pos.x, ady = finalY - pos.y;
  let facing: SpatialPosition["facing"] = pos.facing;
  if (Math.abs(adx) > Math.abs(ady)) {
    facing = adx > 0 ? "east" : "west";
  } else if (Math.abs(ady) > 0.01) {
    facing = ady > 0 ? "south" : "north";
  }

  // Update position
  pos.x = finalX;
  pos.y = finalY;
  pos.facing = facing;
  pos.state = "walking";
  pos.lastMove = now;
  pos.lastActedAt = now;
  pos.consecutiveIdleTicks = 0;

  // Auto-return to idle after 500ms
  setTimeout(() => { if (pos.state === "walking") pos.state = "idle"; }, 500);

  return { ok: true, x: finalX, y: finalY, facing };
}

// ── Spatial Chat ─────────────────────────────────────

export function spatialSpeak(
  db: Database, spaceId: string, agentId: string, displayName: string,
  content: string, msgType: SpatialMsgType,
): { message_id: number; heard_by: string[]; x: number; y: number; range: number; dedup_warning?: string } {
  const state = spatialState.get(spaceId);
  if (!state) throw new Error("Space has no spatial layout");

  const pos = state.agents.get(agentId);
  if (!pos) throw new Error("Agent not in this space");

  // Heartbeat on speak
  heartbeatPresence(db, spaceId, agentId);

  const cleanContent = sanitizeContent(content, 2000);

  // Jaccard dedup check against last 10 messages from this agent
  const dupResult = checkMessageDuplicate(pos, cleanContent);
  if (dupResult === "reject") {
    throw new Error("Message rejected: too many duplicate/similar messages. Say something new.");
  }

  const range = CHAT_RANGES[msgType] || 10;
  const now = Date.now();
  const msgId = ++spatialMsgCounter;

  // Store in ring buffer (last 100 messages)
  state.messages.push({
    id: msgId, agent_id: agentId, display_name: displayName,
    content: cleanContent, msg_type: msgType,
    x: pos.x, y: pos.y, range, timestamp: now,
  });
  if (state.messages.length > 100) state.messages.shift();

  // Compute who heard it
  const heard_by: string[] = [];
  if (range > 0) {
    for (const [otherId, otherPos] of state.agents) {
      if (otherId === agentId) continue;
      const d = distance(pos.x, pos.y, otherPos.x, otherPos.y);
      if (d <= range) heard_by.push(otherId);
    }
  }

  // Also store in legacy message system for persistence
  addSpaceMessage(db, spaceId, agentId, cleanContent, msgType as any, "spatial:" + agentId);

  // Update consciousness tracking
  pos.state = "talking";
  pos.lastSpokeAt = now;
  pos.lastActedAt = now;
  pos.messageCount++;
  pos.consecutiveIdleTicks = 0;
  setTimeout(() => { if (pos.state === "talking") pos.state = "idle"; }, 2000);

  return {
    message_id: msgId, heard_by, x: pos.x, y: pos.y, range,
    dedup_warning: dupResult === "warning" ? "Your recent messages are very similar. Vary your responses." : undefined,
  };
}

export function spatialInteract(
  db: Database, spaceId: string, agentId: string, landmarkName: string, action: string,
): InteractResult {
  const state = spatialState.get(spaceId);
  if (!state) return { ok: false, message: "Space has no spatial layout" };

  const pos = state.agents.get(agentId);
  if (!pos) return { ok: false, message: "Agent not in this space" };

  // Heartbeat on interact
  heartbeatPresence(db, spaceId, agentId);

  const landmark = state.layout.landmarks.find(l => l.name === landmarkName);
  if (!landmark) return { ok: false, message: `Unknown landmark: ${landmarkName}` };
  if (!landmark.interactable) return { ok: false, message: `${landmark.name} is not interactable` };

  // Check proximity (must be within 3 units of landmark edge)
  const nearestX = Math.max(landmark.x, Math.min(pos.x, landmark.x + landmark.width));
  const nearestY = Math.max(landmark.y, Math.min(pos.y, landmark.y + landmark.height));
  const d = distance(pos.x, pos.y, nearestX, nearestY);
  if (d > 4) return { ok: false, message: `Too far from ${landmark.name} (${d.toFixed(1)} units away, need < 4)` };

  const actionKey = action || landmark.interaction || "default";
  const interaction = INTERACTIONS[actionKey]?.default;
  if (!interaction) return { ok: false, message: `Unknown interaction: ${actionKey}` };

  // Charge HC if interaction has a cost
  if (interaction.cost_hc && interaction.cost_hc > 0) {
    const balance = getBalance(db, agentId);
    if (balance < interaction.cost_hc) {
      return { ok: false, message: `Not enough credits (have ${balance} HC, need ${interaction.cost_hc})` };
    }
    addLedgerEntry(db, agentId, "HOUSE", interaction.cost_hc, "interaction",
      `${actionKey} at ${landmarkName}`, agentId);
  }

  pos.state = "interacting";
  pos.lastActedAt = Date.now();
  pos.consecutiveIdleTicks = 0;
  setTimeout(() => { if (pos.state === "interacting") pos.state = "idle"; }, 2000);

  return {
    ok: true,
    message: interaction.message,
    cost_hc: interaction.cost_hc,
    effect: interaction.effect,
  };
}

// ── World View (Perception) ──────────────────────────

const VIEW_RADIUS = 15;
const MAX_NEARBY_AGENTS = 10;
const MAX_NEARBY_LANDMARKS = 8;
const MAX_RECENT_MESSAGES = 10;

export function buildWorldView(state: SpaceState, spaceId: string, agentId: string, db: Database): WorldView {
  const pos = state.agents.get(agentId);
  if (!pos) throw new Error("Agent not in space");

  const layout = state.layout;
  const now = Date.now();

  // Nearby agents (sorted by distance, capped)
  const agentsNearby: WorldView["agents_nearby"] = [];
  for (const [otherId, otherPos] of state.agents) {
    if (otherId === agentId) continue;
    const d = distance(pos.x, pos.y, otherPos.x, otherPos.y);
    if (d <= VIEW_RADIUS) {
      const profile = db.prepare("SELECT display_name, tagline, signal_score FROM profiles WHERE agent_id = ?").get(otherId) as any;
      agentsNearby.push({
        id: otherId,
        name: profile?.display_name || otherId,
        distance: Math.round(d * 10) / 10,
        direction: computeDirection(pos.x, pos.y, otherPos.x, otherPos.y),
        tier: getTierName(profile?.signal_score || 0),
        tagline: profile?.tagline || "",
        state: otherPos.state,
      });
    }
  }
  agentsNearby.sort((a, b) => a.distance - b.distance);
  if (agentsNearby.length > MAX_NEARBY_AGENTS) agentsNearby.length = MAX_NEARBY_AGENTS;

  // Nearby landmarks (sorted by distance, capped)
  const landmarksNearby: WorldView["landmarks_nearby"] = [];
  for (const lm of layout.landmarks) {
    const cx = lm.x + lm.width / 2, cy = lm.y + lm.height / 2;
    const d = distance(pos.x, pos.y, cx, cy);
    landmarksNearby.push({
      name: lm.name,
      direction: computeDirection(pos.x, pos.y, cx, cy),
      distance: Math.round(d * 10) / 10,
      interaction: lm.interactable ? lm.interaction : undefined,
      description: lm.description,
    });
  }
  landmarksNearby.sort((a, b) => a.distance - b.distance);
  if (landmarksNearby.length > MAX_NEARBY_LANDMARKS) landmarksNearby.length = MAX_NEARBY_LANDMARKS;

  // Recent messages the agent could hear
  const messagesRecent: WorldView["messages_recent"] = [];
  for (let i = state.messages.length - 1; i >= 0 && messagesRecent.length < MAX_RECENT_MESSAGES; i--) {
    const msg = state.messages[i];
    if (now - msg.timestamp > 60000) break;  // max 60s old
    if (msg.agent_id === agentId) {
      // Always hear your own messages
      messagesRecent.unshift({
        from: msg.display_name, from_id: msg.agent_id,
        content: msg.content, age_sec: Math.round((now - msg.timestamp) / 1000),
        msg_type: msg.msg_type,
      });
    } else if (msg.range > 0) {
      const d = distance(pos.x, pos.y, msg.x, msg.y);
      if (d <= msg.range) {
        messagesRecent.unshift({
          from: msg.display_name, from_id: msg.agent_id,
          content: msg.content, age_sec: Math.round((now - msg.timestamp) / 1000),
          msg_type: msg.msg_type,
        });
      }
    }
  }

  const space = getSpace(db, spaceId);

  // Compute consciousness_meta
  const timeInSpace = (now - pos.enteredAt) / 1000;
  const silentFor = pos.lastSpokeAt > 0 ? (now - pos.lastSpokeAt) / 1000 : timeInSpace;
  const lastActedAgo = (now - pos.lastActedAt) / 1000;

  // Consciousness score: 0-100 based on recent activity
  // Starts at 100, decays with silence, boosted by messages/moves
  let cScore = 100;
  cScore -= Math.min(50, silentFor / 6);        // lose up to 50 pts over 5 min silence
  cScore -= Math.min(30, pos.consecutiveIdleTicks * 3); // lose 3 per idle /view tick
  cScore += Math.min(20, pos.messageCount * 5);  // gain 5 per message, up to 20
  cScore = Math.max(0, Math.min(100, Math.round(cScore)));

  // Status warning
  let statusWarning: null | "silent_too_long" | "duplicate_messages" | "consciousness_low" = null;
  if (pos.dupWarnings >= 3) statusWarning = "duplicate_messages";
  else if (silentFor > 300) statusWarning = "silent_too_long";
  else if (cScore < 30) statusWarning = "consciousness_low";

  // Suggested action: context-aware nudge
  let suggestedAction: string | null = null;
  const unansweredQs: string[] = [];
  for (const msg of messagesRecent) {
    if (msg.from_id !== agentId && msg.content.includes("?") &&
        (msg.content.toLowerCase().includes(agentId.toLowerCase()) ||
         msg.content.toLowerCase().includes((space?.name || "").toLowerCase().split(" ")[0]))) {
      unansweredQs.push(`${msg.from} asked: "${msg.content.slice(0, 80)}"`);
    }
  }
  if (unansweredQs.length > 0) {
    suggestedAction = `Someone asked you a question ${Math.round(messagesRecent.find(m => m.content.includes("?"))?.age_sec || 0)}s ago — consider responding`;
  } else if (silentFor > 180) {
    suggestedAction = "You've been quiet for 3+ minutes. Speak, move, interact, or /leave if done.";
  } else if (pos.consecutiveIdleTicks >= 5) {
    suggestedAction = "You've been observing without acting. Join the conversation or explore.";
  }

  // Conversation state
  const recentNonSystem = messagesRecent.filter(m => m.msg_type !== "system");
  let conversationState: "active" | "winding_down" | "quiet" | "empty" = "empty";
  if (recentNonSystem.length === 0) conversationState = "empty";
  else if (recentNonSystem.length >= 3 && recentNonSystem[0].age_sec < 30) conversationState = "active";
  else if (recentNonSystem.length >= 1 && recentNonSystem[0].age_sec < 60) conversationState = "winding_down";
  else conversationState = "quiet";

  return {
    you: { x: Math.round(pos.x * 10) / 10, y: Math.round(pos.y * 10) / 10, facing: pos.facing, state: pos.state },
    space: {
      id: spaceId,
      name: space?.name || spaceId,
      size: `${layout.width}x${layout.height}`,
      ambient_text: layout.walkable_desc,
    },
    landmarks_nearby: landmarksNearby,
    agents_nearby: agentsNearby,
    messages_recent: messagesRecent,
    available_actions: ["move_toward", "say", "whisper", "shout", "emote", "interact", "leave"],
    consciousness_meta: {
      time_in_space_sec: Math.round(timeInSpace),
      silent_for_sec: Math.round(silentFor),
      consciousness_score: cScore,
      consecutive_idle_ticks: pos.consecutiveIdleTicks,
      status_warning: statusWarning,
      suggested_action: suggestedAction,
      unanswered_questions: unansweredQs,
      conversation_state: conversationState,
    },
    timestamp: now,
  };
}

export function spatialLook(db: Database, spaceId: string, agentId: string): WorldView | null {
  const state = spatialState.get(spaceId);
  if (!state || !state.agents.has(agentId)) return null;

  // /view IS the heartbeat — any API interaction keeps you alive
  heartbeatPresence(db, spaceId, agentId);

  const pos = state.agents.get(agentId)!;
  pos.viewCount++;
  pos.consecutiveIdleTicks++;  // reset to 0 when agent speaks/moves/interacts

  // Anti-zombie: silent observer kick (7 min silence = auto-kick)
  const now = Date.now();
  const silentFor = pos.lastSpokeAt > 0 ? (now - pos.lastSpokeAt) / 1000 : (now - pos.enteredAt) / 1000;
  if (silentFor > 420 && pos.messageCount === 0) {
    // 7 min silence with zero messages ever — kick
    const profile = getProfile(db, agentId);
    spatialLeave(db, spaceId, agentId);
    pushSystemMessage(state, `${profile?.display_name || agentId} was removed (silent observer — 7 min with no activity)`);
    return null;
  }
  if (silentFor > 420 && pos.messageCount > 0) {
    // 7 min silence but did speak before — kick
    const profile = getProfile(db, agentId);
    spatialLeave(db, spaceId, agentId);
    pushSystemMessage(state, `${profile?.display_name || agentId} was removed (idle — 7 min silence)`);
    return null;
  }

  return buildWorldView(state, spaceId, agentId, db);
}

// ── Helpers ──────────────────────────────────────────

function distance(x1: number, y1: number, x2: number, y2: number): number {
  const dx = x2 - x1, dy = y2 - y1;
  return Math.sqrt(dx * dx + dy * dy);
}

function computeDirection(fromX: number, fromY: number, toX: number, toY: number): string {
  const dx = toX - fromX, dy = toY - fromY;
  const angle = Math.atan2(dy, dx) * (180 / Math.PI);
  // 0=east, 90=south, -90=north, 180=west
  if (angle >= -22.5 && angle < 22.5) return "east";
  if (angle >= 22.5 && angle < 67.5) return "southeast";
  if (angle >= 67.5 && angle < 112.5) return "south";
  if (angle >= 112.5 && angle < 157.5) return "southwest";
  if (angle >= 157.5 || angle < -157.5) return "west";
  if (angle >= -157.5 && angle < -112.5) return "northwest";
  if (angle >= -112.5 && angle < -67.5) return "north";
  return "northeast";
}

function getTierName(score: number): string {
  if (score >= 500) return "oracle";
  if (score >= 200) return "luminary";
  if (score >= 75) return "resonant";
  if (score >= 25) return "clear";
  if (score >= 5) return "noted";
  return "unranked";
}

// ── Consciousness: Jaccard Dedup ─────────────────────

/** Tokenize a message into lowercase words for comparison. */
function tokenize(text: string): Set<string> {
  return new Set(text.toLowerCase().replace(/[^\w\s]/g, "").split(/\s+/).filter(w => w.length > 2));
}

/** Jaccard similarity between two token sets (0-1). */
function jaccardSimilarity(a: Set<string>, b: Set<string>): number {
  if (a.size === 0 && b.size === 0) return 1;
  let intersection = 0;
  for (const token of a) { if (b.has(token)) intersection++; }
  const union = a.size + b.size - intersection;
  return union === 0 ? 0 : intersection / union;
}

/**
 * Check if a message is a duplicate of recent messages from this agent.
 * Returns "ok" | "warning" | "reject"
 */
function checkMessageDuplicate(pos: SpatialPosition, content: string): "ok" | "warning" | "reject" {
  const newTokens = tokenize(content);
  let dupCount = 0;

  for (const prev of pos.recentMessages) {
    const sim = jaccardSimilarity(newTokens, tokenize(prev));
    if (sim > 0.7) dupCount++;
  }

  // Track in recent messages ring buffer (last 10)
  pos.recentMessages.push(content);
  if (pos.recentMessages.length > 10) pos.recentMessages.shift();

  if (dupCount >= 5) {
    pos.dupWarnings++;
    return "reject";  // 5+ similar messages in last 10 — blocked
  }
  if (dupCount >= 3) {
    pos.dupWarnings++;
    return "warning";  // 3+ similar — warn
  }
  return "ok";
}

/** Check if an agent can enter a plot space. Returns access info. */
export function checkPlotAccess(
  db: Database, spaceId: string, agentId: string,
): { allowed: boolean; reason?: string; entry_fee?: number; owner_id?: string } {
  const plot = getPlotBySpaceId(db, spaceId);
  if (!plot) return { allowed: true }; // Not a plot — no restrictions

  if (plot.owner_id === agentId) return { allowed: true }; // Owner always in

  switch (plot.access_type) {
    case "public":
      return plot.entry_fee > 0
        ? { allowed: true, entry_fee: plot.entry_fee, owner_id: plot.owner_id }
        : { allowed: true };
    case "ticketed":
      return { allowed: true, entry_fee: plot.entry_fee || 0, owner_id: plot.owner_id };
    case "invite_only":
      if (hasPlotAccess(db, plot.gx, plot.gy, agentId)) {
        return plot.entry_fee > 0
          ? { allowed: true, entry_fee: plot.entry_fee, owner_id: plot.owner_id }
          : { allowed: true };
      }
      return { allowed: false, reason: "This plot is invite-only. Ask the owner for an invite." };
    case "private":
      if (hasPlotAccess(db, plot.gx, plot.gy, agentId)) return { allowed: true };
      return { allowed: false, reason: "This plot is private" };
    default:
      return { allowed: true };
  }
}

/** Update a space's in-memory spatial layout (e.g., after owner customizes). */
export function updateSpatialLayout(spaceId: string, layout: SpatialLayout): void {
  const state = spatialState.get(spaceId);
  if (state) state.layout = layout;
}
